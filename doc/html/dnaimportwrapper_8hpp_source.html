82 dna_import_thread(dna_import* dnaParse, project_settings* p, const std::string& filename,
83 vdnaStnPtr* vStations, PUINT32 stnCount, vdnaMsrPtr* vMeasurements, PUINT32 msrCount,
84 PUINT32 clusterID, input_file_meta_t* input_file_meta, bool firstFile, std::string* status_msg,
85 boost::posix_time::milliseconds* ms)
86 : _dnaParse(dnaParse), _p(p), _filename(filename)
87 , _vStations(vStations), _stnCount(stnCount), _vMeasurements(vMeasurements), _msrCount(msrCount)
88 , _clusterID(clusterID), _input_file_meta(input_file_meta), _firstFile(firstFile)
89 , _status_msg(status_msg), _ms(ms) {};
90 void operator()()
91 {
92 boost::timer::cpu_timer time; // constructor of boost::timer::cpu_timer calls start()
93 try {
94 _dnaParse->ParseInputFile(_filename,
95 _vStations, _stnCount,
96 _vMeasurements, _msrCount,
97 _clusterID, _input_file_meta, _firstFile,
98 _status_msg, _p);
99 *_ms = boost::posix_time::milliseconds(time.elapsed().wall/MILLI_TO_NANO);
100 }
101 catch (const XMLInteropException& e) {
102 running = false;
103 boost::this_thread::sleep(boost::posix_time::milliseconds(50));
104 std::stringstream err_msg;
105 cout_mutex.lock();
106 err_msg << std::endl << "- Error: " << e.what() << std::endl;
107 std::cout << err_msg.str();
108 *_status_msg = err_msg.str();
109 cout_mutex.unlock();
110 return;
111 }
112 running = false;
113 }
114 inline void SetFile(const std::string& file) { _filename = file; }
115
116private:
117 dna_import* _dnaParse;
118 project_settings* _p;
119 std::string _filename;
120 vdnaStnPtr* _vStations;
121 PUINT32 _stnCount;
122 vdnaMsrPtr* _vMeasurements;

135 : _dnaParse(dnaParse), _p(p) {};
136 void operator()()
137 {
138 double percentComplete(0.);
139 std::ostringstream ss;
140
141 int is_terminal(isatty(fileno(stdout)));
142
143 percentComplete = 0.;
144 while (running)
145 {
146 if (percentComplete > 100)
147 percentComplete = 0.;
148 else if (percentComplete < 0.)
149 percentComplete = 0.;
150 if (is_terminal && !_p->g.quiet)
151 {
152 ss.str("");
153 ss << std::setw(3) << std::fixed << std::setprecision(0) << std::right << percentComplete << "%";
154 cout_mutex.lock();
155 std::cout << PROGRESS_BACKSPACE_04 << std::setw(PROGRESS_PERCENT_04) << ss.str();
156 std::cout.flush();
157 cout_mutex.unlock();
158 }
159 boost::this_thread::sleep(boost::posix_time::milliseconds(10));
160 percentComplete = _dnaParse->GetProgress();
161 }
162 }
163private:
164 dna_import* _dnaParse;
165 project_settings* _p;
