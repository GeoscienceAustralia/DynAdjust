57void dna_io_seg::load_seg_file_header(const std::string& seg_filename, std::istream& seg_file, UINT32& blockCount,
58 UINT32& blockThreshold, UINT32& minInnerStns)
59{
60
61 std::stringstream ss_err;
62 ss_err << "load_seg_file_header(): An error was encountered when reading from " << seg_filename << "." << std::endl;
63
64 char line[PRINT_LINE_LENGTH];
65 memset(line, '\0', PRINT_LINE_LENGTH);
66
67 try {
68 // read header line
69 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
70 seg_file.getline(line, PRINT_LINE_LENGTH); // DYNADJUST SEGME...
71 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
72 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Version
73 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Build
74 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // File created
75 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // File name
76 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
77 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Command line arguments
78 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
79 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Stations file
80 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Measurements file
81 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
82
83 seg_file.getline(line, PRINT_LINE_LENGTH); // Minimum inner stations
84 minInnerStns = strtoul(line+PRINT_VAR_PAD, NULL, 0);
85
86 seg_file.getline(line, PRINT_LINE_LENGTH); // Block size threshold
87 blockThreshold = strtoul(line+PRINT_VAR_PAD, NULL, 0);
88
89 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Starting stations
90 seg_file.getline(line, PRINT_LINE_LENGTH); // ???
91 while (line[0]!= '-')
92 seg_file.getline(line, PRINT_LINE_LENGTH); // ------------------------
93
94 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
95 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // SEGMENTATION SUMMARY
96 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
97
98 // read block count
99 seg_file.getline(line, PRINT_LINE_LENGTH); // No. blocks produced
100 blockCount = strtoul(line+PRINT_VAR_PAD, NULL, 0);
101 }
102 catch (const std::ios_base::failure& f) {
103 ss_err << f.what();
104 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
105 }
106 catch (const std::runtime_error& e) {
107 ss_err << e.what();
108 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
109 }

115void dna_io_seg::load_seg_file(const std::string& seg_filename, UINT32& blockCount,
116 UINT32& blockThreshold, UINT32& minInnerStns,
117 vvUINT32& v_ISL, vvUINT32& v_JSL, vvUINT32& v_CML,
118 bool loadMetrics,
119 pvmsr_t bmsBinaryRecords, pvUINT32 v_measurementCount,
120 pvUINT32 v_unknownsCount, pvUINT32 v_ContiguousNetList,
121 pvUINT32 v_parameterStationCount)
122{
123 std::ifstream seg_file;
124 std::stringstream ss_err;
125 ss_err << "load_seg_file(): An error was encountered when opening " << seg_filename << "." << std::endl;
126
127 try {
128 // open stations seg file. Throws runtime_error on failure.
129 file_opener(seg_file, seg_filename, std::ios::in, ascii, true);
130 }
131 catch (const std::runtime_error& e) {
132 ss_err << e.what();
133 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
134 }
135 catch (...) {
136 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
137 }
138
139 ss_err.str("");
140 ss_err << "load_seg_file(): An error was encountered when reading from " << seg_filename << "." << std::endl;
141
142 UINT32 b, blk, c, i, j, m,
143 blkCount(0), netID(0), jslCount(0), islCount(0), msrCount(0), stnCount(0);
144
145 char line[PRINT_LINE_LENGTH], format_spec_netid[6],
146 format_spec_junct[6], format_spec_inner[6], format_spec_measr[6];
147
148 std::string sBuf, tmp;
149
150 try {
151
152 load_seg_file_header(seg_filename, seg_file, blockCount,
153 blockThreshold, minInnerStns);
154
155 // Resize vectors based upon block count
156 v_ISL.resize(blockCount);
157 v_JSL.resize(blockCount);
158 v_CML.resize(blockCount);
159
160 if (loadMetrics)
161 {
162 v_ContiguousNetList->resize(blockCount);
163 v_measurementCount->resize(blockCount);
164 v_unknownsCount->resize(blockCount);
165 v_parameterStationCount->resize(blockCount);
166 }
167
168 // skip header info
169 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
170 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Block Junction stns Inner stns Measurements Total stns
171
172 snprintf(format_spec_netid, sizeof(format_spec_netid), "%%%d%s", NETID, "lu");
173 snprintf(format_spec_junct, sizeof(format_spec_junct), "%%%d%s", JUNCT, "lu");
174 snprintf(format_spec_inner, sizeof(format_spec_inner), "%%%d%s", INNER, "lu");
175 snprintf(format_spec_measr, sizeof(format_spec_measr), "%%%d%s", MEASR, "lu");
176
177
178 UINT32 t;
179 UINT16 column;
180 // read block sizes
181 for (t=0; t<blockCount; ++t)
182 {
183 column = 0;
184 getline(seg_file, sBuf);
185
186 if (sBuf.compare(0, 20, "--------------------") == 0)
187 throw boost::enable_current_exception(std::runtime_error(" Segmentation file is corrupt."));
188
189 // Block number
190 try {
191 tmp = trimstr(sBuf.substr(column, BLOCK));
192 if (tmp.empty())
193 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Block number."));
194 blkCount = LongFromString<UINT32>(tmp);
195 }
196 catch (...) {
197 ss_err.str("");
198 ss_err << " Segmentation file is corrupt: Could not extract Block number from the record: " << std::endl << " " << sBuf << std::endl;
199 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
200 }
201
202 column += BLOCK;
203
204 // Network ID
205 try {
206 tmp = trimstr(sBuf.substr(column, NETID));
207 if (tmp.empty())
208 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Network ID."));
209 netID = LongFromString<UINT32>(tmp);
210 }
211 catch (...) {
212 ss_err.str("");
213 ss_err << " Segmentation file is corrupt: Could not extract Network ID from the record: " << std::endl << " " << sBuf << std::endl;
214 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
215 }
216
217 column += NETID;
218
219 // Junction station count
220 try {
221 tmp = trimstr(sBuf.substr(column, JUNCT));
222 if (tmp.empty())
223 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Junction station count."));
224 jslCount = LongFromString<UINT32>(tmp);
225 }
226 catch (...) {
227 ss_err.str("");
228 ss_err << " Segmentation file is corrupt: Could not extract Junction station count from the record: " << std::endl << " " << sBuf << std::endl;
229 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
230 }
231
232 column += JUNCT;
233
234 // Inner station count
235 try {
236 tmp = trimstr(sBuf.substr(column, INNER));
237 if (tmp.empty())
238 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Inner station count."));
239 islCount = LongFromString<UINT32>(tmp);
240 }
241 catch (...) {
242 ss_err.str("");
243 ss_err << " Segmentation file is corrupt: Could not extract Inner station count from the record: " << std::endl << " " << sBuf << std::endl;
244 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
245 }
246
247 column += INNER;
248
249 // Measurement count
250 try {
251 tmp = trimstr(sBuf.substr(column, MEASR));
252 if (tmp.empty())
253 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Measurement count."));
254 msrCount = LongFromString<UINT32>(tmp);
255 }
256 catch (...) {
257 ss_err.str("");
258 ss_err << " Segmentation file is corrupt: Could not extract Measurement count from the record: " << std::endl << " " << sBuf << std::endl;
259 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
260 }
261
262 column += MEASR;
263
264 // Total station count
265 try {
266 tmp = trimstr(sBuf.substr(column));
267 if (tmp.empty())
268 throw boost::enable_current_exception(std::runtime_error(" Unable to retrieve Total station count."));
269 stnCount = LongFromString<UINT32>(tmp);
270 }
271 catch (...) {
272 ss_err.str("");
273 ss_err << " Segmentation file is corrupt: Could not extract Total station count from the record: " << std::endl << " " << sBuf << std::endl;
274 throw boost::enable_current_exception(std::runtime_error(ss_err.str().c_str()));
275 }
276
277 if (stnCount != islCount + jslCount)
278 throw boost::enable_current_exception(std::runtime_error(" Segmentation file is corrupt."));
279
280 v_JSL.at(t) = vUINT32(jslCount);
281 v_ISL.at(t) = vUINT32(islCount);
282 v_CML.at(t) = vUINT32(msrCount);
283
284 if (loadMetrics)
285 {
286 v_ContiguousNetList->at(t) = netID;
287 v_measurementCount->at(t) = v_unknownsCount->at(t) = 0;
288 v_parameterStationCount->at(t) = stnCount;
289 }
290 }
291
292 if (blkCount != blockCount)
293 throw boost::enable_current_exception(std::runtime_error("load_seg_file(): Segmentation file is corrupt."));
294
295 // skip header info
296 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
297 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
298 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Individual block data (ASL and AML indices):
299 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
300
301 // initialise block ID
302 netID = 999999;
303
304 // read block data
305 for (b=0; b<blockCount; ++b)
306 {
307 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
308 seg_file.getline(line, PRINT_LINE_LENGTH); // Block #
309 sscanf(line+5, "%ud", &blk);
310 if (b+1 != blk)
311 throw boost::enable_current_exception(std::runtime_error("load_seg_file(): segmentation file is corrupt."));
312
313 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
314 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Junction stns:
315 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Inner stns:
316 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Measurements:
317 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Total stns:
318 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); //
319 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // Inners Junctions Measurements
320 seg_file.ignore(PRINT_LINE_LENGTH, '\n'); // ------------------------
321
322 seg_file.getline(line, PRINT_LINE_LENGTH); // Ok, now the data!
323
324 jslCount = static_cast<UINT32>(v_JSL.at(b).size());
325 islCount = static_cast<UINT32>(v_ISL.at(b).size());
326 msrCount = static_cast<UINT32>(v_CML.at(b).size());
327
328 c = 0;
329 while (strncmp(line, "--------------------", 20) != 0)
330 {
331 if (c < islCount)
332 {
333 sscanf(line, format_spec_inner, &i);
334 v_ISL.at(b).at(c) = i;
335 if (loadMetrics)
336 v_unknownsCount->at(b) += 3;
337 }
338 if (c < jslCount)
339 {
340 sscanf(line+12, format_spec_junct, &j);
341 v_JSL.at(b).at(c) = j;
342 if (loadMetrics)
343 v_unknownsCount->at(b) += 3;
344 }
345 if (c < msrCount)
346 {
347 sscanf(line+24, format_spec_measr, &m);
348 v_CML.at(b).at(c) = m;
349
350 if (loadMetrics)
351 {
352 // Calculate number of 'measurements' in this measurement
353 // No need to check for ignored measurement since the segmentation
354 // algorithm only includes measurements with the ignore flag cleared
355 switch (bmsBinaryRecords->at(m).measType)
356 {
357 case 'G': // GPS Baseline
358 v_measurementCount->at(b) += 3; // three measurements (x, y, z)
359 break;
360 case 'X': // GPS Baseline cluster
361 case 'Y': // GPS point cluster
362 v_measurementCount->at(b) += bmsBinaryRecords->at(m).vectorCount1 * 3;
363 break;
364 case 'D': // Direction set
365 v_measurementCount->at(b) += bmsBinaryRecords->at(m).vectorCount2 - 1;
366 break;
367 case 'A': // Horizontal angle
368 case 'B': // Geodetic azimuth
369 case 'C': // Chord dist
370 case 'E': // Ellipsoid arc
371 case 'H': // Orthometric height
372 case 'I': // Astronomic latitude
373 case 'J': // Astronomic longitude
374 case 'K': // Astronomic azimuth
375 case 'L': // Level difference
376 case 'P': // Geodetic latitude
377 case 'Q': // Geodetic longitude
378 case 'R': // Ellipsoidal height
379 case 'M': // MSL arc
380 case 'S': // Slope distance
381 case 'V': // Zenith distance
382 case 'Z': // Vertical angle
383 default:
384 v_measurementCount->at(b)++; // single measurement quantity
385 break;
386 }
387 }
388 }
389 c++;
390 seg_file.getline(line, PRINT_LINE_LENGTH); // get more data
391 }
392 }
393 }
394 catch (const std::ios_base::failure& f) {
395 ss_err << f.what();
396 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
397 }
398 catch (const std::runtime_error& e) {
399 ss_err << e.what();
400 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
401 }
402 catch (...) {
403 throw boost::enable_current_exception(std::runtime_error(ss_err.str()));
404 }
405
406 seg_file.close();
407
408}

410void dna_io_seg::build_free_stn_availability(vASL& assocStnList, v_freestn_pair& freeStnList)
411{
412 freeStnList.clear();
413
414 // Resize free station availability list. Note - the
415 // vfreeStnAvailability_ constructor sets all stations to 'free'
416 freeStnList.resize(assocStnList.size());
417
418 UINT32 stn_id(0);
419 for_each(
420 freeStnList.begin(),
421 freeStnList.end(),
422 [&stn_id, &assocStnList](freestn_pair& stn){
423 // consume invalid stations
424 if (assocStnList.at(stn_id).IsInvalid())
425 stn.consume();
426 // Set ID
427 stn.stn_index = stn_id++;
428 }
429 );
430}

432void dna_io_seg::create_stn_appearance_list(vv_stn_appear& stnAppearance,
433 const vvUINT32& paramStationList,
434 vASL& assocStnList)
435{
436 // build free station list
437 v_freestn_pair freeStnList;
438 build_free_stn_availability(assocStnList, freeStnList);
439
440 // Make a copy for the reverse pass
441 v_freestn_pair revStnList(freeStnList);
442
443 UINT32 block, blockCount(static_cast<UINT32>(stnAppearance.size()));
444
445 it_vstn_appear _it_appear;
446 it_vUINT32_const _it_const;
447
448 // 1. Create the station 'appearances' map for the forward pass
449 for (block=0; block<blockCount; ++block)
450 {
451 for (_it_const=paramStationList.at(block).begin(),
452 _it_appear=stnAppearance.at(block).begin();
453 _it_appear!=stnAppearance.at(block).end();
454 ++_it_const, ++_it_appear)
455 {
456 // assign id
457 _it_appear->set_id(*_it_const);
458
459 if (freeStnList.at(_it_appear->station_id).isfree())
460 {
461 // assign first appearance flag and consume free station
462 _it_appear->first_fwd();
463
464 freeStnList.at(_it_appear->station_id).consume();
465 }
466 }
467 }
468
469 UINT32 blockR(blockCount- 1);
470
471 // 2. Create the station 'appearances' map for the reverse pass
472 for (block=0; block<blockCount; ++block, --blockR)
473 {
474 for (_it_appear=stnAppearance.at(blockR).begin();
475 _it_appear!=stnAppearance.at(blockR).end();
476 ++_it_appear)
477 {
478 if (revStnList.at(_it_appear->station_id).isfree())
479 {
480 // assign first appearance flag and consume free station
481 _it_appear->first_rev();
482
483 revStnList.at(_it_appear->station_id).consume();
484 }
485 }
486 }
487}

489void dna_io_seg::write_seg_block(std::ostream &os,
490 const vUINT32& vISL, const vUINT32& vJSL, const vUINT32& vCML,
491 const UINT32& currentBlock,
492 const vstn_t* bstBinaryRecords, const vmsr_t* bmsBinaryRecords,
493 bool PRINT_NAMES)
494{
495 char dash;
496
497 os << std::endl << "Block " << currentBlock << std::endl;
498 for (dash=INNER+JUNCT+MEASR+PAD; dash>0; dash--)
499 os << "-";
500 os << std::endl;
501
502 //os << std::setw(JUNCT) << std::left << "Block:" << std::setw(BLOCK) << std::right << currentBlock_ << std::endl;
503 os << std::setw(JUNCT) << std::left << "Junction stns:" << std::setw(BLOCK) << vJSL.size() << std::endl;
504 os << std::setw(JUNCT) << std::left << "Inner stns:" << std::setw(BLOCK) << vISL.size() << std::endl;
505 os << std::setw(JUNCT) << std::left << "Measurements:" << std::setw(BLOCK) << vCML.size() << std::endl;
506 os << std::setw(JUNCT) << std::left << "Total stns:" << std::setw(BLOCK) << (vJSL.size() + vISL.size()) << std::endl << std::endl;
507
508 it_vUINT32_const _it_isl(vISL.begin());
509 it_vUINT32_const _it_jsl(vJSL.begin());
510 it_vUINT32_const _it_msr(vCML.begin());
511
512 os << std::setw(INNER) << std::left << "Inner stns" << std::setw(JUNCT) << std::left << "Junction stns" <<
513 std::setw(MEASR) << "Measurements" << std::setw(PAD) << "Type" << std::endl;
514
515 for (dash=INNER+JUNCT+MEASR+PAD; dash>0; dash--)
516 os << "-";
517 os << std::endl;
518 std::ostringstream tmp;
519
520 vUINT32 msrStations;
521
522 while (1)
523 {
524 tmp.str("");
525 if (_it_isl != vISL.end())
526 {
527 if (PRINT_NAMES) {
528 os << std::setw(INNER) << std::left << ((ostringstream&)(tmp << bstBinaryRecords->at(*_it_isl).stationName)).str();
529 tmp.str("");
530 }
531 else
532 os << std::setw(INNER) << std::left << *_it_isl;
533 _it_isl++;
534 }
535 else
536 os << std::setw(INNER) << std::left << " ";
537 if (_it_jsl != vJSL.end())
538 {
539 if (PRINT_NAMES) {
540 os << std::setw(JUNCT) << std::left << ((ostringstream&)(tmp << bstBinaryRecords->at(*_it_jsl).stationName)).str();
541 tmp.str("");
542 }
543 else
544 os << std::setw(JUNCT) << std::left << *_it_jsl;
545 _it_jsl++;
546 }
547 else
548 os << std::setw(JUNCT) << std::left << " ";
549 if (_it_msr != vCML.end())
550 {
551 if (PRINT_NAMES) {
552
553 // Get all the stations associated with this measurement
554 GetMsrStations(*bmsBinaryRecords, *_it_msr, msrStations);
555
556 os << std::left << bmsBinaryRecords->at(*_it_msr).measType << " (";
557
558 for_each(msrStations.begin(), msrStations.end(),
559 [&os, &bstBinaryRecords](UINT32 stn){
560 os << bstBinaryRecords->at(stn).stationName << " ";
561 });
562
563 os << ") ";
564
565 if (bmsBinaryRecords->at(*_it_msr).clusterID > 0)
566 os << "cluster " << bmsBinaryRecords->at(*_it_msr).clusterID;
567
568 os << std::endl;
569 }
570 else
571 os << std::setw(MEASR) << std::left << *_it_msr << std::setw(PAD) << std::left << bmsBinaryRecords->at(*_it_msr).measType << std::endl;
572 _it_msr++;
573 }
574 else
575 os << std::endl;
576
577 // continue until all lists are exhausted
578 if (_it_msr==vCML.end() &&
579 _it_jsl==vJSL.end() &&
580 _it_isl==vISL.end())
581 break;
582 }
583
584 for (dash=INNER+JUNCT+MEASR+PAD; dash>0; dash--)
585 os << "-";
586 os << std::endl;
587
588}

590void dna_io_seg::write_seg_file(const std::string& seg_filename, const std::string& bst_filename, const std::string& bms_filename,
591 const UINT32& min_inner_stns, const UINT32& max_block_stns,
592 const std::string& seg_starting_stns, const vstring& vinitialStns,
593 const std::string& command_line_arguments,
594 vvUINT32& v_ISL, vvUINT32& v_JSL, vvUINT32& v_CML,
595 vUINT32& v_ContiguousNetList, const pvstn_t bstBinaryRecords, const pvmsr_t bmsBinaryRecords)
596{
597 std::ofstream seg_file;
598 std::stringstream ss;
599 ss << "write_seg_file(): An error was encountered when opening " << seg_filename << "." << std::endl;
600
601 try {
602 // Create segmentation file. Throws runtime_error on failure.
603 file_opener(seg_file, seg_filename);
604 }
605 catch (const std::runtime_error& e) {
606 ss << e.what();
607 throw boost::enable_current_exception(std::runtime_error(ss.str()));
608 }
609 catch (...) {
610 throw boost::enable_current_exception(std::runtime_error(ss.str()));
611 }
612
613 // Print formatted header
614 print_file_header(seg_file, "DYNADJUST SEGMENTATION OUTPUT FILE");
615
616 seg_file << std::setw(PRINT_VAR_PAD) << std::left << "File name:" << boost::filesystem::system_complete(seg_filename).string() << std::endl << std::endl;
617
