142void StnTally::coutSummary(std::ostream &os, const std::string& title)
143{
144 // Print title
145 os << title << " " << TotalCount() << " stations:" << std::endl;
146 UINT32 i, w(PRINT_VAR_PAD+NUMERIC_WIDTH);
147 // Print line
148 os << " ";
149 for (i=0; i<w; ++i)
150 os << "-";
151 os << std::endl;
152 // Print stations
153 if (CCC)
154 os << std::left << std::setw(PRINT_VAR_PAD) << " (CCC) 3D constrained:" << std::right << std::setw(NUMERIC_WIDTH) << CCC << std::endl;
155 if (FFF)
156 os << std::left << std::setw(PRINT_VAR_PAD) << " (FFF) 3D free:" << std::right << std::setw(NUMERIC_WIDTH) << FFF << std::endl;
157 if (CCF)
158 os << std::left << std::setw(PRINT_VAR_PAD) << " (CCF) 2D constrained, 1D free:" << std::right << std::setw(NUMERIC_WIDTH) << CCF << std::endl;
159 if (FFC)
160 os << std::left << std::setw(PRINT_VAR_PAD) << " (FFC) 2D free, 1D constrained:" << std::right << std::setw(NUMERIC_WIDTH) << FFC << std::endl;
161 if (FCF)
162 os << std::left << std::setw(PRINT_VAR_PAD) << " (FCF) custom 2D constraints:" << std::right << std::setw(NUMERIC_WIDTH) << FCF << std::endl;
163 if (CFF)
164 os << std::left << std::setw(PRINT_VAR_PAD) << " (CFF) custom 2D constraints:" << std::right << std::setw(NUMERIC_WIDTH) << CFF << std::endl;
165 if (CFC)
166 os << std::left << std::setw(PRINT_VAR_PAD) << " (CFC) custom 3D constraints:" << std::right << std::setw(NUMERIC_WIDTH) << CFC << std::endl;
167 os << " ";
168 for (i=0; i<w; ++i)
169 os << "-";
170 os << std::endl;
171 os << std::left << std::setw(PRINT_VAR_PAD) << " Total" << std::right << std::setw(NUMERIC_WIDTH) << TotalCount() << std::endl;
172}
