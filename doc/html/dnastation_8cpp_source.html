266void CDnaStation::SetConstraints(const std::string& sConstraints)
267{
268 // capture string, trim whitespace
269 m_strConstraints = trimstr(sConstraints);
270
271 // No string provided? Fill with FFF
272 if (sConstraints.empty())
273 m_strConstraints = "FFF";
274
275 // Greater than 3 characters? Trim to 3 characters
276 if (m_strConstraints.length() > 3)
277 m_strConstraints = m_strConstraints.substr(0, 3);
278
279 // Less than 3 characters? Pad with 'F'
280 if (m_strConstraints.length() < 3)
281 m_strConstraints.append(size_t(3 - m_strConstraints.length()), 'F');
282
283 m_cLatConstraint = (char)(*m_strConstraints.substr(0, 1).c_str());
284 m_cLonConstraint = (char)(*m_strConstraints.substr(1, 1).c_str());
285 m_cHtConstraint = (char)(*m_strConstraints.substr(2, 1).c_str());
286
287 // Free in all 3 dimensions
288 if (boost::iequals(m_strConstraints, "FFF"))
289 m_constraintType = free_3D;
290 // Constrained in all 3 dimensions
291 else if (boost::iequals(m_strConstraints, "CCC"))
292 m_constraintType = constrained_3D;
293 // Horizontal or 2D adjustment
294 else if (boost::iequals(m_strConstraints, "FFC"))
295 m_constraintType = free_2D;
296 // Vertical or 1D adjustment
297 else if (boost::iequals(m_strConstraints, "CCF"))
298 m_constraintType = free_1D;
299 else
300 m_constraintType = custom_constraint;
301}

360{

362 if (m_strHemisphereZone.empty())
363 return;

370// Reduce from Cartesian or Projection to Geographic
371// m_dXAxis always contains X, Easting and Latitude
372// m_dYAxis always contains Y, Northing and Longitude
373void CDnaStation::ReduceStations_LLH(const CDnaEllipsoid* m_eEllipsoid, const CDnaProjection* m_pProjection)
374{
375 switch (m_ctType)
376 {
377 case XYZ_type_i:
378 // Convert from cartesian to geographic (radians)
379 CartToGeo<double>(m_dXAxis, m_dYAxis, m_dZAxis, &m_dXAxis, &m_dYAxis, &m_dHeight, m_eEllipsoid);
380 m_dZAxis= 0.;
381 m_htType = ELLIPSOIDAL_type_i;
382
383 // Force the current type to be geographic (ellipsoid height).
384 // The user-supplied type is retained in m_ctTypeSupplied
385 m_ctType = LLh_type_i;
386 m_strType = LLh_type;
387 break;
388 case UTM_type_i:
389 // Convert from projection to geographic (radians)
390 GridToGeo<double, UINT32>(m_dXAxis, m_dYAxis, m_zone, &m_dXAxis, &m_dYAxis,
391 m_eEllipsoid->GetSemiMajor(), m_eEllipsoid->GetInverseFlattening(), // ellipsoid parameters
392 m_pProjection->GetFalseEasting(), m_pProjection->GetFalseNorthing(), // projection parameters
393 m_pProjection->GetCentralScaleFactor(), m_pProjection->GetLongCentralMeridianZone1(),
394 m_pProjection->GetZoneWidth());
395
396 // Force the current type to be geographic (orthometric height).
397 // The user-supplied type is retained in m_ctTypeSupplied
398 m_ctType = LLH_type_i;
399 m_strType = LLH_type;
400 break;
401 default:
402 break;
403 }
404}

406// Reduce from Geographic or Projection to Cartesian
407// m_dXAxis always contains X, Easting and Latitude
408// m_dYAxis always contains Y, Northing and Longitude
409void CDnaStation::ReduceStations_XYZ(const CDnaEllipsoid* m_eEllipsoid, const CDnaProjection* m_pProjection)
410{
411 switch (m_ctType)
412 {
413 case UTM_type_i:
414 // Convert from projection to geographic (radians)
415 GridToGeo<double, UINT32>(m_dXAxis, m_dYAxis, m_zone, &m_dXAxis, &m_dYAxis,
416 m_eEllipsoid->GetSemiMajor(), m_eEllipsoid->GetInverseFlattening(), // ellipsoid parameters
417 m_pProjection->GetFalseEasting(), m_pProjection->GetFalseNorthing(), // projection parameters
418 m_pProjection->GetCentralScaleFactor(), m_pProjection->GetLongCentralMeridianZone1(),
419 m_pProjection->GetZoneWidth());
420 [[fallthrough]];
421 case LLH_type_i:
422 case LLh_type_i:
423 m_dcurrentLatitude = m_dXAxis;
424 m_dcurrentLongitude = m_dYAxis;
425 m_dcurrentHeight = m_dHeight;
426 if (m_ctType == LLH_type_i)
427 m_dcurrentHeight += m_fgeoidSep;
428 // Convert from geographic (radians) to cartesian
429 // Assumes height is ellipsoidal
430 GeoToCart<double>(m_dcurrentLatitude, m_dcurrentLongitude, m_dcurrentHeight, &m_dXAxis, &m_dYAxis, &m_dZAxis, m_eEllipsoid);
431 break;
432 default:
433 break;
434 }
435
436 // Force the current type to be geographic. The user-supplied type is retained in m_ctTypeSupplied
437 m_ctType = XYZ_type_i;
438 m_strType = XYZ_type;
439}

441bool CDnaStation::IsValidConstraint(const std::string& sConst)
442{
443 if (boost::iequals(sConst, "CCC"))
444 return true;
445 if (boost::iequals(sConst, "CCF"))
446 return true;
447 if (boost::iequals(sConst, "CFF"))
448 return true;
449 if (boost::iequals(sConst, "FFF"))
450 return true;
451 if (boost::iequals(sConst, "FFC"))
452 return true;
453 if (boost::iequals(sConst, "FCC"))
454 return true;
455 if (boost::iequals(sConst, "FCF"))
456 return true;
457 if (boost::iequals(sConst, "CFC"))
458 return true;
459 return false;
460}

463_COORD_TYPE_ CDnaStation::GetCoordTypeC(const std::string& sType)
464{
465 // case insensitive
466 if (boost::iequals(sType, XYZ_type))
467 return XYZ_type_i;
468 else if (boost::iequals(sType, UTM_type))
469 return UTM_type_i; // height is assumed to be orthometric
470 else if (boost::iequals(sType, ENU_type))
471 return ENU_type_i;
472
473 // case sensitive
474 else if (boost::equals(sType, LLh_type))
475 return LLh_type_i; // ellipsoid height
476
477 // default
478 else if (boost::equals(sType, LLH_type))
479 return LLH_type_i; // orthometric height (default)
480
481 // If this point is reached, sType is an unknown coordinate type, so throw!
482 std::stringstream ss;
483 ss << " '" << sType << "' is not a recognised coordinate type." << std::endl;
484 throw boost::enable_current_exception(std::runtime_error(ss.str()));
485
486 return LLH_type_i;
487}

708void CDnaStation::WriteDNAXMLStn(std::ofstream* dna_ofstream,
709 const CDnaEllipsoid* ellipsoid, const CDnaProjection* projection,
710 const double& lat_east_x, const double& lon_north_y, const double& ht_zone_z,
711 INPUT_FILE_TYPE t, const dna_stn_fields* dsw)
712{
713 //m_ctTypeSupplied = LLh_type_i;
714
715 std::string hemisphereZone(m_strHemisphereZone);
716 std::string coordinateType(LLH_type);
717
718 switch (m_ctTypeSupplied)
719 {
720 case XYZ_type_i:
721 coordinateType = XYZ_type;
722 break;
723 case LLh_type_i:
724 coordinateType = LLh_type;
725 break;
726 case UTM_type_i:
727 coordinateType = UTM_type;
728 break;
729 case ENU_type_i: // Not supported yet, so revert to Lat, Long, Height
730 coordinateType = LLH_type;
731 m_ctTypeSupplied = LLH_type_i;
732 break;
733 case LLH_type_i:
734 coordinateType = LLH_type;
735 break;
736 default:
737 break;
738 }
739
740 // Convert height to orthometric?
741 // Decisions for printing (estimated) station height for various input coordinate types
742 // and geoid model. Estimated stations are in LLh.
743 // h = H + Na
744 //
745 // Input Coord type Geoid loaded? Geoid not loaded? Result
746 // ---------------------------------------------------------------------------------
747 // LLH ht_z - m_fgeoidSep ht_z (m_fgeoidSep = 0) H (orthometric)
748 // UTM ht_z - m_fgeoidSep ht_z (m_fgeoidSep = 0) H (orthometric)
749 // ENU ht_z - m_fgeoidSep ht_z (m_fgeoidSep = 0) H (orthometric)
750 // LLh ht_z ht_z h (ellipsoid)
751 // XYZ LLh->XYZ LLh->XYZ X, Y, Z
752 double lat_east_x_mod(lat_east_x);
753 double lon_north_y_mod(lon_north_y);
754 double ht_zone_z_mod(ht_zone_z);
755
756 switch (m_ctTypeSupplied)
757 {
758 case LLH_type_i:
759 case UTM_type_i:
760 // reduce to orthometric.
761 // If geoid hasn't been loaded, m_fgeoidSep will be 0.0
762 ht_zone_z_mod -= m_fgeoidSep;
763 break;
764 default:
765 break;
766 }
767
768 // Convert coordinates to cartesian or utm?
769 // Convert radians values to degrees, minutes and seconds?
770 std::stringstream ss;
771 double zone;
772 switch (m_ctTypeSupplied)
773 {
774 case XYZ_type_i:
775 GeoToCart<double>(lat_east_x, lon_north_y, ht_zone_z,
776 &lat_east_x_mod, &lon_north_y_mod, &ht_zone_z_mod, ellipsoid);
777 break;
778
779 case UTM_type_i:
780 GeoToGrid<double>(lat_east_x, lon_north_y,
781 &lat_east_x_mod, &lon_north_y_mod, &zone, ellipsoid, projection, true);
782
783 ss << std::fixed << std::setprecision(0) << zone;
784 hemisphereZone = ss.str();
785 break;
786
787 case LLh_type_i:
788 case LLH_type_i:
789 lat_east_x_mod = RadtoDms(lat_east_x);
790 lon_north_y_mod = RadtoDmsL(lon_north_y);
791 break;
792 default:
793 break;
794 }
795
796 switch (t)
797 {
798 case dna:
799
800 // Write DNA station file
801 WriteDNAStn(dna_ofstream, coordinateType,
802 lat_east_x_mod, lon_north_y_mod, ht_zone_z_mod,
803 hemisphereZone, *dsw);
804 break;
805
806 case dynaml:
807
808 // Write DynAdjust XML station file
809 WriteDynaMLStn(dna_ofstream, coordinateType,
810 lat_east_x_mod, lon_north_y_mod, ht_zone_z_mod,
811 hemisphereZone);
812 break;
813 default:
814 break;
815 }
816}
