27// Returns the number of fields with valid data
28int GetFields(char *line, char delim, bool multiple_delim_as_one, const char *fmt, ...)
29{
30 va_list ap;
31 size_t length = strlen(line);
32
33 // 1. determine "expected" number of fields
34 size_t field_count = strlen(fmt);
35 size_t* field = new size_t[field_count+1];
36
37 size_t i(0), j(0);
38
39 // 2. Determine indices of the beginning of each field
40
41 // first field commences at 0
42 field[j++] = 0;
43
44 // 3. Determine indices of remaining fields, but don't let j
45 // go beyond the number of 'wanted' fields (field_count)
46 while (line[i] != '\n' && i < length && j <= field_count)
47 //while (line[i] != '\n' && i < length)
48 //while(line[i] != '\n' && line[i] != '\0')
49 {
50 // replace all delimiters with NULL
51 if (line[i] == delim)
52 {
53 if (multiple_delim_as_one && i < length-1)
54 {
55 if (line[i+1] == delim)
56 {
57 i++;
58 continue;
59 }
60 }
61 line[i] = '\0';
62 field[j++] = i+1;
63 }
64 i++;
65 }
66
67 // 4. terminate end of the wanted string with null
68 line[i] = '\0';
69 i = 0;
70 va_start(ap, fmt);
71
72 int success(0);
73
74 char* p;
75
76 // 5. get the data
77 while (i < j && *fmt != '\0')
78 {
79 p = &line[field[i]];
80
81 switch(*fmt) {
82 case 'd':
83 if (strlen(p) == 0)
84 *(va_arg(ap, int *)) = INT_MIN;
85 else
86 {
87 success++;
88 *(va_arg(ap, int *)) = atoi(p);
89 }
90 break;
91 case 'f':
92 if (strlen(p) == 0)
93 *(va_arg(ap, double *)) = MAX_DBL_VALUE;
94 else
95 {
96 success++;
97 *(va_arg(ap, double *)) = atof(p);
98 }
99 break;
100 case 's':
101 if (strlen(p) != 0)
102 success++;
103 strcpy((va_arg(ap, char *)), p);
104 break;
105 default :
106 delete[] field;
107 return 0;
108 }
109 i++;
110 fmt++;
111 }
112
113 va_end(ap);
114 delete[] field;
115
116 // 6. Return number of fields with non-empty data
117 return success;
118}

231{
232 std::stringstream str;
233 str << "!#=DNA " << fileVersion << " " << fileType;
234
235 boost::gregorian::date today(boost::gregorian::day_clock::local_day());
236
237 // creation date
238 str << std::setw(14) << std::right << stringFromDate<boost::gregorian::date>(today);
239 // default reference frame
240 str << std::setw(14) << std::right << reference_frame;
241 // default epoch
242 str << std::setw(14) << std::right << epoch_version;
243 // count of records
244 str << std::setw(10) << std::right << count;
245
246 os << str.str() << std::endl;
247
248 // Provide comments on the version of DynAdjust that is creating this file
249 os << "* ";
250 output_createdby(os);
251 os << ". " << std::endl;
252 os << "* ";
253 output_version(os);
254 os << ". " << std::endl;
255 os << "* ";
256 output_build(os);
257 os << std::endl;
