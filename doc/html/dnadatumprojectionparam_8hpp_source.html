59// nu
60template <class T>
61T primeVertical_(const T& semiMajor, const T& e1Sqd, const T& latitude)
62{
63 return (semiMajor / sqrt(1.0 - e1Sqd * (sin(latitude)*sin(latitude))));
64}
65
66// rho
67template <class T>
68double primeMeridian_(const T& semiMajor, const T& e1Sqd, const T& latitude)
69{
70 T dDel = sqrt(1.0 - e1Sqd * (sin(latitude)*sin(latitude)));
71 return (semiMajor * ((1.0 - e1Sqd) / (dDel * dDel * dDel)) );
72}
73
74// nu and rho
75template <class T>
76void primeVerticalandMeridian_(const T& semiMajor, const T& e1Sqd, const T& latitude, T& nu, T& rho)
77{
78 T dDel = sqrt(1.0 - e1Sqd * (sin(latitude)*sin(latitude)));
79
80 nu = semiMajor / dDel;
81 rho = semiMajor * ((1.0 - e1Sqd) / (dDel * dDel * dDel));
82}
83
84// average radius of curvature
85template <class T>
86T averageRadiusofCurvature_(const T& semiMajor, const T& e1Sqd, const T& latitude)
87{
88 T nu, rho;
89 primeVerticalandMeridian_(semiMajor, e1Sqd, latitude, nu, rho);
90 return sqrt(nu * rho);
91}
