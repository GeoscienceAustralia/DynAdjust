65template <class S=std::string, class U>
66S TransformationType(const U& u)
67{
68 switch (u)
69 {
70 case __static_to_static__:
71 return "SS";
72 case __static_to_dynamic__:
73 return "SD";
74 case __static_to_step__:
75 // static (S) to intermediate (I)
76 return "SI";
77 case __dynamic_to_step__:
78 // dynamic (D) to intermediate (I)
79 return "DI";
80 case __dynamic_to_static__:
81 return "DS";
82 case __dynamic_to_dynamic__:
83 return "DD";
84 case __step_to_dynamic__:
85 // intermediate (I) to dynamic (D)
86 return "ID";
87 case __step_to_static__:
88 // intermediate (I) to static (S)
89 return "IS";
90 case __plate_motion_model__:
91 return "PM";
92 default:
93 return " ";
94 }
95}

154// SPECIAL NOTE ON ROTATION RATES FOR IERS PARAMETERS
155// ----------------------------------------------------------
156//
157// There are two different ways of applying the sign conventions for the rotations.
158// In both cases a positive rotation is an anti-clockwise rotation, when viewed along
159// the positive axis towards the origin but:
160//
161// 1. The IERS assumes the rotations to be of the points around the
162// cartesian axes, while;
163// 2. The method historically used in Australia assumes the rotations
164// to be of the Cartesian axes around the points.
165//
166// Although these two conventions exist, to enforce the property that all rotations
167// describe anticlockwise rotation as positive when viewed along the axis towards the
168// origin, the rotation of the coordinate axes around the points should be a
169// skew-symmetric matrix with the opposite sign to the rotation of the point(s)
170// around the coordinate axis.
171//
172// To avoid confusion and to achieve consistency in the implementation of the
173// transformation functions, all published IERS rotations and associated rotation rates
174// have been reversed.
175//
177
178
180// ITRF2000 <-> ITRF...
181// http://itrf.ensg.ign.fr/doc_ITRF/Transfo-ITRF2000_ITRFs.txt
182
184// SPECIAL NOTE ON IERS PARAMETERS for ITRF2000 -> ITRFxx
185// ----------------------------------------------------------
186//
187// Despite what the website says, the linear values for translation are in centimetres,
188// not millimetres.
189//
190// To avoid confusion and to achieve consistency in the implementation of the
191// transformation functions, all published IERS translations and associated
192// translation rates have been converted to millimetres.
193//
