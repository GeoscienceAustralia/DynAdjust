50template <typename T>
51T real_line_length_ascii(const T& line_length_ascii)
52{
53 // return the real length of a line based on the system definition
54 // of a "new line". On Windows, a new line is comprised of two characters
55 // namely, carriage return (CR, '\r') and line feed (LF, '\n'). On Unix
56 // and Unix-like systems (Linux, mac OS, BSD, etc.), a new line is
57 // comprised of a line feed (LF, '\n') only.
58 //
59
60#if defined(__linux) || defined(sun) || defined(__unix__) || defined(__APPLE__)
61 // "real" line length = line_length_ascii + "\n"
62 return line_length_ascii + 1;
63
64#else // #if defined(_WIN32) || defined(__WIN32__)
65 // "real" line length = line_length_ascii + "\r\n"
66 return line_length_ascii + 2;
67
68#endif
69}

83template <typename T>
84void file_opener(
85 T& stream,
86 const std::string& str,
87 std::ios_base::openmode mode=std::ios_base::out, // default: output
88 const iosMode type=ascii, // default: ascii
89 bool fileMustExist=false) // default: no need for file to exist
90{
91 try {
92 stream.open(str.c_str(), mode);
93 stream.exceptions (/*std::ios_base::eofbit | */std::ios_base::badbit | std::ios_base::failbit);
94 stream.iword(0) = type;
95 }
96 catch (const std::ios_base::failure& f) {
97 std::stringstream ss;
98 if (fileMustExist && !boost::filesystem::exists(str.c_str()))
99 ss << "file_opener(): Can't find " << str << ".";
100 else
101 ss << "file_opener(): An error was encountered when opening " <<
102 str << ". \n Check that the file is not already opened.";
103 ss << std::endl << f.what();
104 throw boost::enable_current_exception(std::runtime_error(ss.str()));
105 }
106
107 if (!stream.good()) {
108 std::stringstream ss;
109 ss << "file_opener(): An error was encountered when opening " <<
110 str << ". \n Check that the file is not already opened.";
111 throw boost::enable_current_exception(std::runtime_error(ss.str()));
112 }
113}

115template <typename T>
116void print_file_header(
117 T& stream,
118 const std::string& header)
119{
120 // Print formatted header
121 stream << OUTPUTLINE << std::endl;
122 stream << header << std::endl << std::endl;
123
124 // version
125 output_version(stream, true);
126 stream << std::endl;
127
128 // build
129 output_build(stream, true);
130 stream << std::endl;
131
132 // File creation time
133 stream << std::setw(PRINT_VAR_PAD) << std::left << "File created:";
134 std::ostringstream datetime_ss;
135 boost::posix_time::time_facet* p_time_output = new boost::posix_time::time_facet;
136 std::locale special_locale (std::locale(""), p_time_output);
137 // special_locale takes ownership of the p_time_output facet
138 datetime_ss.imbue (special_locale);
139 (*p_time_output).format("%A, %d %B %Y, %X");
140 datetime_ss << boost::posix_time::second_clock::local_time();
141 stream << datetime_ss.str().c_str() << std::endl;
142}
