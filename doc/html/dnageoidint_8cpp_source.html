113// http://www.codeproject.com/KB/MFC/checkapprunning.aspx
114// Checks, if an application with this name is running
115//
116// bShow ..... TRUE: bring application to foreground, if running
117// FALSE: only check, don't move to the application
118//
119// return: FALSE: application is not running
120// TRUE: application runs
121BOOL AppIsAlreadyRunning(BOOL bShow=TRUE)
122{
123 BOOL bRunning=FALSE;
124 CString strAppName;
125 strAppName.Format("%s", __BINARY_NAME__);
126 strAppName += _T(".exe");
127 DWORD dwOwnPID = GetProcessId(GetCurrentProcess());
128 HANDLE hSnapShot=CreateToolhelp32Snapshot(TH32CS_SNAPPROCESS,0);
129 PROCESSENTRY32* processInfo=new PROCESSENTRY32;
130 processInfo->dwSize=sizeof(PROCESSENTRY32);
131 int index=0;
132 while(Process32Next(hSnapShot,processInfo)!=FALSE)
133 {
134 if (!strcmp(processInfo->szExeFile,strAppName))
135 {
136 if (processInfo->th32ProcessID != dwOwnPID)
137 {
138 if (bShow)
139 EnumWindows(ShowAppEnum, processInfo->th32ProcessID);
140 bRunning=TRUE;
141 break;
142 }
143 }
144 }
145 CloseHandle(hSnapShot);
146 delete processInfo;
147 return bRunning;
148}
