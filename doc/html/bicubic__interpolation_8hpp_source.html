65// Given arrays y[1..4], y1[1..4], y2[1..4], and y12[1..4], containing the function, gradients,
66// and cross derivative at the four grid points of a rectangular grid cell (numbered counterclockwise
67// from the lower left), and given d1 and d2, the length of the grid cell in the 1- and
68// 2-directions, this routine returns the table c[1..4][1..4] that is used by routine bcuint
69// for bicubic interpolation.
70template <class T>
71void bcucof(T y[], T y1[], T y2[], T y12[], T c[4][4])
72{
73 int l(0), k(0), j(0), i(0);
74 T xx(0.), cl[16], x[16];
75
76 for (i=0; i<4; i++) // Pack a temporary vector x.
77 {
78 x[i] = y[i];
79 x[i+4] = y1[i];
80 x[i+8] = y2[i];
81 x[i+12] = y12[i];
82 }
83
84 for (i=0; i<16; i++) // Matrix multiply by the stored table.
85 {
86 xx = 0.0;
87 for (k=0; k<16; k++)
88 xx += wt[i][k] * x[k];
89 cl[i] = xx;
90 }
91
92 l = 0;
93
94 for (i=0; i<4; i++) // Unpack the result into the output table.
95 {
96 for (j=0; j<4; j++)
97 c[i][j] = cl[l++];
98 }
99}
