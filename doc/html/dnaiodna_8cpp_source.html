175void dna_io_dna::read_ren_data(std::ifstream* ptr, pv_string_vstring_pair stnRenaming)
176{
177 std::string sBuf;
178
179 string_vstring_pair stnNames;
180 vstring altStnNames;
181
182 while (!ptr->eof()) // while EOF not found
183 {
184 try {
185 getline((*ptr), sBuf);
186 }
187 catch (...) {
188 if (ptr->eof())
189 return;
190 std::stringstream ss;
191 ss << "read_ren_data(): Could not read from the renaming file." << std::endl;
192 boost::enable_current_exception(std::runtime_error(ss.str()));
193 }
194
195 // blank or whitespace?
196 if (trimstr(sBuf).empty())
197 continue;
198
199 // Ignore lines with blank station name
200 if (trimstr(sBuf.substr(dsl_.stn_name, dsw_.stn_name)).empty())
201 continue;
202
203 // Ignore lines with comments
204 if (sBuf.compare(0, 1, "*") == 0)
205 continue;
206
207 std::string tmp;
208
209 // Add the preferred name
210 try {
211 tmp = trimstr(sBuf.substr(dsl_.stn_name, dsw_.stn_name));
212 stnNames.first = tmp;
213 }
214 catch (...) {
215 std::stringstream ss;
216 ss << "read_ren_data(): Could not extract station name from the record: " <<
217 std::endl << " " << sBuf << std::endl;
218 boost::enable_current_exception(std::runtime_error(ss.str()));
219 }
220
221 // Alternative names
222 altStnNames.clear();
223 UINT32 positionEndName(dsw_.stn_name+dsw_.stn_name);
224 UINT32 positionNextName(dsw_.stn_name);
225 while (sBuf.length() > positionNextName)
226 {
227 if (sBuf.length() > positionEndName)
228 tmp = trimstr(sBuf.substr(positionNextName, dsw_.stn_name));
229 else
230 tmp = trimstr(sBuf.substr(positionNextName));
231
232 if (!tmp.empty())
233 altStnNames.push_back(tmp);
234
235 positionNextName = positionEndName;
236 positionEndName += dsw_.stn_name;
237 }
238
239 // sort for faster searching
240 std::sort(altStnNames.begin(), altStnNames.end());
241
242 // Add the alternate names
243 stnNames.second = altStnNames;
244
245 // Okay, add the pair to the list
246 stnRenaming->push_back(stnNames);
247 }
248}

251void dna_io_dna::read_dna_header(std::ifstream* ptr, std::string& version, INPUT_DATA_TYPE& idt,
252 CDnaDatum& referenceframe,
253 std::string& fileEpsg, std::string& fileEpoch, std::string& geoidversion, UINT32& count)
254{
255 std::string sBuf;
256 getline((*ptr), sBuf);
257 sBuf = trimstr(sBuf);
258 m_filespecifiedReferenceFrame_ = false;
259 m_filespecifiedEpoch_ = false;
260
261 // Set the default version
262 version = "1.00";
263
264 // Attempt to get the file's version
265 try {
266 if (boost::iequals("!#=DNA", sBuf.substr(0, 6)))
267 version = trimstr(sBuf.substr(6, 6));
268 }
269 catch (const std::runtime_error& e) {
270 throw boost::enable_current_exception(std::runtime_error(e.what()));
271 }
272
273 // Attempt to get the file's type
274 try {
275 determineDNASTNFieldParameters<UINT16>(version, dsl_, dsw_);
276 determineDNAMSRFieldParameters<UINT16>(version, dml_, dmw_);
277 }
278 catch (const std::runtime_error& e) {
279 std::stringstream ssError;
280 ssError << "- Error: Unable to determine DNA file version." << std::endl <<
281 sBuf << std::endl << " " << e.what() << std::endl;
282 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
283 }
284
285 // Version 1
286 if (boost::iequals(version, "1.00"))
287 {
288 idt = unknown; // could be stn or msr
289 count = 100; // set default 100 stations
290 return;
291 }
292
293 std::string type;
294 // Attempt to get the file's type
295 try {
296 type = trimstr(sBuf.substr(12, 3));
297 }
298 catch (const std::runtime_error& e) {
299 std::stringstream ssError;
300 ssError << "- Error: File type has not been provided in the header" << std::endl <<
301 sBuf << std::endl << e.what() << std::endl;
302 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
303 }
304
305 // Station file
306 if (boost::iequals(type, "stn"))
307 idt = stn_data;
308 // Measurement file
309 else if (boost::iequals(type, "msr"))
310 idt = msr_data;
311 // Geoid information file
312 else if (boost::iequals(type, "geo"))
313 idt = geo_data;
314 // Station renaming
315 else if (boost::iequals(type, "ren"))
316 idt = ren_data;
317 else
318 {
319 idt = unknown;
320 std::stringstream ssError;
321 ssError << "The supplied filetype '" << type << "' is not recognised" << std::endl;
322 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
323 }
324
325 if (sBuf.length() < 29)
326 {
327 count = 100;
328 return;
329 }
330
331 std::string file_referenceframe;
332 // Attempt to get the default reference frame
333 try {
334 if (sBuf.length() > 42)
335 file_referenceframe = trimstr(sBuf.substr(29, 14));
336 else if (sBuf.length() > 29)
337 file_referenceframe = trimstr(sBuf.substr(29));
338 else
339 file_referenceframe = "";
340 }
341 catch (...) {
342 // datum not provided?
343 file_referenceframe = "";
344 }
345
346 if (!file_referenceframe.empty())
347 m_filespecifiedReferenceFrame_ = true;
348
349 std::string epoch_version;
350 // Attempt to get the default epoch / geoid version
351 try {
352 if (sBuf.length() > 56)
353 epoch_version = trimstr(sBuf.substr(43, 14));
354 else if (sBuf.length() > 43)
355 epoch_version = trimstr(sBuf.substr(43));
356 else
357 epoch_version = "";
358 }
359 catch (...) {
360 // epoch / version not provided?
361 epoch_version = "";
362 }
363
364 if (!epoch_version.empty())
365 m_filespecifiedEpoch_ = true;
366
367 // Try to get the reference frame
368 // For the first file, and unless the user has specified a reference frame,
369 // the reference frame will be used to set the frame for all other files.
370 try {
371
372 switch (idt)
373 {
374 case stn_data:
375 case msr_data:
376
377 // Capture file epsg
378 if (file_referenceframe.empty())
379 // Empty? Get the epsg code from the project datum
380 fileEpsg = referenceframe.GetEpsgCode_s();
381 else
382 fileEpsg = epsgStringFromName<std::string>(file_referenceframe);
383
384 // capture file epoch
385 if (epoch_version.empty())
386 {
387 // Empty? Get the epoch of the nominated epsg (set above)
388 fileEpoch = referenceepochFromEpsgString<std::string>(fileEpsg);
389 }
390 else
391 fileEpoch = epoch_version;
392
393 break;
394 case geo_data:
395 geoidversion = epoch_version;
396 break;
397 default:
398 break;
399 }
400 }
401 catch (const std::runtime_error& e) {
402 std::stringstream ssError;
403 ssError << "The supplied frame (" << file_referenceframe << ") is not recognised" << std::endl <<
404 e.what() << std::endl;
405 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
406 }
407
408 // Attempt to get the data count
409 try {
410 if (sBuf.length() > 66)
411 count = val_uint<UINT32, std::string>(trimstr(sBuf.substr(57, 10)));
412 else if (sBuf.length() > 57)
413 count = val_uint<UINT32, std::string>(trimstr(sBuf.substr(57)));
414 else
415 count = 100;
416 }
417 catch (...) {
418 count = 100;
419 }
420}

423void dna_io_dna::write_stn_file(pvstn_t vbinary_stn, const std::string& stnfilename, const std::string& networkname,
424 const CDnaDatum& datum, const CDnaProjection& projection, bool flagUnused,
425 const std::string& comment)
426{
427 // Print DNA stations from a vector of stn_t
428 std::ofstream dna_stn_file;
429
430 try {
431 // Create file pointer
432 create_file_stn(&dna_stn_file, stnfilename);
433
434 // Write header and comment
435 write_stn_header(&dna_stn_file, vbinary_stn, networkname, datum, flagUnused, comment);
436
437 // Sort on original file order
438 prepare_sort_list(static_cast<UINT32>(vbinary_stn->size()));
439 CompareStnFileOrder<station_t, UINT32> stnorderCompareFunc(vbinary_stn);
440 std::sort(vStationList_.begin(), vStationList_.end(), stnorderCompareFunc);
441
442 dnaStnPtr stnPtr(new CDnaStation(datum.GetName(), datum.GetEpoch_s()));
443
444 // print stations
445 // Has the user specified --flag-unused-stations, in wich case, do not
446 // print stations marked unused?
447 if (flagUnused)
448 {
449 // Print stations in DNA format
450 for_each(vStationList_.begin(), vStationList_.end(),
451 [&dna_stn_file, &stnPtr, &projection, &datum, &vbinary_stn, this](const UINT32& stn) {
452 if (stnPtr->IsNotUnused())
453 {
454 stnPtr->SetStationRec(vbinary_stn->at(stn));
455 stnPtr->WriteDNAXMLStnCurrentEstimates(&dna_stn_file, datum.GetEllipsoidRef(),
456 const_cast<CDnaProjection*>(&projection), dna, &dsw_);
457 }
458 });
459 }
460 else
461 {
462 // Print stations in DNA format
463 for_each(vStationList_.begin(), vStationList_.end(),
464 [&dna_stn_file, &stnPtr, &projection, &datum, &vbinary_stn, this](const UINT32& stn) {
465 stnPtr->SetStationRec(vbinary_stn->at(stn));
466 stnPtr->WriteDNAXMLStnCurrentEstimates(&dna_stn_file, datum.GetEllipsoidRef(),
467 const_cast<CDnaProjection*>(&projection), dna, &dsw_);
468 });
469 }
470
471 dna_stn_file.close();
472
473 }
474 catch (const std::ifstream::failure& f) {
475 throw boost::enable_current_exception(std::runtime_error(f.what()));
476 }
477}
