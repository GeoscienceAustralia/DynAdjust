125void CDnaEllipsoid::CalculateEllipsoidParams()
126{
127 m_dB = m_dA * (1.0 - (1.0 / m_dInv_f));
128 double dAsqd = m_dA * m_dA;
129 double dBsqd = m_dB * m_dB;
130 m_dEccen1_sqd = (dAsqd - dBsqd) / dAsqd; // first ecc squared
131 m_dEccen1 = sqrt(m_dEccen1_sqd); // first ecc
132 m_dEccen2_sqd = (dAsqd - dBsqd) / dBsqd; // second ecc squared
133 m_dEccen2 = sqrt(m_dEccen2_sqd); // second ecc
134 m_dF = 1.0 / m_dInv_f;
135}
