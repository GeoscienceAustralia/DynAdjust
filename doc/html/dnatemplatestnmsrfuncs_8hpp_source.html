57template <typename T>
58// Get all the statons asociated with a measurement
59void GetGXMsrStations(std::vector<CDnaGpsBaseline>* vgpsBsls, std::vector<T>& msrStations)
60{
61 msrStations.clear();
62
63 // Step 1. Set ignore flag for all baselines to true
64 for_each(vgpsBsls->begin(), vgpsBsls->end(),
65 [&msrStations](CDnaGpsBaseline& bsl) {
66 msrStations.push_back(bsl.GetFirst());
67 msrStations.push_back(bsl.GetTarget());
68 });
69
70 // Strip duplicate entries
71 strip_duplicates(msrStations);
72}

75template <typename T>
76// Get all the statons asociated with a measurement
77void GetMsrStations(const vmsr_t& binaryMsrs, const T& bmsIndex, std::vector<T>& msrStations)
78{
79 msrStations.clear();
80 T clusterID(binaryMsrs.at(bmsIndex).clusterID);
81
82 it_vmsr_t_const _it_msr_start = binaryMsrs.begin() + bmsIndex;
83 it_vmsr_t_const _it_msr_prev = _it_msr_start;
84
85 bool bCluster(false);
86
87 // I don't think this is necessary for X measruements, since
88 // bmsIndex points to the first element in a cluster. Perhaps keep
89 // this here to safe guard against cases where bmsIndex is not the
90 // first???
91 switch (binaryMsrs.at(bmsIndex).measType)
92 {
93 case 'D':
94 case 'X':
95 case 'Y':
96 // move to the beginning of the cluster
97 while (_it_msr_start != binaryMsrs.begin())
98 {
99 if ((--_it_msr_prev)->clusterID != clusterID)
100 break;
101 _it_msr_start--;
102 }
103 bCluster = true;
104 break;
105 }
106
107 while (_it_msr_start->clusterID == clusterID) // Will be true for non-cluster measurements
108 {
109 if (_it_msr_start->measType != 'D')
110 {
111 if (_it_msr_start->measStart != xMeas)
112 {
113 if (++_it_msr_start == binaryMsrs.end())
114 break;
115 continue;
116 }
117 }
118
119 // add the relevant stations
120 // Station 1 is in all measurement types
121 msrStations.push_back(_it_msr_start->station1);
122 // Station 2 is in two- and three-station measurements
123 if (MsrTally::Stations(_it_msr_start->measType) >= TWO_STATION)
124 msrStations.push_back(_it_msr_start->station2);
125 // Station 3 is only in three-station measurements
126 if (MsrTally::Stations(_it_msr_start->measType) == THREE_STATION)
127 msrStations.push_back(_it_msr_start->station3);
128
129 if (!bCluster)
130 break;
131
132 if (++_it_msr_start == binaryMsrs.end())
133 break;
134 }
135
136 // Strip duplicate entries
137 strip_duplicates(msrStations);
138}

140template <typename T>
141// Get all the binary measurement indices involved in a measurement
142// On return, msrIndices will contain only one index. For clusters
143// and directions, msrIndices will be more than one.
144void GetMsrIndices(const vmsr_t& binaryMsrs, const T& bmsIndex, std::vector<T>& msrIndices)
145{
146 msrIndices.clear();
147
148 // First, sift out conventional measurement types and return immediately.
149 switch (binaryMsrs.at(bmsIndex).measType)
150 {
151 case 'D':
152 case 'X':
153 case 'Y':
154 // Clusters dealt with after this switch statement
155 break;
156 default:
157 // basic measurement type for which there is only one index, so
158 // add the index and return
159 msrIndices.push_back(bmsIndex);
160 return;
161 }
162
163 // Ok, now deal with cluster measurements
164 UINT32 clusterID(binaryMsrs.at(bmsIndex).clusterID);
165
166 it_vmsr_t_const _it_msr_start(binaryMsrs.begin() + bmsIndex);
167 it_vmsr_t_const _it_msr_prev(_it_msr_start);
168
169 // move to the beginning of the cluster
170 while (_it_msr_start != binaryMsrs.begin())
171 {
172 if ((--_it_msr_prev)->clusterID != clusterID)
173 break;
174 --_it_msr_start;
175 }
176
177
178 while (_it_msr_start->clusterID == clusterID && _it_msr_start != binaryMsrs.end())
179 {
180 if (_it_msr_start->ignore)
181 {
182 if (++_it_msr_start == binaryMsrs.end())
183 break;
184 continue;
185 }
186
187 if (_it_msr_start->measType != 'D')
188 {
189 if (_it_msr_start->measStart > xMeas)
190 {
191 if (++_it_msr_start == binaryMsrs.end())
192 break;
193 continue;
194 }
195 }
196
197 msrIndices.push_back(static_cast<T>(std::distance(binaryMsrs.begin(), _it_msr_start)));
198
199 if (++_it_msr_start == binaryMsrs.end())
200 break;
201 }
202
203 // Sort and strip duplicate entries
204 strip_duplicates(msrIndices);
205}

207template <typename T>
208// Get the first binary measurement index for a measurement
209T GetFirstMsrIndex(const vmsr_t& binaryMsrs, const T& bmsIndex)
210{
211 // First, sift out conventional measurement types and return immediately.
212 switch (binaryMsrs.at(bmsIndex).measType)
213 {
214 case 'D':
215 case 'X':
216 case 'Y':
217 // Clusters dealt with after this switch statement
218 break;
219 default:
220 // basic measurement type for which there is only one index, so
221 // return bmsIndex
222 return bmsIndex;
223 }
224
225 // Ok, now deal with cluster measurements
226 UINT32 clusterID(binaryMsrs.at(bmsIndex).clusterID);
227
228 it_vmsr_t_const _it_msr_start(binaryMsrs.begin() + bmsIndex);
229 it_vmsr_t_const _it_msr_prev(_it_msr_start);
230
231 // move to the beginning of the cluster
232 while (_it_msr_start != binaryMsrs.begin())
233 {
234 if ((--_it_msr_prev)->clusterID != clusterID)
235 break;
236 --_it_msr_start;
237 }
238
239 // return the index of the measurement's first record in the binary measurement file
240 return static_cast<UINT32>(std::distance(binaryMsrs.begin(), _it_msr_start));
241}

277template <typename T, typename Iter>
278void RenameStationsMsr(T* msr, Iter begin, Iter end)
279{
280 _it_string_vstring_pair it;
281 Iter original = begin;
282
283 // Search the aliases for the first station name
284 // Emulate for_each behaviour
285 while (begin != end)
286 {
287 if (binary_search(begin->second.begin(), begin->second.end(), msr->GetFirst()))
288 {
289 msr->SetFirst(begin->first);
290 break;
291 }
292 ++begin;
293 }
294
295 // Is this measurement a one-station measurement?
296 if (msr->m_MSmeasurementStations == ONE_STATION)
297 return;
298
299 // Search the aliases for the second station name
300 // Emulate for_each behaviour
301 begin = original;
302 while (begin != end)
303 {
304 if (binary_search(begin->second.begin(), begin->second.end(), msr->GetTarget()))
305 {
306 msr->SetTarget(begin->first);
307 break;
308 }
309 ++begin;
310 }
311
312 // Is this measurement a two-station measurement?
313 if (msr->m_MSmeasurementStations == TWO_STATION)
314 return;
315
316 // Search the aliases for the third station name
317 // Emulate for_each behaviour
318 begin = original;
319 while (begin != end)
320 {
321 if (binary_search(begin->second.begin(), begin->second.end(), msr->GetTarget2()))
322 {
323 msr->SetTarget2(begin->first);
324 break;
325 }
326 ++begin;
327 }
328}

330template <typename T, typename Iter>
331void IgnoreGXMeasurements(T* msr, Iter begin, Iter end)
332{
333
334 // Search the aliases for the first station name
335 // Emulate for_each behaviour
336 if (binary_search(begin, end, msr->GetClusterID()))
337 {
338 msr->SetIgnore(true);
339 for_each(msr->GetBaselines_ptr()->begin(),
340 msr->GetBaselines_ptr()->end(),
341 [] (CDnaGpsBaseline& bsl) {
342 bsl.SetIgnore(true);
343 });
344 }

348template <typename T, typename msriterator>
349// Copy the cluster measurement
350void CopyClusterMsr(T& cluster, const msriterator _it_msr, T& clusterCopy)
351{
352 clusterCopy.clear();
353
354 // First, sift out conventional measurement types and return immediately.
355 switch (_it_msr->measType)
356 {
357 case 'D':
358 case 'X':
359 case 'Y':
360 // Clusters dealt with after this switch statement
361 break;
362 default:
363 // basic measurement type for which there is only one element
364 clusterCopy.push_back(*_it_msr);
365 return;
366 }
367
368 // Ok, now deal with cluster measurements
369 UINT32 clusterID(_it_msr->clusterID);
370
371 msriterator _it_msr_start(_it_msr);
372 msriterator _it_msr_temp(_it_msr_start);
373
374 while (_it_msr_start != cluster.begin())
375 {
376 if ((--_it_msr_temp)->clusterID != clusterID)
377 break;
378 --_it_msr_start;
379 }
380
381 msriterator _it_msr_last(_it_msr);
382
383 while (_it_msr_last != cluster.end())
384 {
385 if ((++_it_msr_temp)->measType != 'Y')
386 {
387 ++_it_msr_last;
388 break;
389 }
390 ++_it_msr_last;
391 }
392
393 clusterCopy.insert(clusterCopy.begin(), _it_msr_start, _it_msr_last);
394}

499template <typename T = dnaStnPtr, typename S = stringstring_doubledouble_pair, typename U = double>
500class NearbyStation_LowAcc {
501public:
502 NearbyStation_LowAcc (const U& tolerance, std::vector<S>* stns)
503 : _tolerance(tolerance), _stns(stns), _dist(0.) {}
504
505 bool operator()(const T& left, const T& right) {
506 if ((_dist = GreatCircleDistance(
507 left->GetLatitude(),
508 left->GetLongitude(),
509 right->GetLatitude(),
510 right->GetLongitude())) < _tolerance)
511 {
512 _stns->push_back(stringstring_doubledouble_pair(
513 string_string_pair(left->GetName(), right->GetName()),
514 doubledouble_pair(_dist, right->GetHeight()-left->GetHeight())));
515 return true;
516 }
517 return false;
518 }
519
520 U _tolerance;
521 std::vector<S>* _stns;
522 double _dist;
523};

549 U _tolerance;
550 std::vector<S>* _stns;
551 U _dAzimuth;
552 double _dist;
553 E _ellipsoid; // GDA by default

734template<typename T>
735void ResetMeasurementPtr(dnaMsrPtr* msrPtr, const T& cType)
736{
737 switch (cType)
738 {
739 case 'A': // Horizontal angle
740 msrPtr->reset(new CDnaAngle);
741 break;
742 case 'B': // Geodetic azimuth
743 case 'K': // Astronomic azimuth
744 msrPtr->reset(new CDnaAzimuth);
745 break;
746 case 'C': // Chord dist
747 case 'E': // Ellipsoid arc
748 case 'M': // MSL arc
749 case 'S': // Slope distance
750 msrPtr->reset(new CDnaDistance);
751 break;
752 case 'D': // Direction set
753 msrPtr->reset(new CDnaDirectionSet);
754 break;
755 case 'G': // GPS Baseline (treat as single-baseline cluster)
756 case 'X': // GPS Baseline cluster
757 msrPtr->reset(new CDnaGpsBaselineCluster);
758 break;
759 case 'H': // Orthometric height
760 msrPtr->reset(new CDnaHeight);
761 break;
762 case 'I': // Astronomic latitude
763 case 'J': // Astronomic longitude
764 case 'P': // Geodetic latitude
765 case 'Q': // Geodetic longitude
766 msrPtr->reset(new CDnaCoordinate);
767 break;
768 case 'L': // Level difference
769 msrPtr->reset(new CDnaHeightDifference);
770 break;
771 case 'R': // Ellipsoidal height
772 msrPtr->reset(new CDnaHeight);
773 break;
774 case 'V': // Zenith distance
775 case 'Z': // Vertical angle
776 msrPtr->reset(new CDnaDirection);
777 break;

815// used to sort measurements
816template <typename M = CDnaMeasurement>
817class CompareMsr {
818public:
819 bool operator()(const boost::shared_ptr<M> left, const boost::shared_ptr<M> right) {
820 if (left->GetIgnore() == right->GetIgnore()) {
821 if (boost::iequals(left->GetType(), right->GetType()))
822 {
823 switch (left->GetTypeC())
824 {
825 case 'A': // Horizontal angles
826 return *(dynamic_cast<const CDnaAngle*>(&(*left))) <
827 *(dynamic_cast<const CDnaAngle*>(&(*right)));
828
829 case 'C': // Chord dist
830 case 'E': // Ellipsoid arc
831 case 'M': // MSL arc
832 case 'S': // Slope distance
833 return *(dynamic_cast<const CDnaDistance*>(&(*left))) <
834 *(dynamic_cast<const CDnaDistance*>(&(*right)));
835
836 case 'D': // Direction set
837 return *(dynamic_cast<const CDnaDirectionSet*>(&(*left))) <
838 *(dynamic_cast<const CDnaDirectionSet*>(&(*right)));
839
840 case 'B': // Geodetic azimuth
841 case 'K': // Astronomic azimuth
842 return *(dynamic_cast<const CDnaAzimuth*>(&(*left))) <
843 *(dynamic_cast<const CDnaAzimuth*>(&(*right)));
844
845 case 'I': // Astronomic latitude
846 case 'J': // Astronomic longitude
847 case 'P': // Geodetic latitude
848 case 'Q': // Geodetic longitude
849 return *(dynamic_cast<const CDnaCoordinate*>(&(*left))) <
850 *(dynamic_cast<const CDnaCoordinate*>(&(*right)));
851
852 case 'H': // Orthometric height
853 case 'R': // Ellipsoidal height
854 return *(dynamic_cast<const CDnaHeight*>(&(*left))) <
855 *(dynamic_cast<const CDnaHeight*>(&(*right)));
856
857 case 'L': // Level difference
858 return *(dynamic_cast<const CDnaHeightDifference*>(&(*left))) <
859 *(dynamic_cast<const CDnaHeightDifference*>(&(*right)));
860
861 case 'V': // Zenith distance
862 case 'Z': // Vertical angle
863 return *(dynamic_cast<const CDnaDirection*>(&(*left))) <
864 *(dynamic_cast<const CDnaDirection*>(&(*right)));
865
866 case 'G': // GPS Baseline (treat as single-baseline cluster)
867 case 'X': // GPS Baseline cluster
868 return *(dynamic_cast<const CDnaGpsBaselineCluster*>(&(*left))) <
869 *(dynamic_cast<const CDnaGpsBaselineCluster*>(&(*right)));
870
871 case 'Y': // GPS point cluster
872 return *(dynamic_cast<const CDnaGpsPointCluster*>(&(*left))) <
873 *(dynamic_cast<const CDnaGpsPointCluster*>(&(*right)));
874 }
875
876 return left < right;
877 }
878 else
879 return left->GetType() < right->GetType(); }
880 return left->GetIgnore() < right->GetIgnore();
881 }
