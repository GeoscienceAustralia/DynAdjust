30void dna_io_tpb::load_tpb_file(const std::string& tpb_filename, v_string_v_doubledouble_pair& global_plates)
31{
32 // From the standard:
33 // PB2002_plates.dig.For some applications it is necessary to
34 // represent plates by closed outlines.They include computing the
35 // areas of plates, determining which plate a given point lies
36 // within, and mapping plates as regions of contrasting color.For
37 // such applications, file PB2002_plates.dig is provided.It
38 // contains 52 segments, each titled with the two - letter identifier
39 // of a plate.Each segment is a closed curve outlining that plate
40 // in the counterclockwise direction(as seen from outside the
41 // Earth).The last point in the segment is identical to the first
42 // point.Because each plate boundary necessarily appears twice in
43 // this file, it is about twice as large as the first.
44 //
45 // Tectonic plate boundary file structure is as follows.
46 // Note - coordinates are in degrees and are given in a counterclockwise
47 // direction. Since reftran uses the boost geometry package to interpolate
48 // which plate a point lies in, the polygon node coordinates will need
49 // to be reversed and converted to decimal degrees.
50 //
51 // <plate identifier>
52 // <long,lat>
53 // <long,lat>
54 // ...
55 // <long,lat>
56 // *** end of line segment ***
57 // <plate identifier>
58 // <long,lat>
59 // <long,lat>
60 // ...
61 // <long,lat>
62 // *** end of line segment ***
63
64 std::ifstream tpb_file;
65 std::stringstream ss;
66 ss << "load_tpb_file(): An error was encountered when opening " << tpb_filename << "." << std::endl;
67
68 try {
69 // open ascii plate boundaries file. Throws runtime_error on failure.
70 file_opener(tpb_file, tpb_filename, std::ios::in, ascii, true);
71 }
72 catch (const std::runtime_error& e) {
73 ss << e.what();
74 throw boost::enable_current_exception(std::runtime_error(ss.str()));
75 }
76 catch (...) {
77 throw boost::enable_current_exception(std::runtime_error(ss.str()));
78 }
79
80 ss.str("");
81 ss << "load_tpb_file(): An error was encountered when reading from " << tpb_filename << "." << std::endl;
82
83 v_doubledouble_pair plate_nodes;
84 std::string plate_name, node_coordinates;
85 vstring coordinates;
86
87 try {
88
89 global_plates.clear();
90
91 while (!tpb_file.eof()) // while EOF not found
92 {
93 // get the plate identifier
94 getline(tpb_file, plate_name);
95
96 // blank or whitespace?
97 if (trimstr(plate_name).empty())
98 continue;
99
100 // Ignore lines with comments
101 if ((plate_name.compare(0, 1, "*") == 0) &&
102 !boost::iequals(plate_name.substr(0, 7), "*** end"))
103 continue;
104
105 str_toupper<int>(plate_name);
106
107 // get the next set of plate coordinates
108 getline(tpb_file, node_coordinates);
109
110 while (!boost::iequals(node_coordinates.substr(0, 7), "*** end"))
111 {
112 // Extract coordinates from comma delimited string
113 SplitDelimitedString<std::string>(
114 node_coordinates, // the comma delimited string
115 std::string(","), // the delimiter
116 &coordinates); // the respective values
117
118 plate_nodes.push_back(doubledouble_pair(
119 DoubleFromString<double>(coordinates.at(0)),
120 DoubleFromString<double>(coordinates.at(1))));
121
122 // get the plate identifier
123 getline(tpb_file, node_coordinates);
124 }
125
126 global_plates.push_back(string_v_doubledouble_pair(
127 plate_name, plate_nodes));
128
129 plate_nodes.clear();
130 }
131
132 tpb_file.close();
133 }
134 catch (const std::ios_base::failure& f) {
135 if (tpb_file.eof())
136 {
137 tpb_file.close();
138 return;
139 }
140 ss << f.what();
141 throw boost::enable_current_exception(std::runtime_error(ss.str()));
142 }
143 catch (const std::runtime_error& e) {
144 ss << e.what();
145 throw boost::enable_current_exception(std::runtime_error(ss.str()));
146 }
147 catch (...) {
148 if (tpb_file.eof())
149 {
150 tpb_file.close();
151 return;
152 }
153 throw boost::enable_current_exception(std::runtime_error(ss.str()));
154 }
155
156 return;
157}

159void dna_io_tpb::load_tpp_file(const std::string& tpp_filename, v_plate_motion_eulers& plate_pole_parameters)
160{
161 // Tectonic pole parameters file structure is as follows.
162 // Note - coordinates are in degrees.
163 //
164 // <plate identifier> <latitude> <longitue> <pole rotation rate> <pole parameters author>
165 //
166
167 std::ifstream tpp_file;
168 std::stringstream ss;
169 ss << "load_tpp_file(): An error was encountered when opening " << tpp_filename << "." << std::endl;
170
171 try {
172 // open ascii pole parameters file. Throws runtime_error on failure.
173 file_opener(tpp_file, tpp_filename, std::ios::in, ascii, true);
174 }
175 catch (const std::runtime_error& e) {
176 ss << e.what();
177 throw boost::enable_current_exception(std::runtime_error(ss.str()));
178 }
179 catch (...) {
180 throw boost::enable_current_exception(std::runtime_error(ss.str()));
181 }
182
183 ss.str("");
184 ss << "load_tpp_file(): An error was encountered when reading from " << tpp_filename << "." << std::endl;
185
186 plate_motion_euler pmm;
187 std::string record;
188
189 try {
190
191 plate_pole_parameters.clear();
192
193 while (!tpp_file.eof()) // while EOF not found
194 {
195 // get the plate parameters record
196 getline(tpp_file, record);
197
198 // blank or whitespace?
199 if (trimstr(record).empty())
200 continue;
201
202 // Ignore lines with comments
203 if (record.compare(0, 1, "*") == 0)
204 continue;
205
206 // Extract plate name
207 pmm.plate_name = record.substr(0, 2);
208
209 // Extract coordinates from comma delimited string
210 pmm.pole_latitude = DoubleFromString<double>(trimstr(record.substr(3, 10)));
211 pmm.pole_longitude = DoubleFromString<double>(trimstr(record.substr(12, 20)));
212
213 // extract pole rotation rate
214 pmm.pole_rotation_rate = DoubleFromString<double>(trimstr(record.substr(22, 29)));
215
216 // extract plate motion model author
217 pmm.pole_param_author = record.substr(32);
218
219 plate_pole_parameters.push_back(pmm);
220 }
221
222 tpp_file.close();
223 }
224 catch (const std::ios_base::failure& f) {
225 if (tpp_file.eof())
226 {
227 tpp_file.close();
228 return;
229 }
230 ss << f.what();
231 throw boost::enable_current_exception(std::runtime_error(ss.str()));
232 }
233 catch (const std::runtime_error& e) {
234 ss << e.what();
235 throw boost::enable_current_exception(std::runtime_error(ss.str()));
236 }
237 catch (...) {
238 if (tpp_file.eof())
239 {
240 tpp_file.close();
241 return;
242 }
243 throw boost::enable_current_exception(std::runtime_error(ss.str()));
244 }
245
246 return;
247}

249bool dna_io_tpb::validate_plate_files(v_string_v_doubledouble_pair& global_plates,
250 v_plate_motion_eulers& plate_pole_parameters, std::string& message)
251{
252 if (plate_pole_parameters.size() != global_plates.size())
253 {
254 message = "There number of plates in the plate boundaries file and the pole parameters file are not equal.";
255 return false;
256 }
257
258 //for (it_plate_motion_euler i= plate_pole_parameters.begin(), it_v_string_v_doubledouble_pair j = global_plates.begin();
259 // i!=plate_pole_parameters.end(), j!=global_plates.end();
260 // ++i, ++j)
261
262 it_v_string_v_doubledouble_pair j = global_plates.begin();
263
264 for (it_plate_motion_euler i = plate_pole_parameters.begin();
265 i != plate_pole_parameters.end() && j != global_plates.end();
266 ++i, ++j)
267 {
268 if (!boost::equals(i->plate_name, j->first))
269 {
270 message = "There is a mismatch between the plates in the plate boundaries file and the pole parameters file.";
271 return false;
272 }
273 }
274
275 return true;
276}
