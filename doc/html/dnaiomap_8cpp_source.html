34void dna_io_map::load_map_file(const std::string& map_filename, pv_string_uint32_pair stnsMap)
35{
36 std::ifstream map_file;
37 std::stringstream ss;
38 ss << "load_map_file(): An error was encountered when opening " << map_filename << "." << std::endl;
39
40 try {
41 // open stations map file. Throws runtime_error on failure.
42 file_opener(map_file, map_filename, std::ios::in | std::ios::binary, binary, true);
43 }
44 catch (const std::runtime_error& e) {
45 ss << e.what();
46 throw boost::enable_current_exception(std::runtime_error(ss.str()));
47 }
48 catch (...) {
49 throw boost::enable_current_exception(std::runtime_error(ss.str()));
50 }
51
52 ss.str("");
53 ss << "load_map_file(): An error was encountered when reading from " << map_filename << "." << std::endl;
54
55 stnsMap->clear();
56 UINT32 mapsize;
57 string_uint32_pair stnID;
58 char stationName[STN_NAME_WIDTH]; // 56 characters
59
60 try {
61 // read the file information
62 readFileInfo(map_file);
63
64 // read the number of records
65 map_file.read(reinterpret_cast<char *>(&mapsize), sizeof(UINT32));
66 stnsMap->reserve(mapsize);
67
68 // read the records
69 for (UINT32 i=0; i<mapsize; i++)
70 {
71 map_file.read(const_cast<char *>(stationName), sizeof(stationName));
72 stnID.first = stationName;
73 map_file.read(reinterpret_cast<char *>(&stnID.second), sizeof(UINT32));
74 stnsMap->push_back(stnID);
75 }
76 }
77 catch (const std::ios_base::failure& f) {
78 ss << f.what();
79 throw boost::enable_current_exception(std::runtime_error(ss.str()));
80 }
81 catch (const std::runtime_error& e) {
82 ss << e.what();
83 throw boost::enable_current_exception(std::runtime_error(ss.str()));
84 }
85 catch (...) {
86 throw boost::enable_current_exception(std::runtime_error(ss.str()));
87 }
88
89 map_file.close();
90
91 if (stnsMap->size() < mapsize)
92 throw boost::enable_current_exception(
93 std::runtime_error("load_map_file(): Could not allocate sufficient memory for the Station map."));
94}

96void dna_io_map::write_map_file(const std::string& map_filename, pv_string_uint32_pair stnsMap)
97{
98 std::ofstream map_file;
99 std::stringstream ss;
100 ss << "write_map_file(): An error was encountered when opening " << map_filename << "." << std::endl;
101
102 try {
103 // Open station map file. Throws runtime_error on failure.
104 file_opener(map_file, map_filename,
105 std::ios::out | std::ios::binary, binary);
106 }
107 catch (const std::runtime_error& e) {
108 ss << e.what();
109 throw boost::enable_current_exception(std::runtime_error(ss.str()));
110 }
111 catch (...) {
112 throw boost::enable_current_exception(std::runtime_error(ss.str()));
113 }
114
115 ss.str("");
116 ss << "write_map_file(): An error was encountered when writing to " << map_filename << "." << std::endl;
117
118 v_string_uint32_pair::const_iterator _it_stnmap(stnsMap->begin());
119 UINT32 mapval(static_cast<UINT32>(stnsMap->size()));
120 try {
121 // write the file information
122 writeFileInfo(map_file);
123
124 // write the data
125 map_file.write(reinterpret_cast<char *>(&mapval), sizeof(UINT32));
126 char stationName[STN_NAME_WIDTH];
127 memset(stationName, '\0', sizeof(stationName));
128 for (; _it_stnmap!=stnsMap->end(); ++_it_stnmap)
129 {
130 strcpy(stationName, _it_stnmap->first.c_str());
131 mapval = _it_stnmap->second;
132 map_file.write(stationName, sizeof(stationName));
133 map_file.write(reinterpret_cast<char *>(&mapval), sizeof(UINT32));
134 }
135 }
136 catch (const std::ios_base::failure& f) {
137 ss << f.what();
138 throw boost::enable_current_exception(std::runtime_error(ss.str()));
139 }
140 catch (const std::runtime_error& e) {
141 ss << e.what();
142 throw boost::enable_current_exception(std::runtime_error(ss.str()));
143 }
144 catch (...) {
145 throw boost::enable_current_exception(std::runtime_error(ss.str()));
146 }
147
148 map_file.close();
149}

152void dna_io_map::write_map_file_txt(const std::string& map_filename, pv_string_uint32_pair stnsMap)
153{
154 std::ofstream map_file;
155 std::stringstream ss;
156 ss << "write_map_file_txt(): An error was encountered when opening " << map_filename << "." << std::endl;
157
158 try {
159 // Create station map text file. Throws runtime_error on failure.
160 file_opener(map_file, map_filename);
161 }
162 catch (const std::runtime_error& e) {
163 ss << e.what();
164 throw boost::enable_current_exception(std::runtime_error(ss.str()));
165 }
166 catch (...) {
167 throw boost::enable_current_exception(std::runtime_error(ss.str()));
168 }
169
170 ss.str("");
171 ss << "write_map_file_txt(): An error was encountered when writing to " << map_filename << "." << std::endl;
172
173 try {
174 std::stringstream ss_map;
175 ss_map << stnsMap->size() << " stations";
176 map_file << std::left << std::setw(STATION) << ss_map.str();
177 map_file << std::setw(HEADER_20) << std::right << "Stn. index" << std::endl;
178 v_string_uint32_pair::const_iterator _it_stnmap(stnsMap->begin());
179 for (_it_stnmap=stnsMap->begin(); _it_stnmap!=stnsMap->end(); ++_it_stnmap)
180 {
181 map_file << std::setw(STATION) << std::left <<
182 _it_stnmap->first.c_str() <<
183 std::setw(HEADER_20) << std::right << _it_stnmap->second << std::endl;
184 }
185 }
186 catch (const std::ios_base::failure& f) {
187 ss << f.what();
188 throw boost::enable_current_exception(std::runtime_error(ss.str()));
189 }
190 catch (const std::runtime_error& e) {
191 ss << e.what();
192 throw boost::enable_current_exception(std::runtime_error(ss.str()));
193 }
194 catch (...) {
195 throw boost::enable_current_exception(std::runtime_error(ss.str()));
196 }
197
198 map_file.close();
199}
