165// Custom pair type to map the location of stations in blocks
166// Since every station in the network will appear in this list,
167// valid_station will be used to determine whether this station
168// is used or not.
169template <class T=UINT32, class U=bool>
170struct stn_block_map_t
171{
172 T block_no;
173 U first_appearance_fwd;
174 U first_appearance_rev;
175 U valid_stn;
176
177 stn_block_map_t()
178 : block_no(0)
179 , first_appearance_fwd(false), first_appearance_rev(false)
180 , valid_stn(false) {}
181 stn_block_map_t(const T& block, const U fwd, const U rev, const U validity)
182 : block_no(block)
183 , first_appearance_fwd(fwd), first_appearance_rev(rev)
184 , valid_stn(validity) {}
185 stn_block_map_t (const stn_block_map_t<T> &p)
186 : block_no(p.block_no)
187 , first_appearance_fwd(p.first_appearance_fwd)
188 , first_appearance_rev(p.first_appearance_rev)
189 , valid_stn(p.valid_stn) {}
190 void firstAppearanceFwd(const T& block) {
191 block_no = block;
192 first_appearance_fwd = true;
193 }
194 void firstAppearanceRev() {
195 first_appearance_rev = true;
196 }
197};

547// data struct for storing station information to binary station file
548typedef struct stn_t {
549 stn_t(const short& u=0)
550 : suppliedStationType(LLH_type_i), initialLatitude(0.), currentLatitude(0.), initialLongitude(0.), currentLongitude(0.)
551 , initialHeight(0.), currentHeight(0.), suppliedHeightRefFrame(ELLIPSOIDAL_type_i)
552 , geoidSep(0.), meridianDef(0.), verticalDef(0.), zone(u)
553 , fileOrder(0), nameOrder(0), clusterID(0), unusedStation(FALSE)
554 {
555 memset(stationName, '\0', sizeof(stationName));
556 memset(stationNameOrig, '\0', sizeof(stationNameOrig));
557 memset(stationConst, '\0', sizeof(stationConst));
558 memset(stationType, '\0', sizeof(stationType));
559 memset(description, '\0', sizeof(description));
560 // GDA2020, lat, long, height
561 memset(epsgCode, '\0', sizeof(epsgCode));
562 snprintf(epsgCode, sizeof(epsgCode), "7843");
563 memset(epoch, '\0', sizeof(epoch));
564 memset(plate, '\0', sizeof(plate));
565 }

569 char stationConst[STN_CONST_WIDTH]; // constraint: lat, long, height
570 char stationType[STN_TYPE_WIDTH]; // type: LLH, UTM, XYZ
571 UINT16 suppliedStationType; // type supplied by the user (required for adjust)
572 double initialLatitude; // initial estimate
573 double currentLatitude; // current estimate
574 double initialLongitude;
575 double currentLongitude;
576 double initialHeight; // initialHeight and currentHeight are always assumed to be
577 double currentHeight; // ellipsoidal (ELLIPSOIDAL_type_i). If the user runs geoid using
578 // the convert option, then suppliedHeightRefFrame is set to
579 // ORTHOMETRIC_type_i and currentHeight is set to (initialHeight + N).
580 UINT16 suppliedHeightRefFrame; // Used to signify which reference frame supplied height refers to
581 float geoidSep; // ellipsoid / geoid separation
582 double meridianDef; // deflection in meridian (N/S)
583 double verticalDef; // deflection in vertical (E/W)
584 short zone;
585 char description[STN_DESC_WIDTH]; // 128 characters
586 UINT32 fileOrder; // original file order

592 // if datum is dynamic, Epoch is YYYY MM DD
593 // if datum is static, Epoch is ignored

595} station_t;

647{
648 std::string source_files("");
649 bool this_file;
650
651 for (UINT32 i(0); i<vfile_meta->size(); ++i)
652 {
653 this_file= false;
654 switch (file_type)
655 {
656 case stn_data:
657 if (vfile_meta->at(i).datatype == stn_data ||
658 vfile_meta->at(i).datatype == stn_msr_data)
659 this_file= true;

662 if (vfile_meta->at(i).datatype == msr_data ||

664 this_file= true;

668 vfile_meta->at(i).datatype == msr_data ||
669 vfile_meta->at(i).datatype == stn_msr_data)
670 this_file= true;
671 break;
672 }
673
674 if (this_file)
675 {
676 source_files += vfile_meta->at(i).filename;
677 source_files += " ";
678 }
679 }
