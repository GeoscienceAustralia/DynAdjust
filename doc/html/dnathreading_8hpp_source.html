38template<typename T>
39class concurrent_block_adjustment
40{
41public:
42 concurrent_block_adjustment()
43 : forward_is_running(false), reverse_is_running(false), combine_is_running(false) {
44 }
45 virtual ~concurrent_block_adjustment() {}
46
47 inline void resize_forward_run(const T size) {
48 // a lock is not needed since this function is called
49 // prior to creating threads
50 //lock_guard<mutex> lock_fwd(forward_mutex);
51 v_fwd_blocks.resize(size);
52 T id(0);
53 for_each(v_fwd_blocks.begin(), v_fwd_blocks.end(),
54 [this, &id] (sequential_adj_t<T, bool>& a) {
55 a.block_index = id++;
56 });
57 }
58
59 inline void resize_reverse_run(const T size) {
60 // a lock is not needed since this function is called
61 // prior to creating threads
62 //lock_guard<mutex> lock_rev(reverse_mutex);
63 v_rev_blocks.resize(size);
64 T id(0);
65 for_each(v_rev_blocks.begin(), v_rev_blocks.end(),
66 [this, &id] (sequential_adj_t<T, bool>& a) {
67 a.block_index = id++;
68 });
69 }
70 inline void resize_runs(const T size) {
71 resize_forward_run(size);
72 resize_reverse_run(size);
73 }
74
75 inline void reset_forward_run() {
76 // a lock is not needed since this function is called
77 // prior to creating threads
78 //lock_guard<mutex> lock_fwd(forward_mutex);
79 for_each(v_fwd_blocks.begin(), v_fwd_blocks.end(),
80 boost::mem_fn(&sequential_adj::nosolution));
81 }
82
83 inline void reset_reverse_run() {
84 // a lock is not needed since this function is called
85 // prior to creating threads
86 //lock_guard<mutex> lock_rev(reverse_mutex);
87 for_each(v_rev_blocks.begin(), v_rev_blocks.end(),
88 boost::mem_fn(&sequential_adj::nosolution));
89 }
90
91 inline void reset_adjustment_runs() {
92 reset_forward_run();
93 reset_reverse_run();
94 }
95
96 inline std::string print_adjusted_forward_blocks() {
97 std::stringstream ss("");
98 boost::lock_guard<boost::mutex> lock(forward_mutex);
99 ss << "Forward blocks (" << v_fwd_blocks.size() << "):" << std::endl;
100 for (UINT32 i=0; i<v_fwd_blocks.size(); ++i)
101 if (v_fwd_blocks.at(i).adjusted)
102 ss << v_fwd_blocks.at(i).block_index << " ";
103 return ss.str();
104 }
105
106 inline std::string print_adjusted_reverse_blocks() {
107 boost::lock_guard<boost::mutex> lock(reverse_mutex);
108 std::stringstream ss;
109 ss << "Reverse blocks (" << v_fwd_blocks.size() << "):" << std::endl;
110 for (UINT32 i=0; i<v_rev_blocks.size(); ++i)
111 if (v_rev_blocks.at(i).adjusted)
112 ss << v_rev_blocks.at(i).block_index << " ";
113 return ss.str();
114 }
115
116 inline std::string print_adjusted_blocks() {
117 std::stringstream ss;
118 ss << std::endl <<
119 print_adjusted_forward_blocks() << std::endl <<
120 print_adjusted_reverse_blocks() << std::endl;
121 return ss.str();
122 }

124 // Set started state to true
125 inline void combine_run_started() { // combine run
126 boost::lock_guard<boost::mutex> lock(cmb_state_mutex);
127 combine_is_running = true;
128 }
129 inline void forward_run_started() { // forward run
130 boost::lock_guard<boost::mutex> lock(fwd_state_mutex);
131 forward_is_running = true;
132 }
133 inline void reverse_run_started() { // reverse run
134 boost::lock_guard<boost::mutex> lock(rev_state_mutex);
135 reverse_is_running = true;
136 }
137
138 // Set finished state to true
139 inline void combine_run_stopped() { // combine run
140 boost::lock_guard<boost::mutex> lock(cmb_state_mutex);
141 combine_is_running = false;
142 }
143 inline void forward_run_finished() { // forward run
144 boost::lock_guard<boost::mutex> lock(fwd_state_mutex);
145 forward_is_running = false;
146 }
147 inline void reverse_run_finished() { // reverse run
148 boost::lock_guard<boost::mutex> lock(rev_state_mutex);
149 reverse_is_running = false;
150 }
151
152 // Get running state
153 inline bool get_combine_state() { // combine run
154 boost::lock_guard<boost::mutex> lock(cmb_state_mutex);
155 return combine_is_running;
156 }
157 inline bool get_forward_state() { // forward run
158 boost::lock_guard<boost::mutex> lock(fwd_state_mutex);
159 return forward_is_running;
160 }
161 inline bool get_reverse_state() { // reverse run
162 boost::lock_guard<boost::mutex> lock(rev_state_mutex);
163 return reverse_is_running;
164 }
165
166 inline bool is_forward_finished() { // forward run
167 boost::lock_guard<boost::mutex> lock(fwd_state_mutex);
168 return forward_is_running == false;
169 }
170 inline bool is_reverse_finished() { // reverse run
171 boost::lock_guard<boost::mutex> lock(rev_state_mutex);
172 return reverse_is_running == false;
173 }
174
175 // It is dangerous to aquire a lock if one is already held
176 // (e.g. lock fwd_state_mutex, then lock rev_state_mutex)
177 // so use unlock for forward before locking reverse
178 inline bool any_still_running() {
179 if (get_forward_state())
180 return true;
181 boost::lock_guard<boost::mutex> lock_r(rev_state_mutex);
182 return reverse_is_running;
183 }
184 inline bool finished_all_runs() {
185 if (get_forward_state())
186 return false;
187 boost::lock_guard<boost::mutex> lock_r(rev_state_mutex);
188 return !reverse_is_running;
189 }
190
191 inline void set_forward_block_adjusted(const T& block) {
192 boost::lock_guard<boost::mutex> lock(forward_mutex);
193 v_fwd_blocks.at(block).solution();
194 }
195 inline void set_reverse_block_adjusted(const T& block) {
196 boost::lock_guard<boost::mutex> lock(reverse_mutex);
197 v_rev_blocks.at(block).solution();
198 }
199
200 inline bool forward_block_adjusted(const T& block) {
201 boost::lock_guard<boost::mutex> lock(forward_mutex);
202 return v_fwd_blocks.at(block).adjusted;
203 }
204 inline bool reverse_block_adjusted(const T& block) {
205 boost::lock_guard<boost::mutex> lock(reverse_mutex);
206 return v_rev_blocks.at(block).adjusted;
207 }
208private:
209 bool forward_is_running;
210 bool reverse_is_running;
211 bool combine_is_running;
212
213 std::vector< sequential_adj_t<T, bool> > v_fwd_blocks;
214 std::vector< sequential_adj_t<T, bool> > v_rev_blocks;
215
216 mutable boost::mutex forward_mutex, reverse_mutex, combine_mutex;
217 mutable boost::mutex fwd_state_mutex, rev_state_mutex, cmb_state_mutex;
218
219};

222template<typename T>
223class concurrent_queue
224{
225public:
226 concurrent_queue()
227 : more_blocks_coming_(true) {}
228 virtual ~concurrent_queue() {}
229
230 inline void push_data(const std::vector<T>& data)
231 {
232 boost::lock_guard<boost::mutex> ql(queue_mutex);
233 // push a vector of data elements
234 for_each(
235 data.begin(),
236 data.end(),
237 [this](const T& t){ // use lambda expression
238 this->the_queue.push(t);
239 }
240 );
241 }
242
243 inline void push_and_notify(const T data)
244 {
245 boost::lock_guard<boost::mutex> ql(queue_mutex);
246 the_queue.push(data);
247 // notify all (potential) threads waiting on the_queue
248 the_condition.notify_all();
249 }
250
251 inline bool not_empty() const
252 {
253 boost::lock_guard<boost::mutex> ql(queue_mutex);
254 return !the_queue.empty();
255 }
256
257 inline bool is_empty() const
258 {
259 boost::lock_guard<boost::mutex> ql(queue_mutex);
260 return the_queue.empty();
261 }
262
263 inline bool front_and_pop(T& popped_value)
264 {
265 boost::lock_guard<boost::mutex> ql(queue_mutex);
266 if (the_queue.empty())
267 return false;
268 popped_value = the_queue.front();
269 the_queue.pop();
270 return true;
271 }
272
273 inline void notify_all()
274 {
275 the_condition.notify_all();
276 }
277
278 inline bool wait_front_and_pop(T& popped_value)
279 {
280 // wait if the queue is empty
281 boost::unique_lock<boost::mutex> ql(queue_mutex);
282 the_condition.wait(ql, [this] {return !this->the_queue.empty();});
283 if (the_queue.empty())
284 return false;
285 popped_value = the_queue.front();
286 the_queue.pop();
287 return true;
288 }
289
290 inline void wait_if_queue_is_empty()
291 {
292 boost::unique_lock<boost::mutex> ql(queue_mutex);
293 the_condition.wait(ql, [this] { return !this->the_queue.empty();});
294 }
295
296 inline void queue_exhausted()
297 {
298 boost::lock_guard<boost::mutex> ql(exhausted_mutex);
299 more_blocks_coming_ = false;
300 the_condition.notify_all();
301 }
302
303 inline void reset_blocks_coming()
304 {
305 boost::lock_guard<boost::mutex> el(exhausted_mutex);
306 more_blocks_coming_ = true;
307 }
308
309 inline bool more_blocks_coming()
310 {
311 boost::lock_guard<boost::mutex> el(exhausted_mutex);
312 return more_blocks_coming_;
313 }
314
315 inline bool is_queue_exhausted()
316 {
317 // First, check that there are no more blocks coming
318 if (more_blocks_coming())
319 return false;
320
321 // Second, is there any data in the queue? If so, then
322 // that's the tell-tale sign we've got more work to do.
323 return is_empty();
324 }
325
326
327
328private:
329 mutable boost::mutex queue_mutex;
330 mutable boost::mutex exhausted_mutex;
331 std::queue<T> the_queue;
332 boost::condition_variable the_condition;
333
334 bool more_blocks_coming_;
335};
