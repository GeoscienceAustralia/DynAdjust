110void CDnaProjectFile::LoadProjectFile()
111{
112 // load project file using p.g.project_file
113 std::ifstream dnaproj_file;
114
115 std::stringstream err_msg;
116 err_msg << "LoadProjectFile(): An error was encountered when opening " <<
117 settings_.g.project_file << "." << std::endl;
118
119 try {
120 // create binary aml file. Throws runtime_error on failure.
121 file_opener(dnaproj_file, settings_.g.project_file,
122 std::ios::in, ascii, true);
123 }
124 catch (const std::runtime_error& e) {
125 err_msg << e.what();
126 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
127 }
128 catch (...) {
129 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
130 }
131
132 err_msg.str("");
133 err_msg << "LoadProjectFile(): An error was encountered when loading from " <<
134 settings_.g.project_file << "." << std::endl;
135
136 std::string str, line, var, val;
137 std::stringstream ss;
138
139 settingMode mSetting(unknownSetting);
140
141 while (!dnaproj_file.eof())
142 {
143 try {
144 getline(dnaproj_file, line, '\n');
145 }
146 catch (std::ifstream::failure& f)
147 {
148 if (dnaproj_file.eof())
149 break;
150
151 err_msg << f.what() << std::endl;
152 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
153 }
154 catch (...)
155 {
156 if (dnaproj_file.eof())
157 break;
158
159 err_msg << "Could not read file." << std::endl;
160 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
161 }
162
163
164 // Blank line?
165 if (line.empty())
166 continue;
167
168 // Line too short?
169 if (line.length() <= PRINT_VAR_PAD)
170 continue;
171
172 // #variables
173 if (line.find(section_variables) != std::string::npos)
174 {
175 mSetting = switchSetting;
176 continue;
177 }
178
179 // #general
180 if (line.find(section_general) != std::string::npos)
181 {
182 mSetting = generalSetting;
183 continue;
184 }
185
186 // #import
187 if (line.find(section_import) != std::string::npos)
188 {
189 mSetting = importSetting;
190 continue;
191 }
192
193 // #reftran
194 if (line.find(section_reftran) != std::string::npos)
195 {
196 mSetting = reftranSetting;
197 continue;
198 }
199
200 // #general
201 if (line.find(section_geoid) != std::string::npos)
202 {
203 mSetting = geoidSetting;
204 continue;
205 }
206
207 // #segment
208 if (line.find(section_segment) != std::string::npos)
209 {
210 mSetting = segmentSetting;
211 continue;
212 }
213
214 // #adjust
215 if (line.find(section_adjust) != std::string::npos)
216 {
217 mSetting = adjustSetting;
218 continue;
219 }
220
221 // #output
222 if (line.find(section_output) != std::string::npos)
223 {
224 mSetting = outputSetting;
225 continue;
226 }
227
228 // #plot
229 if (line.find(section_plot) != std::string::npos)
230 {
231 mSetting = plotSetting;
232 continue;
233 }
234
235 // #display
236 if (line.find(section_display) != std::string::npos)
237 {
238 mSetting = displaySetting;
239 continue;
240 }
241
242 // Line?
243 if (line.find("----------") != std::string::npos)
244 continue;
245
246 // Now get the variables and their value relating to the setting
247 var = trimstr(line.substr(0, PRINT_VAR_PAD));
248 if (line.length() > PRINT_VAR_PAD)
249 val = trimstr(line.substr(PRINT_VAR_PAD));
250 else
251 val.clear();
252
253 try {
254 switch (mSetting)
255 {
256 case generalSetting:
257 LoadSettingGeneral(mSetting, var, val);
258 break;
259 case importSetting:
260 LoadSettingImport(mSetting, var, val);
261 break;
262 case reftranSetting:
263 LoadSettingReftran(var, val);
264 break;
265 case geoidSetting:
266 LoadSettingGeoid(var, val);
267 break;
268 case segmentSetting:
269 LoadSettingSegment(var, val);
270 break;
271 case adjustSetting:
272 LoadSettingAdjust(var, val);
273 LoadSettingOutput(var, val);
274 break;
275 case outputSetting:
276 LoadSettingOutput(var, val);
277 break;
278 case plotSetting:
279 LoadSettingOutput(var, val);
280 break;
281 case displaySetting:
282 break;
283 default:
284 break;
285 }
286 }
287 catch (const std::runtime_error& e) {
288 err_msg << e.what();
289 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
290 }
291 catch (...) {
292 throw boost::enable_current_exception(std::runtime_error(err_msg.str()));
293 }
294 }
295
296 dnaproj_file.clear();
297 dnaproj_file.close();
298}

310void CDnaProjectFile::InitialiseImportSettings()
311{
312 InitialiseGeneralSettings();
313
314 if (settings_.i.input_files.empty() &&
315 settings_.i.seg_file.empty())
316 {
317 std::stringstream ss;
318 ss << "Nothing to do - the project file does not list any files to import. " << std::endl << std::endl;
319 throw boost::enable_current_exception(std::runtime_error(ss.str()));
320 }
321
322 std::string firstPart(settings_.g.output_folder + FOLDER_SLASH + settings_.g.network_name + ".");
323
324 if (!settings_.i.bst_file.empty())
325 {
326 if (!boost::filesystem::exists(settings_.i.bst_file))
327 // Does the file exist? No.
328 // Assume it is a filename contained in the input folder.
329 // import will throw an exception if it cannot be found.
330 settings_.i.bst_file = formPath<std::string>(settings_.g.input_folder,
331 leafStr<std::string>(settings_.i.bst_file));
332 }
333 else
334 settings_.i.bst_file = firstPart + "bst"; // binary stations file
335
336 if (!settings_.i.bms_file.empty())
337 {
338 if (!boost::filesystem::exists(settings_.i.bms_file))
339 // Does the file exist? No.
340 // Assume it is a filename contained in the input folder.
341 // import will throw an exception if it cannot be found.
342 settings_.i.bms_file = formPath<std::string>(settings_.g.input_folder,
343 leafStr<std::string>(settings_.i.bms_file));
344 }
345 else
346 settings_.i.bms_file = firstPart + "bms"; // binary measurements file
347
348 settings_.i.asl_file = firstPart + "asl"; // associated stations list
349 settings_.i.aml_file = firstPart + "aml"; // associated measurements list
350 settings_.i.map_file = firstPart + "map"; // station names map
351 settings_.i.dst_file = firstPart + "dst"; // fuplicate stations
352 settings_.i.dms_file = firstPart + "dms"; // duplicate measurements
353 settings_.i.imp_file = firstPart + "imp"; // log
354
355 if (settings_.o._msr_to_stn == 1)
356 settings_.o._m2s_file = firstPart + "m2s"; // measurement to stations table
357
358 if (!settings_.i.seg_file.empty())
359 {
360 if (!boost::filesystem::exists(settings_.i.seg_file))
361 // Does the file exist? No.
362 // Assume it is a filename contained in the input folder.
363 // import will throw an exception if it cannot be found.
364 settings_.i.seg_file = formPath<std::string>(settings_.g.input_folder,
365 leafStr<std::string>(settings_.i.seg_file));
366 }
367
369 // GNSS scaling
370 if (fabs(settings_.i.pscale - 1.0) > PRECISION_1E5 ||
371 fabs(settings_.i.lscale - 1.0) > PRECISION_1E5 ||
372 fabs(settings_.i.hscale - 1.0) > PRECISION_1E5 ||
373 fabs(settings_.i.vscale - 1.0) > PRECISION_1E5 ||
374 !settings_.i.scalar_file.empty())
375 settings_.i.apply_scaling = 1;
376
377 if (!settings_.i.scalar_file.empty())
378 {
379 if (!boost::filesystem::exists(settings_.i.scalar_file))
380 // Does the file exist? No.
381 // Assume it is a filename contained in the input folder.
382 // import will throw an exception if it cannot be found.
383 settings_.i.scalar_file = formPath<std::string>(settings_.g.input_folder,
384 leafStr<std::string>(settings_.i.scalar_file));
385 }
386
387 // Create file name based on the provided block
388 std::string fileName(settings_.g.network_name);
389 if (settings_.i.import_block)
390 {
391 std::stringstream blk("");
392 blk << ".block-" << settings_.i.import_block_number;
393 fileName += blk.str();
394 }
395
396 if (settings_.i.export_dynaml)
397 {
398 if (settings_.i.export_single_xml_file)
399 settings_.i.xml_outfile = formPath<std::string>(settings_.g.output_folder,
400 fileName, "xml");
401 else
402 {
403 settings_.i.xml_stnfile = formPath<std::string>(settings_.g.output_folder,
404 fileName + "stn", "xml");
405 settings_.i.xml_msrfile = formPath<std::string>(settings_.g.output_folder,
406 fileName + "msr", "xml");
407 }
408 }
409
410 if (settings_.i.export_dna_files)
411 {
412 settings_.i.dna_stnfile = formPath<std::string>(settings_.g.output_folder,
413 fileName, "stn");
414 settings_.i.dna_msrfile = formPath<std::string>(settings_.g.output_folder,
415 fileName, "msr");
416 }
417
418 if (settings_.i.simulate_measurements)
419 settings_.i.simulate_msrfile = formPath<std::string>(settings_.g.output_folder,
420 settings_.g.network_name, "simulated.msr");
421
422}

541 case PhasedMode:

620void CDnaProjectFile::LoadSettingGeneral(const settingMode mSetting, const std::string& var, std::string& val)
621{
622 // general settings
623 if (boost::iequals(var, QUIET))
624 {
625 if (val.empty())
626 return;
627 settings_.g.quiet = yesno_uint<UINT16, std::string>(val);
628 }
629 else if (boost::iequals(var, VERBOSE))
630 {
631 if (val.empty())
632 return;
633 settings_.g.verbose = boost::lexical_cast<UINT16, std::string>(val);
634 }
635 else if (boost::iequals(var, PROJECT_FILE))
636 {
637 if (val.empty())
638 return;
639 settings_.g.project_file = val;
640 }
641 else if (boost::iequals(var, DYNADJUST_LOG_FILE))
642 {
643 if (val.empty())
644 return;
645 settings_.g.log_file = val;
646 }
647 else if (boost::iequals(var, NETWORK_NAME))
648 {
649 if (val.empty())
650 return;
651 string_string_pair t;
652 switch (mSetting)
653 {
654 case switchSetting:
655 t.first = var.c_str();
656 t.second = val.substr(0,4).c_str();
657 t.second = trimstr(t.second);
658 if (t.second.substr(0,2) == "${" && t.second.at(3) == '}')
659 {
660 settings_.g.variables.push_back(t);
661 std::sort(settings_.g.variables.begin(), settings_.g.variables.end(), PairCompareFirst<std::string, std::string>());
662 }
663 break;
664 case generalSetting:
665 settings_.g.network_name = val;
666 break;
667 default:
668 break;
669 }
670 }
671 else if (boost::iequals(var, INPUT_FOLDER))
672 {
673 if (val.empty())
674 return;
675 settings_.g.input_folder = val;
676 }
677 else if (boost::iequals(var, OUTPUT_FOLDER))
678 {
679 if (val.empty())
680 return;
681 settings_.g.output_folder = val;
682 }
683

686void CDnaProjectFile::LoadSettingImport(const settingMode mSetting, const std::string& var, std::string& val)
687{
688 // Import settings
689 if (boost::iequals(var, IMPORT_FILE))
690 {
691 if (val.empty())
692 return;
693 settings_.i.input_files.push_back(formPath<std::string>(settings_.g.input_folder, val));
694 }
695 else if (boost::iequals(var, REFERENCE_FRAME))
696 {
697 if (val.empty())
698 return;
699 settings_.i.reference_frame = val;
700 }
701 else if (boost::iequals(var, EPOCH))
702 {
703 if (val.empty())
704 return;
705 settings_.i.epoch = val;
706 }
707 else if (boost::iequals(var, OVERRIDE_INPUT_FRAME))
708 {
709 if (val.empty())
710 return;
711 settings_.i.override_input_rfame = yesno_uint<UINT16, std::string>(val);;
712 }
713 else if (boost::iequals(var, IMPORT_GEO_FILE))
714 {
715 if (val.empty())
716 return;
717 settings_.i.geo_file = formPath<std::string>(settings_.g.input_folder, val);
718 settings_.i.import_geo_file = 1;
719 }
720 else if (boost::iequals(var, BIN_STN_FILE))
721 {
722 string_string_pair t;
723 switch (mSetting)
724 {
725 case switchSetting:
726 if (val.empty())
727 return;
728 t.first = var.c_str();
729 t.second = val.substr(0,4).c_str();
730 t.second = trimstr(t.second);
731 if (t.second.substr(0,2) == "${" && t.second.at(3) == '}')
732 {
733 settings_.g.variables.push_back(t);
734 std::sort(settings_.g.variables.begin(), settings_.g.variables.end(), PairCompareFirst<std::string, std::string>());
735 }
736 break;
737 case importSetting:
738 if (val.empty())
739 // default export bst file - create in output folder
