134// data struct for storing measurement information to binary measurement file
135typedef struct msr_t {
136 msr_t()
137 : measType('\0'), measStart(0), measurementStations(1), ignore(false), station1(0), station2(0)
138 , station3(0), vectorCount1(0), vectorCount2(0), clusterID(0), fileOrder(0)
139 , term1(0.), term2(0.), term3(0.), term4(0.), scale1(1.), scale2(1.), scale3(1.), scale4(1.)
140 , measAdj(0.), measCorr(0.), measAdjPrec(0.), residualPrec(0.)
141 , NStat(0.), TStat(0.), PelzerRel(0.)
142 , preAdjCorr(0.), preAdjMeas(0.)
143 {
144 memset(coordType, '\0', sizeof(coordType));
145 // GDA94, lat, long, height
146 memset(epsgCode, '\0', sizeof(epsgCode));
147 snprintf(epsgCode, sizeof(epsgCode), DEFAULT_EPSG_S);
148 memset(epoch, '\0', sizeof(epoch));
149 }
150
151 char measType; // 'A', 'S', 'X', ... , etc.
152 char measStart; // Start of a measurement (0=start or X, 1=Y, 2=Z, 3=covX, 4=covY, 5=covZ)
153 char measurementStations; // One-, two- or three-station measurement
154 char epsgCode[7]; // epsg ID, i.e. NNNNN (where NNNNN is in the range 0-32767)
155 char epoch[12]; // date, i.e. "DD.MM.YYYY" (10 chars)
156 // if datum is dynamic, Epoch is YYYY MM DD
157 // if datum is static, Epoch is ignored
158 char coordType[4]; // "LLH", "UTM", ... , etc.
159 bool ignore;
160 UINT32 station1; // stations 1, 2 and 3 are indices to
161 UINT32 station2; // record numbers in the station
162 UINT32 station3; // record file.
163 UINT32 vectorCount1; // number of directions, GpsPoints or GpsBaselines
164 UINT32 vectorCount2; // number of covariances for GpsPoint or GpsBaseline
165 // number of non-ignored directions
166 UINT32 clusterID; // cluster ID (which cluster this measurement belongs to)
167 UINT32 fileOrder; // original file order
168 double term1; // measurement, X, Y, Z, dX, dY, dZ value
169 // direction
170 double term2; // measurement, XX, XY or XZ variance
171 // direction variance
172 double term3; // instrument height, YY or YZ variance
173 double term4; // target height or ZZ variance
174 double scale1; // phi, n or X scalar
175 // derived angle corrected for deflection of the vertical
176 double scale2; // lambda, e or Y scalar
177 // derived angle variance
178 double scale3; // height, up or Z scalar
179 // derived angle covariance
180 double scale4; // matrix scalar
181 double measAdj;
182 double measCorr;
183 double measAdjPrec;
184 double residualPrec;
185 double NStat;
186 double TStat;
187 double PelzerRel;
188 double preAdjCorr;
189 double preAdjMeas;
190} measurement_t;
