148// Expects yy as two digit year, and doy as three digit day of year
149std::stringstream dna_io_snx::parse_date_from_yy_doy(const UINT32& yy, const UINT32& doy, DATE_FORMAT_TYPE date_type, const std::string& separator)
150{
151 std::stringstream ss_doy, ss_year, ss_date;
152
153 // format the day of year so that boost date_input_facet is happy
154 if (doy < 100)
155 ss_doy << "0";
156 if (doy < 10)
157 ss_doy << "0";
158
159 ss_doy << doy;
160
161 // create four-character year?
162 switch (date_type)
163 {
164 case yyyy_doy:
165 case doy_yyyy:
166 // SINEX time specification:
167 // _______________________________________________________________
168 // | Time: | YY:DDD:SSSSS. "UTC" | I2.2, |
169 // | | YY = last 2 digits of the year, | 1H:,I3.3, |
170 // | | if YY <= 50 implies 21-st century, | 1H:,I5.5 |
171 // | | if YY > 50 implies 20-th century, | |
172 // | | DDD = 3-digit day in year, | |
173 // | | SSSSS = 5-digit seconds in day. | |
174 // |_______|_______________________________________|_______________|
175 if (yy <= 50)
176 ss_year << "20";
177 else //(average_year > 50)
178 ss_year << "19";
179 break;
180 default:
181 break;
182 }
183
184 if (yy < 10)
185 ss_year << "0";
186 ss_year << yy;
187
188 switch (date_type)
189 {
190 case yyyy_doy:
191 case yy_doy:
192 ss_date << ss_year.str() << separator << ss_doy.str();
193 break;
194 case doy_yy:
195 case doy_yyyy:
196 ss_date << ss_doy.str() << separator << ss_year.str();
197 break;
198 }
199
200 return ss_date;
201}

203// date is a string "YY:DDD:SSSSS"
204std::stringstream dna_io_snx::parse_date_from_string(const std::string& date_str, DATE_FORMAT_TYPE date_type, DATE_TERMINAL_TYPE terminal_type, const std::string& separator)
205{
206 // Year
207 UINT32 yy = LongFromString<UINT32>(date_str.substr(0, 2));
208
209 // Day of year (assume a colon separates the two values)
210 UINT32 doy = LongFromString<UINT32>(date_str.substr(3, 3));
211
212 if (yy == 0 && doy == 0)
213 {
214 boost::gregorian::date today(boost::gregorian::day_clock::local_day());
215 std::stringstream date_ss;
216
217 switch (terminal_type)
218 {
219 case date_from:
220 // Set a date before the advent of GPS
221 date_ss << "001 " << TIME_IMMEMORIAL;
222 return date_ss;
223 case date_to:
224 // Set a date 100 years into the future
225 date_ss << "365 " << (today.year() + 100);
226 return date_ss;
227 }
228 }
229
230 return parse_date_from_yy_doy(yy, doy, date_type, separator);
231}

233bool dna_io_snx::parse_sinex_header(std::ifstream** snx_file, CDnaDatum& datum, UINT32& lineNo)
234{
235 /*
236 %=SNX 2.00 AUS 11:182:57860 IGS 11:157:00000 11:157:86370 P 00054 0 S
237 */
238 char c = static_cast<char>((*snx_file)->peek());
239 if (c != '%')
240 {
241 // The SINEX file doesn't have a header.
242 return false;
243 }
244
245 std::string sBuf, sinex_code;
246
247 UINT32 average_year, average_doy;
248 std::stringstream ss;
249
250 lineNo++;
251 try {
252 getline((**snx_file), sBuf);
253 }
254 catch (...) {
255 ss << "parse_sinex_header(): Could not read from the SINEX file. " << std::endl << ".";
256 throw boost::enable_current_exception(std::runtime_error(ss.str()));
257 }
258
259 try {
260 // capture epoch of data, calculate average and test for year cross over
261 year_doy_Average(
262 LongFromString<UINT32>(sBuf.substr(32, 2)),
263 LongFromString<UINT32>(sBuf.substr(45, 2)),
264 LongFromString<UINT32>(sBuf.substr(35, 3)),
265 LongFromString<UINT32>(sBuf.substr(48, 3)),
266 average_year,
267 average_doy);
268
269 ss = parse_date_from_yy_doy(average_year, average_doy, doy_yyyy, std::string(" "));
270 datum.SetEpoch(dateFromStringstream_doy_year<boost::gregorian::date, std::stringstream>(ss));
271 }
272 catch (...)
273 {
274 ss << "parse_sinex_header(): Could not extract date and time information from the header record: " << std::endl << " " << sBuf << ".";
275 throw boost::enable_current_exception(std::runtime_error(ss.str()));
276 }
277
278 return true;
279}

332void dna_io_snx::format_station_names(v_discontinuity_tuple* stn_discontinuities, bool& m_discontsSortedbyName,
333 vdnaStnPtr* vStations, vdnaMsrPtr* vMeasurements)
334{
335 UINT32 site, doy, year;
336 std::string site_name;
337
338#ifdef _MSDEBUG
339 std::string date_str;
340#endif
341
342 _it_vdiscontinuity_tuple _it_discont(stn_discontinuities->end()), _it_discont_site;
343
344 boost::gregorian::date site_date;
345
346 if (!m_discontsSortedbyName)
347 {
348 std::sort(stn_discontinuities->begin(), stn_discontinuities->end(),
349 CompareSiteTuplesByName<discontinuity_tuple>());
350 m_discontsSortedbyName = true;
351 }
352 std::sort(siteOccurrence_.begin(), siteOccurrence_.end(),
353 CompareSiteTuplesByName<site_id>());
354
355 _it_discont = stn_discontinuities->begin();
356
357 for (site=0; site<siteOccurrence_.size(); ++site)
358 {
359 site_name = siteOccurrence_.at(site).site_name;
360
361 // At this point, _it_discont will point to either:
362 // 1. the first element in the vector
363 // 2. the first occurrence of discontinuity for site_name
364 // To prevent searching through the entire array again on 2. , check if
365 // _it_discont->site_name = site_name
366 if (!boost::equals(site_name, _it_discont->site_name))
367 {
368 // Save the iterator
369 _it_discont_site = _it_discont;
370
371 if ((_it_discont = binary_search_discontinuity_site(
372 _it_discont,
373 stn_discontinuities->end(),
374 site_name)) == stn_discontinuities->end())
375 {
376 // Site not found in discontinuity file, reset iterator to last saved state
377 _it_discont = _it_discont_site;
378 continue;
379 }
380 }
381
382 // Is this site a discontinuity site?
383 if (!_it_discont->discontinuity_exists)
384 continue;
385
386 // Capture the (start date of the site)
387 site_date = dateFromString_doy_year<boost::gregorian::date, std::string>(siteOccurrence_.at(site).formatted_date);
388
389#ifdef _MSDEBUG
390 //date_str = stringFromDate(_it_discont->date_start, "%y:%j:00000");
391#endif
392 _it_discont_site = _it_discont;
393
394 // find the next occurrence of this site
395 while (boost::equals(site_name, _it_discont_site->site_name))
396 {
397 // Test if the start epoch of this site is within this discontinuity window
398 if (site_date >= _it_discont_site->date_start &&
399 site_date < _it_discont_site->date_end)
400// if (siteOccurrence_.at(site).solution_id == _it_discont_site->solution_id)
401 {
402 std::stringstream ss;
403 doy = _it_discont_site->date_start.day_of_year();
404 year = _it_discont_site->date_start.year();
405 ss << siteOccurrence_.at(site).site_name << "_";
406
407 // format date using the start of the window defining the
408 // period up to which a discontinuity has been identified
409 ss << year;
410 if (doy < 100)
411 ss << "0";
412 if (doy < 10)
413 ss << "0";
414 ss << doy;
415
416 siteOccurrence_.at(site).formatted_name = ss.str();
417 //TRACE("Site %s\n", siteOccurrence_.at(site).formatted_name.c_str());
418 break;
419 }
420 _it_discont_site++;
421 }
422 }
423
424 if (siteOccurrence_.size() != vStations->size() ||
425 siteOccurrence_.size() != vMeasurements->at(0)->GetTotal())
426 {
427 std::stringstream ss;
428 ss << "format_station_names(): Inconsistent number of stations in SINEX file.";
429 throw boost::enable_current_exception(std::runtime_error(ss.str()));
430 }
431
432 _it_vsite_id_tuple _it_site(siteOccurrence_.begin());
433
434 // sort on file index (default sort)
435 std::sort(siteOccurrence_.begin(), siteOccurrence_.end(),
436 CompareSiteTuples<site_id>());
437
438 // Now update station vector
439 for_each(vStations->begin(), vStations->end(),
440 [&_it_site](const dnaStnPtr& stn) {
441 if (!boost::equals(_it_site->site_name, _it_site->formatted_name))
442 {
443 stn->SetOriginalName();
444 stn->SetName(_it_site->formatted_name);
445 }
446 _it_site++;
447 });
448
449 // re-initialise iterator
450 _it_site = siteOccurrence_.begin();
451
452 _it_vdnamsrptr _it_msr(vMeasurements->begin());
453 std::vector<CDnaGpsPoint>* vgpsPnts(_it_msr->get()->GetPoints_ptr());
454
455 // Now update measurement vector
456 for_each(vgpsPnts->begin(), vgpsPnts->end(),
457 [&_it_site](CDnaGpsPoint& pnt) {
458 if (!boost::equals(_it_site->site_name, _it_site->formatted_name))
459 pnt.SetFirst(_it_site->formatted_name);
460 _it_site++;
461 });
462}

467void dna_io_snx::parse_sinex_block(std::ifstream** snx_file, const char* sinexRec, vdnaStnPtr* vStations, PUINT32 stnCount,
468 vdnaMsrPtr* vMeasurements, PUINT32 msrCount, PUINT32 clusterID,
469 StnTally& parsestn_tally, MsrTally& parsemsr_tally, CDnaDatum& datum,
470 UINT32& fileOrder, UINT32& lineNo, UINT32& columnNo)
471{
472 // write comments
473 if (strncmp(sinexRec, "+FILE/REFERENCE", 15) == 0 ||
474 strncmp(sinexRec, "+SITE/RECEIVER", 14) == 0 ||
475 strncmp(sinexRec, "+SITE/ANTENNA", 13) == 0 ||
476 strncmp(sinexRec, "+SITE/GPS_PHASE_CENTER", 22) == 0 ||
477 strncmp(sinexRec, "+SITE/ECCENTRICITY", 18) == 0)
478 {
479 // nothing to do.
480 return;
481 }
482
483 // According to the SINEX standard:
484 // - BLOCKS start with "+" in the first column followed by standardized block labels.
485 // - Each BLOCK ends with "-" and the block label.
486 // - BLOCK data starts in column 2 or higher.
487 // - BLOCKS can be in any order, provided that they start with (+) and
488 // end with (-) BLOCK labels.
489 //
490 // Therefore, don't assume the SOLUTION/* block order is followed.
491 // However, assume SITE/ID comes first, so that discontinuities can be parsed in an
492 // intelligent way.
493
494 if (strncmp(sinexRec, "+SITE/ID", 8) == 0)
495 {
496 // OK, found general site list
497 //parse_sinex_sites(snx_file, lineNo, columnNo);
498 return;
499 }
500
501 if (strncmp(sinexRec, "+SOLUTION/EPOCHS", 16) == 0)
502 {
503 // OK, found epochs associated with stations
504 parse_sinex_epochs(snx_file, lineNo, columnNo);
505 return;
506 }
507
508 if (strncmp(sinexRec, "+SOLUTION/ESTIMATE", 18) == 0)
509 {
510 // OK, found some stations
511 parse_sinex_stn(snx_file, sinexRec, vStations, stnCount,
512 parsestn_tally, datum, fileOrder, lineNo, columnNo);
513 return;
514 }
515
516 if (strncmp(sinexRec, "+SOLUTION/MATRIX_ESTIMATE", 25) == 0)
517 {
518 // Ok, found some measurements
519 parse_sinex_msr(snx_file, sinexRec, vStations, vMeasurements, clusterID, msrCount,
520 parsemsr_tally, datum, lineNo);
521 return;
522 }

526void dna_io_snx::parse_sinex_discontinuities(std::ifstream* snx_file,
527 v_discontinuity_tuple* stn_discontinuities, bool& m_discontsSortedbyName,
528 UINT32& lineNo, UINT32& columnNo)
529{
530 std::string sBuf("+"), stn, stn_prev(""), model;
531 UINT32 file_rec(0), solution_id;
532 boost::gregorian::date discont_start_date, discont_end_date;
533
534 std::stringstream ss, discont_from, discont_to;
535
536 stn_discontinuities->clear();
537 m_discontsSortedbyName = false;
538
539 while (sBuf.at(0) != '-')
540 {
541 lineNo++;
542 getline((*snx_file), sBuf);
543
544 // A comment
545 if (sBuf.at(0) == '*')
546 continue;
547
548 if (sBuf.at(0) == '-')
549 break;
550
551 try {
552 // station
553 stn = trimstr(sBuf.substr(1, 4)); // station name
554
555 // solution ID
556 solution_id = val_uint<UINT32, std::string>(
557 trimstr(sBuf.substr(9, 4))); // solution id
558
559 // model
560 // Commonly used models for time series analysis are:
561 // "M" Models Codes for coordinates time series Analysis:
562 // P = Position
563 // V = Velocity
564 // A = Annual Periodicity
565 // S = Semi-Annual Periodicity
566 // E = Exponential Decay
567 // X = Exclude/Delete
568 model = trimstr(sBuf.substr(42, 1)); // model
569
570 // Is this a discontinuity in position? If not, continue to next
571 if (!boost::iequals(model, "P"))
572 continue;
573
574 // If there are multiple solutions, then there must be a discontinuity, but only if
575 // the model is the same. For this version of DynAdjust, work with position only (i.e. disregard velocity)
576 if (solution_id > 1)
577 containsDiscontinuities_ = true;
578
580 // Dates FROM and TO define the temporal extents of time series divided by discontinuities.
581 // To understand how to apply these dates, the following rules apply:
582 // - If FROM and TO both equal 00:000:00000, there is no known (or recorded) discontinuity
583 // at this site (in which case, this site does not need renaming)
584 // - If FROM equals 00:000:00000 and TO is a realistic date, then this is the first leg
585 // in a sequence of time series, in which case solution_id should equal 1
586 // - If FROM and TO are realistic dates, then this is an intermediate leg in a sequence
587 // - If FROM is a realistic date and TO equals 00:000:00000, then this is the last leg in
588 // the sequence of time series.
589
590 // start epoch (yy:doy:sssss)
591 discont_from = parse_date_from_string(trimstr(sBuf.substr(16, 6)), doy_yyyy, date_from, std::string(" "));
592 discont_start_date = dateFromStringstream_doy_year<boost::gregorian::date, std::stringstream>(discont_from);
593
594 // end epoch (yy:doy:sssss)
595 discont_to = parse_date_from_string(trimstr(sBuf.substr(29, 6)), doy_yyyy, date_to, std::string(" "));
596 discont_end_date = dateFromStringstream_doy_year<boost::gregorian::date, std::stringstream>(discont_to);
597
599
600 stn_discontinuities->push_back(
601 discontinuity_tuple_t<UINT32, std::string, boost::gregorian::date, bool>(
602 file_rec, // file_index
603 solution_id, // solution_id
604 stn, // site_name
605 discont_start_date, // discontinuity start
606 discont_end_date, // discontinuity end
607 false) // discontinuity_exists
608 );
609
610 file_rec++;
611 }
612 catch (...) {
613 ss.str("");
614 columnNo = 1;
615 ss << "parse_sinex_discontinuities(): Could not extract station name from the record: " << std::endl << " " << sBuf << ".";
616 throw boost::enable_current_exception(std::runtime_error(ss.str()));
617 }
618 }
619
620 // No discontinuities to handle?
621 if (!containsDiscontinuities_)
622 return;
623
624 // sort the station discontinuities on station name
625 std::sort(stn_discontinuities->begin(), stn_discontinuities->end(),
626 CompareSiteTuplesByName<discontinuity_tuple>());
627
628 // Identify if discontinuity exists (by equal station names) and
629 // set flag to indicate that this station is a known discontinuity station
630 _it_vdiscontinuity_tuple it_discont, it_discont_next;
631 for (it_discont = stn_discontinuities->begin();
632 it_discont != stn_discontinuities->end(); it_discont++)
633 {
634 it_discont_next = it_discont + 1;
635
636 if (it_discont_next == stn_discontinuities->end())
637 break;
638
639 // Are station names equal?
640 if (boost::equals(it_discont->site_name, it_discont_next->site_name))
641 {
642 // Then this is a discontinuity site!
643 it_discont->discontinuity_exists = true;
644 it_discont_next->discontinuity_exists = true;
645 }
646 }
647

835void dna_io_snx::reduce_sinex_sites()
836{
837 // sort the sites on station name
838 std::sort(siteOccurrence_.begin(), siteOccurrence_.end(),
839 CompareSiteTuplesByName<site_id>());
840
841 // Identify if discontinuity exists from internal evidence (i.e. equal station names). If so,
842 // increment the station occurrence to indicate that this station is duplicated because
843 // of discontinuity.
844 _it_vsite_id_tuple it_site, it_site_next;
845 for (it_site = siteOccurrence_.begin();
846 it_site != siteOccurrence_.end(); it_site++)
847 {
848 it_site_next = it_site + 1;
849
850 if (it_site_next == siteOccurrence_.end())
851 {
852 // this is the last occurrence
853 it_site->last_occurrence = true;
854 break;
855 }
856
857 // if this station occurs again:
858 // - increment the next occurrence of it
859 // - set the index of the first occurrence
860 if (boost::equals(it_site->site_name, it_site_next->site_name))
861 {
862 // increment next occurrence. Note, if SOLUTION/EPOCHS block has been read,
863 // this will have already been set
864 if (!solutionEpochsRead_)
865 it_site_next->solution_id = it_site->solution_id + 1;
866
867 // If this code is reached, then there must be a discontinuities in the file!
868 containsDiscontinuities_ = true;
869 }
870 else
871 // this is the last occurrence
872 it_site->last_occurrence = true;
873
874 }
875
876 // sort on file index (default sort)
877 std::sort(siteOccurrence_.begin(), siteOccurrence_.end(),
878 CompareSiteTuples<site_id>());
879}

1060 {

1064 " reformat the SINEX file so that the +SOLUTION/EPOCHS block appears before the +SOLUTION/ESTIMATE block.";
1065 throw boost::enable_current_exception(std::runtime_error(ss.str()));

1069 parsemsr_tally.Y = static_cast<UINT32>(vStations->size() * 3);
1070 (*msrCount) = static_cast<UINT32>(vStations->size() * 3);
1071
1072 UINT32 i, count, param1, param2, dimension(static_cast<UINT32>(vStations->size() * 3));
1073 double dparam[3];
1074
1075 UINT32 dimension_all = dimension;
1076
1077 if (containsDiscontinuities_)
1078 dimension_all = static_cast<UINT32>(siteOccurrence_.size() * 3);
1079
1080 if (containsVelocities_)
1081 dimension_all *= 2;

1083 // If this file contains discontinuity sites and the user wishes to retain them,
1084 // then the size of the variance matrix will be the size of the stn discontinuity

1086 if (containsDiscontinuities_ && applyDiscontinuities_)
1087 dimension = static_cast<UINT32>(siteOccurrence_.size() * 3);
1088
1089 // default covariance matrix assumes
1090 matrix_2d covariance_all(dimension_all, dimension_all);
1091 char cBuf[MAX_RECORD_LENGTH];
1092 char* p;
1093
1094 // Get all covariances (whether discontinuities or velocities exist or not) and store
1095 // in covariance_all. The unwanted elements will be stripped later
1096 while (sBuf.at(0) != '-')
1097 {
1098 lineNo++;

1100
1101 // Comments
1102 if (cBuf[0] == '*')
1103 continue;
1104
1105 if (cBuf[0] == '-')
1106 break;

1108 p = cBuf;
1109 while (*p == ' ')
1110 p++;
1111
1112 if ((count = GetFields(p, ' ', true, "ddfff", &param1, &param2, &dparam[0], &dparam[1], &dparam[2])) < 3)

1120 if (param1 > dimension_all)
1121 break;
1122
1123 // Fill covariance matrix.
1124 for (i=0; i<(count-2); i++)

1126 }
1127

1129 if (sinexRec[26] == LOWER_TRIANGLE)
1130 covariance_all.fillupper();

1137 // Hence, the purpose of this block is to remove:
1138 // 1. velocity elements (DynAdjust doesn't handle velocities yet)
1139 // 2. discontinuities if they exist and the user doesn't want them (on account of not
1140 // providing a discontinuities file)

1159 // the z component of the velocity

1161 row_v += 3;
1162#ifdef _MSDEBUG
1163 //TRACE("Skip to row: %d\n", row_v+1);
1164#endif
1165 continue;
1166 }
1167#ifdef _MSDEBUG
1168 //else
1169 //TRACE("Keep this row: %d\n", row_v);
1170#endif
1171 }
1172
1173 // At this point, unnecessary rows (discontinuity sites with or without velocities) have been skipped
1174 // Now copy the data
1175 // X row
1176 for (col_v=row_v, col=row; col_v<dimension_all; ++col_v)
1177 {
1178 // Do we need to skip this site (in the case of unwanted discontinuities?
1179 if (containsDiscontinuities_ && !applyDiscontinuities_)
1180 {
1181 stn_discont = (UINT32)floor((col_v+1) / 3.);
1182 if (containsVelocities_)
1183 stn_discont /= 2;
1184
1185 // Is this the last occurrence? If not, skip it
1186 if (!siteOccurrence_.at(stn_discont).last_occurrence)

1188 // skip to z component
1189 col_v += 2;
1190 // If this site also includes velocities, then skip to
1191 // the z component of the velocity
1192 if (containsVelocities_)
1193 col_v += 3;
1194 continue;
1195 }
1196 }

1198 covariance.put(row, col++, covariance_all.get(row_v, col_v++));
1199 covariance.put(row, col++, covariance_all.get(row_v, col_v++));
1200 covariance.put(row, col++, covariance_all.get(row_v, col_v));
1201
1202 // skip velocity elements
1203 if (containsVelocities_)
1204 col_v += 3;
1205 }
1206
1207 row_v++;
1208 row++;
1209
1210
1211 // Y row
1212 for (col_v=row_v, col=row; col_v<dimension_all; ++col_v)
1213 {
1214 // Do we need to skip this site (in the case of unwanted discontinuities?
1215 if (containsDiscontinuities_ && !applyDiscontinuities_)
1216 {
1217 stn_discont = (UINT32)floor((col_v+1) / 3.);
1218 if (containsVelocities_)
1219 stn_discont /= 2;
1220
1221 // Is this the last occurrence? If not, skip it
1222 if (!siteOccurrence_.at(stn_discont).last_occurrence)
1223 {
1224 // skip to z component
1225 col_v += 2;
1226 // If this site also includes velocities, then skip to
1227 // the z component of the velocity
1228 if (containsVelocities_)
1229 col_v += 3;
1230 continue;
1231 }
1232 }
1233
1234 covariance.put(row, col++, covariance_all.get(row_v, col_v++));
1235 covariance.put(row, col++, covariance_all.get(row_v, col_v));
1236 if (col-row > 2)
1237 covariance.put(row, col++, covariance_all.get(row_v, ++col_v));
1238
1239 // skip velocity elements
1240 if (containsVelocities_)
1241 col_v += 3;
1242 }
1243
1244 row_v++;
1245 row++;
1246
1247 // Z row
1248 for (col_v=row_v, col=row; col_v<dimension_all; ++col_v)
1249 {
1250 // Do we need to skip this site (in the case of unwanted discontinuities?
1251 if (containsDiscontinuities_ && !applyDiscontinuities_)
1252 {
1253 stn_discont = (UINT32)floor((col_v+1) / 3.);
1254 if (containsVelocities_)
1255 stn_discont /= 2;
1256

1258 {
1259 // Is this the last occurrence? If not, skip it
1260 if (!siteOccurrence_.at(stn_discont).last_occurrence)
1261 {
1262 // skip to z component
1263 col_v += 2;
1264 // If this site also includes velocities, then skip to
1265 // the z component of the velocity
1266 if (containsVelocities_)
1267 col_v += 3;
1268 continue;
1269 }
1270 }
1271 }
1272

1274 if (col-row > 1)
1275 covariance.put(row, col++, covariance_all.get(row_v, ++col_v));
1276 if (col-row > 2)
1277 covariance.put(row, col++, covariance_all.get(row_v, ++col_v));
1278
1279 // skip velocity elements
1280 if (containsVelocities_)
1281 col_v += 3;
1282 }
1283
1284 // skip velocity elements
1285 if (containsVelocities_)
1286 row_v += 3;
1287 row++;
1288 }
1289 }
1290 else
1291 covariance = covariance_all;

1293 dnaMsrPtr dnaGpsPoint;

1295 dnaCovariancePtr dnaPointCovariance;
