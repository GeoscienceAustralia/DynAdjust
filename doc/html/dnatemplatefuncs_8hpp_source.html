98// copy_if continues to the next element when the first return from
99// Predicate Pred is true. This function iterates through all elements
100template <typename InputIterator, typename Predicate>
101void copy_if_all_occurrences(InputIterator begin, InputIterator end, Predicate pred)
102{
103 InputIterator next;
104 while (begin != end) {
105 next = begin+1;
106 while (next != end)
107 {
108 if (pred(*begin, *next))
109 next++;
110 else
111 next = end;
112 }
113 ++begin;
114 }
115}

191// insert leading zeros into date string
192template <class S = std::string>
193S FormatDateString(const S& datestring)
194{
195 std::vector<S> tokenList;
196 S delimiter(".");
197 SplitDelimitedString<S>(datestring, delimiter, &tokenList);
198
199 if (tokenList.size() < 3)
200 return "";
201
202 if (tokenList.at(0).size() < 2)
203 tokenList.at(0).insert(0, "0");
204 if (tokenList.at(1).size() < 2)
205 tokenList.at(1).insert(0, "0");
206
207 std::stringstream datess;
208 datess << tokenList.at(0) << "." << tokenList.at(1) << "." << tokenList.at(2);
209 return datess.str();
210}

213//delimiter should be a space, commar or similar
214// Expected input is -38 43 28.24255
215// Use of NSEW to indicate negative/positive hemispheres
216// is not supported yet.
217template <class T>
218T ParseDmsString(const std::string& dmsString, const std::string& delimiter)
219{
220 std::vector<std::string> tokenList;
221 SplitDelimitedString<std::string>(dmsString, delimiter, &tokenList);
222
223 // "-38 43 28.24255"
224 // "-38" -> 38.000000000
225 T dms = DoubleFromString<T>(tokenList.at(0));
226
227 if (tokenList.size() < 2)
228 return dms;
229
230 // "43" -> 0.430000000
231 // -> 38.430000000
232 T min = DoubleFromString<T>(tokenList.at(1));
233 min /= 100.;
234 if (dms < 0.0)
235 dms -= min;
236 else
237 dms += min;
238
239 if (tokenList.size() < 3)
240 return dms;
241
242 // "28.24255" -> 0.002824255
243 // -> 38.432824255
244 T sec = DoubleFromString<T>(tokenList.at(2));
245 sec /= 10000.;
246 if (dms < 0.0)
247 dms -= sec;
248 else
249 dms += sec;
250
251 return dms;
252
253}

349template <class T>
350std::string FormatDnaDmsString(const T& dDegMinSec, int precision)
351{
352 std::stringstream ss;
353 if (precision < 4)
354 precision = 4;
355 ss << std::fixed << std::setprecision(precision) << dDegMinSec;
356 std::string strBuf(ss.str());
357
358 size_t decimal(0);
359 std::string d, m, s;
360
361 // Format degrees minutes and seconds
362 if ((decimal = strBuf.find('.')) != std::string::npos)
363 {
364 // found a decimal point!
365 if (decimal == 0) // decimal point at start, ie >> .0123
366 {
367 strBuf.insert(decimal, "0");
368 decimal++;
369 }
370
371 // degrees
372 d = strBuf.substr(0, decimal);
373
374 // minutes
375 m = strBuf.substr(decimal+1, 2);
376
377 // seconds
378 s = strBuf.substr(decimal+3);
379 if (s.length() > 2)
380 s.insert(2, ".");
381
382 // cater for rounding
383 if (s.substr(0, 2) == "60")
384 {
385 s.replace(0, 2, "00");
386 int mm = atoi(m.c_str()) + 1;
387 ss.str("");
388 ss << mm;
389 m = ss.str();
390 if (mm < 10)
391 m.insert(0, "0");
392 }
393
394 if (m.substr(0, 2) == "60")
395 {
396 m.replace(0, 2, "00");
397 int dd = atoi(d.c_str()) + 1;
398 ss.str("");
399 ss << dd;
400 d = ss.str();
401 }
402
403 ss.str("");
404 ss << std::setw(3) << d << std::setw(3) << std::right << m << " " << std::setw(6) << s;
405 strBuf = ss.str();
406 }
407
408 return strBuf;
409}

423// M = measurement_t, U = UINT32
424template<typename M, typename U>
425class CompareClusterID
426{
427public:
428 CompareClusterID()
429 : _m(0), _id(0) {}
430 CompareClusterID(const std::vector<M>* m, const U& id=0)
431 : _m(m), _id(id) {}
432 // used for lower_bound, upper_bound, etc...
433 bool operator()(const boost::shared_ptr<U> lhs, const U& rhs) {
434 return (*(lhs.get()) < _m->at(rhs).clusterID);
435 }
436 bool operator()(const U& lhs, const boost::shared_ptr<U> rhs) {
437 return (_m->at(lhs).clusterID < *(rhs.get()));
438 }
439 // Sort by clusterID then by file order.
440 // The additional, secondary sort by fileOrder is necessary
441 // if the same order is expected from gcc sort and vc9 sort.
442 // Strangely, vc9's sort implementation ptoduces different results to
443 // gcc if sorted only by clusterID. The difference only occurs when two
444 // measurements have the same clusterID.
445 // This has an impact is on the selection of the next station from the
446 // free list. It is not that one segmentation result is incorrect,
447 // rather, it is just difficult to compare results directly if
448 // segmentation blocks are different. 20.07.2009.
449 bool operator()(const U& lhs, const U& rhs) {
450 if (_m->at(lhs).clusterID == _m->at(rhs).clusterID)
451 return _m->at(lhs).fileOrder < _m->at(rhs).fileOrder;
452 return _m->at(lhs).clusterID < _m->at(rhs).clusterID;
453 }
454 bool operator()(const U& id) {
455 return _m->at(id).clusterID == _id;
456 }
457 inline void SetAMLPointer(const std::vector<M>* m) { _m = m; }
458 inline void SetClusterID(const U& id) { _id = id; }
459 inline bool IsAMLPointerNull() { return _m == NULL; }
460private:
461 const std::vector<M>* _m;
462 U _id;
463};
