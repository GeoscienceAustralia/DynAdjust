32void PrintSummaryMessage(dna_adjust* netAdjust, const project_settings* p, boost::posix_time::milliseconds *elapsed_time)
33{
34 if (p->g.quiet)
35 return;
36
37 cout_mutex.lock();
38 std::cout.flush();
39 UINT32 currentIteration(0);
40
41 // any messages left
42 while (netAdjust->NewMessagesAvailable())
43 {
44 if (!netAdjust->GetMessageIteration(currentIteration))
45 break;
46 std::stringstream ss("");
47 ss << " Iteration " << std::right << std::setw(2) << std::fixed << std::setprecision(0) << currentIteration;
48 ss << ", max station corr: " << std::right << std::setw(12) << netAdjust->GetMaxCorrection(currentIteration) << std::endl;
49 std::cout << PROGRESS_BACKSPACE_28 << std::setw(28) << std::left << ss.str();
50 }
51
52 if (p->a.report_mode)
53 {
54 std::cout << "+ Printing results of last adjustment only" << std::endl;
55 cout_mutex.unlock();
56 return;
57 }
58
59 std::cout << std::left << "+ Done." << std::endl;
60
61 UINT32 block_count(netAdjust->blockCount());
62 std::string block_str(" block");
63 if (block_count > 1)
64 block_str.append("s");
65 block_str.append(".");
66
67 switch (p->a.adjust_mode)
68 {
69 case Phased_Block_1Mode:
70 case PhasedMode:
71 if (netAdjust->GetStatus() == ADJUST_SUCCESS)
72 std::cout << "+ Successfully adjusted " << block_count << block_str;
73 else
74 std::cout << "+ Attempted to adjust " << netAdjust->blockCount() << block_str;
75 std::cout << std::endl;
76 }
77
78
79 std::cout << "+ Solution: ";
80
81 if (netAdjust->GetStatus() != ADJUST_SUCCESS)
82 {
83 std::cout << "failed to converge after ";
84 if (p->a.adjust_mode == Phased_Block_1Mode ||
85 p->a.max_iterations == 1)
86 std::cout << "one iteration." << std::endl;
87 else
88 std::cout << p->a.max_iterations << " iterations." << std::endl;
89
90 if (netAdjust->GetStatus() > ADJUST_THRESHOLD_EXCEEDED)
91 {
92 std::cout << std::endl << "+ Open " << leafStr<std::string>(p->o._adj_file) << " to view the adjustment details." << std::endl << std::endl;
93 cout_mutex.unlock();
94 return;
95 }
96 }
97 else
98 {
99 switch (p->a.adjust_mode)
100 {
101 case Phased_Block_1Mode:
102 std::cout << "estimates solved for Block 1 only." << std::endl <<
103 std::endl <<
104 "- Warning: Depending on the quality of the apriori station estimates, further" << std::endl <<
105 " iterations may be needed. --block1-phased mode should only be used once" << std::endl <<
106 " rigorous estimates have been produced for the entire network." << std::endl << std::endl;
107 break;
108 default:
109 std::cout << "converged after " << netAdjust->CurrentIteration() << " iteration";
110 if (netAdjust->CurrentIteration() > 1)
111 std::cout << "s";
112 std::cout << "." << std::endl;
113 }
114 }
115
116 std::cout << formatedElapsedTime<std::string>(elapsed_time, "+ Network adjustment took ") << std::endl;
117 cout_mutex.unlock();
118
119}

157void GenerateStatistics(dna_adjust* netAdjust, const project_settings* p)
158{
159 // Generate statistics
160 // Don't produce statistics only for block 1 only adjustments
161 if (p->a.adjust_mode != Phased_Block_1Mode)
162 {
163 if (!p->g.quiet)
164 {
165 std::cout << "+ Generating statistics...";
166 std::cout.flush();
167 }
168 netAdjust->GenerateStatistics();
169 if (!p->g.quiet)
170 {
171 std::cout << " done." << std::endl;
172
173 std::cout << "+ Adjustment results:" << std::endl << std::endl;
174 std::cout << "+" << OUTPUTLINE << std::endl;
175 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Number of unknown parameters" << std::fixed << std::setprecision(0) << netAdjust->GetUnknownsCount();
176 if (netAdjust->GetAllFixed())
177 std::cout << " (All stations held constrained)";
178 std::cout << std::endl;
179
180 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Number of measurements" << std::fixed << std::setprecision(0) << netAdjust->GetMeasurementCount();
181
182 if (netAdjust->GetPotentialOutlierCount() > 0)
183 {
184 std::cout << " (" << netAdjust->GetPotentialOutlierCount() << " potential outlier";
185 if (netAdjust->GetPotentialOutlierCount() > 1)
186 std::cout << "s";
187 std::cout << ")";
188 }
189 std::cout << std::endl;
190 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Degrees of freedom" << std::fixed << std::setprecision(0) << netAdjust->GetDegreesOfFreedom() << std::endl;
191 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Chi squared" << std::fixed << std::setprecision(2) << netAdjust->GetChiSquared() << std::endl;
192 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Rigorous sigma zero" << std::fixed << std::setprecision(3) << netAdjust->GetSigmaZero() << std::endl;
193 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Global (Pelzer) Reliability" << std::fixed << std::setw(8) << std::setprecision(3) << netAdjust->GetGlobalPelzerRel() << "(excludes non redundant measurements)" << std::endl << std::endl;
194
195 std::stringstream ss("");
196 ss << std::left << " Chi-Square test (" << std::setprecision(1) << std::fixed << p->a.confidence_interval << "%)";
197 std::cout << std::setw(PRINT_VAR_PAD) << std::left << ss.str();
198 ss.str("");
199 ss << std::fixed << std::setprecision(3) <<
200 netAdjust->GetChiSquaredLowerLimit() << " < " <<
201 netAdjust->GetSigmaZero() << " < " <<
202 netAdjust->GetChiSquaredUpperLimit();
203 std::cout << std::setw(CHISQRLIMITS) << std::left << ss.str();
204 ss.str("");
205
206 if (netAdjust->GetDegreesOfFreedom() < 1)
207 ss << "NO REDUNDANCY";
208 else
209 {
210 ss << "*** ";
211 switch (netAdjust->GetTestResult())
212 {
213 case test_stat_pass:
214 ss << "PASSED"; // within upper and lower
215 break;
216 case test_stat_warning:
217 ss << "WARNING"; // less than lower limit
218 break;
219 case test_stat_fail:
220 ss << "FAILED"; // greater than upper limit
221 break;
222 }
223 ss << " ***";
224 }
225
226 std::cout << std::setw(PASS_FAIL) << std::right << ss.str() << std::endl;
227 std::cout << "+" << OUTPUTLINE << std::endl << std::endl;
228 }
229 }
230 else
231 std::cout << std::endl;
232}

426int ParseCommandLineOptions(const int& argc, char* argv[], const boost::program_options::variables_map& vm, project_settings& p)
427{
428 // capture command line arguments
429 for (int cmd_arg(0); cmd_arg<argc; ++cmd_arg)
430 {
431 p.a.command_line_arguments += argv[cmd_arg];
432 p.a.command_line_arguments += " ";
433 }
434
435 if (vm.count(PROJECT_FILE))
436 {
437 if (boost::filesystem::exists(p.g.project_file))
438 {
439 try {
440 CDnaProjectFile projectFile(p.g.project_file, adjustSetting);
441 p = projectFile.GetSettings();
442 }
443 catch (const std::runtime_error& e) {
444 std::cout << std::endl << "- Error: " << e.what() << std::endl;
445 return EXIT_FAILURE;
446 }
447
448 return EXIT_SUCCESS;
449 }
450
451 std::cout << std::endl << "- Error: project file " << p.g.project_file << " does not exist." << std::endl << std::endl;
452 return EXIT_FAILURE;
453 }
454
455 if (!vm.count(NETWORK_NAME))
456 {
457 std::cout << std::endl << "- Nothing to do - no network name specified. " << std::endl << std::endl;
458 return EXIT_FAILURE;
459 }
460
461 p.g.project_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "dnaproj");
462
463 if (boost::filesystem::exists(p.g.project_file))
464 {
465 // update import settings from dnaproj file
466 try {
467 CDnaProjectFile projectFile(p.g.project_file, importSetting);
468 p.i = projectFile.GetSettings().i;
469 }
470 catch (...) {
471 // do nothing
472 }
473
474 // update geoid file name from dnaproj file (blank if geoid was not executed)
475 try {
476 CDnaProjectFile projectFile(p.g.project_file, geoidSetting);
477 p.n = projectFile.GetSettings().n;
478 }
479 catch (...) {
480 // do nothing
481 }
482
483 // update reftran settings from dnaproj file
484 // Note, if reftran was not executed, reference frame and epoch will be set to
485 // the frame and epoch captured on import.
486 try {
487 CDnaProjectFile projectFile(p.g.project_file, reftranSetting);
488 p.r = projectFile.GetSettings().r;
489 }
490 catch (...) {
491 // do nothing
492 }
493 }
494
495 // binary station file location (output)
496 if (vm.count(BIN_STN_FILE))
497 p.a.bst_file = formPath<std::string>(p.g.input_folder, p.a.bst_file);
498 else
499 p.a.bst_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "bst");
500
501 // binary station file location (output)
502 if (vm.count(BIN_MSR_FILE))
503 p.a.bms_file = formPath<std::string>(p.g.input_folder, p.a.bms_file);
504 else
505 p.a.bms_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "bms");
506
507 if (!boost::filesystem::exists(p.a.bst_file) || !boost::filesystem::exists(p.a.bms_file))
508 {
509 cout_mutex.lock();
510 std::cout << std::endl << "- Nothing to do: ";
511
512 if (p.g.network_name.empty())
513 std::cout << std::endl << "network name has not been specified specified, and " << std::endl << " ";
514 std::cout << p.a.bst_file << " and " << p.a.bms_file << " do not exist." << std::endl << std::endl;
515 cout_mutex.unlock();
516 return EXIT_FAILURE;
517 }
518
519 // output settings
520 if (vm.count(OUTPUT_ADJ_STN_ITER))
521 p.o._adj_stn_iteration = 1;
522 if (vm.count(OUTPUT_ADJ_MSR_ITER))
523 p.o._adj_msr_iteration = 1;
524 if (vm.count(OUTPUT_CMP_MSR_ITER))
525 p.o._cmp_msr_iteration = 1;
526 if (vm.count(OUTPUT_ADJ_STAT_ITER))
527 p.o._adj_stat_iteration = 1;
528 if (vm.count(OUTPUT_ADJ_MSR) || // print adjusted measurements?
529 vm.count(OUTPUT_ADJ_GNSS_UNITS) || // print alternative units for adjusted GNSS measurements?
530 vm.count(OUTPUT_ADJ_MSR_SORTBY) || // print sort adjusted measurements?
531 vm.count(OUTPUT_ADJ_MSR_TSTAT)) // print t-statistic for adjusted measurements?
532 p.o._adj_msr_final = 1;
533
534 if (vm.count(OUTPUT_ADJ_STN_BLOCKS))
535 p.o._output_stn_blocks = 1;
536 if (vm.count(OUTPUT_ADJ_MSR_BLOCKS))
537 p.o._output_msr_blocks = 1;
538 if (vm.count(OUTPUT_ADJ_STN_SORT_ORDER))
539 p.o._sort_stn_file_order = 1;
540
541 if (vm.count(MODE_SIMULTANEOUS))
542 p.a.adjust_mode = SimultaneousMode; // default
543 else if (vm.count(MODE_PHASED_BLOCK1))
544 p.a.adjust_mode = Phased_Block_1Mode;
545 else if (vm.count(MODE_PHASED_MT))
546 {
547 p.a.multi_thread = 1;
548 p.a.adjust_mode = PhasedMode;
549 }
550 else if (vm.count(MODE_PHASED))
551 p.a.adjust_mode = PhasedMode;
552 else if (vm.count(MODE_SIMULATION))
553 p.a.adjust_mode = SimulationMode;
554
555 // Report mode?
556 if (vm.count(MODE_ADJ_REPORT))
557 {
558 p.a.report_mode = true;
559 p.a.max_iterations = 0;
560 }
561 else if (p.a.max_iterations < 1)
562 p.a.report_mode = true;
563
564 if (vm.count(STAGED_ADJUSTMENT))
565 {
566 p.a.stage = true;
567 p.a.multi_thread = false;
568 p.a.adjust_mode = PhasedMode;
569 //p.o._output_stn_blocks = true;
570 }
571
572 // Force inverse method for measurement variances to be the same as that which
573 // is used for the inversion of the normals
574 //if (vm.count(LSQ_INVERSE_METHOD))
575 // p.a.inverse_method_msr = p.a.inverse_method_lsq;
576 if (vm.count(SCALE_NORMAL_UNITY))
577 p.a.scale_normals_to_unity = 1;
578 if (vm.count(OUTPUT_ADJ_MSR_TSTAT))
579 p.o._adj_msr_tstat = 1;
580 if (vm.count(OUTPUT_ADJ_MSR_DBID))
581 p.o._database_ids = 1;
582 if (vm.count(OUTPUT_IGNORED_MSRS))
583 p.o._print_ignored_msrs = 1;
584 if (vm.count(PURGE_STAGE_FILES))
585 p.a.purge_stage_files = 1;
586 if (vm.count(RECREATE_STAGE_FILES))
587 p.a.recreate_stage_files = 1;
588 if (vm.count(TYPE_B_GLOBAL))
589 p.o._apply_type_b_global = 1;
590 if (vm.count(TYPE_B_FILE))
591 p.o._apply_type_b_file = 1;
592
593 p.s.asl_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "asl"); // associated stations list
594 p.s.aml_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "aml"); // associated measurements list
595 p.a.map_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "map"); // station names map
596
597 // has a seg file name been specified?
598 if (vm.count(SEG_FILE))
599 p.a.seg_file = formPath<std::string>(p.g.input_folder, p.a.seg_file);
600 else
601 p.a.seg_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "seg");
602
603 if (vm.count(OUTPUT_APU_CORRELATIONS))
604 p.o._output_pu_covariances = 1;
605
606 // Set up file names dependent on adjustment mode
607 p.o._xyz_file = p.o._adj_file =
608 formPath<std::string>(p.g.output_folder, p.g.network_name);
609
610 if (vm.count(OUTPUT_POS_UNCERTAINTY))
611 {
612 p.o._positional_uncertainty = 1;
613 p.o._apu_file = p.o._adj_file;
614 }
615
616 if (vm.count(OUTPUT_STN_COR_FILE))
617 p.o._cor_file = p.o._adj_file;
618
619 switch (p.a.adjust_mode)
620 {
621 case Phased_Block_1Mode:
622 case PhasedMode:
623
624 p.o._adj_file += ".phased";
625 p.o._xyz_file += ".phased";
626
627 if (vm.count(OUTPUT_POS_UNCERTAINTY))
628 p.o._apu_file += ".phased";
629
630 if (vm.count(OUTPUT_STN_COR_FILE))
631 p.o._cor_file += ".phased";
632
633 if (p.a.adjust_mode == Phased_Block_1Mode)
634 {
635 p.o._adj_file += "-block1";
636 p.o._xyz_file += "-block1";
637
638 if (vm.count(OUTPUT_POS_UNCERTAINTY))
639 p.o._apu_file += "-block1";
640
641 if (vm.count(OUTPUT_STN_COR_FILE))
642 p.o._cor_file += "-block1";
643
644 }
645 else if (p.a.stage)
646 {
647 p.o._adj_file += "-stage";
648 p.o._xyz_file += "-stage";
649
650 if (vm.count(OUTPUT_POS_UNCERTAINTY))
651 p.o._apu_file += "-stage";
652
653 if (vm.count(OUTPUT_STN_COR_FILE))
654 p.o._cor_file += "-stage";
655
656 }
657 else if (p.a.multi_thread)
658 {
659 p.o._adj_file += "-mt";
660 p.o._xyz_file += "-mt";
661
662 if (vm.count(OUTPUT_POS_UNCERTAINTY))
663 p.o._apu_file += "-mt";
664
665 if (vm.count(OUTPUT_STN_COR_FILE))
666 p.o._cor_file += "-mt";
667 }
668 break;
669 case SimultaneousMode:
670 p.o._adj_file += ".simult";
671 p.o._xyz_file += ".simult";
