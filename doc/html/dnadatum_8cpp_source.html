108void CDnaDatum::initialiseDatumFromEpsgCode()
109{
110 // Initialise ellipsoid using epsg code and set epoch if static
111 ellipsoid_.SetEllipsoid(epsgCode_);
112
113 // datum type?
114 if (isEpsgDatumStatic(epsgCode_))
115 datumType_ = STATIC_DATUM;
116 else
117 datumType_ = DYNAMIC_DATUM;
118
119 // Set name
120 datumName_ = datumFromEpsgCode<std::string, UINT32>(epsgCode_);
121}

126std::string CDnaDatum::GetEpoch_s() const
127{
128 if (isEpsgWGS84Ensemble(epsgCode_) && isTimeImmemorial(epoch_))
129 return "";
130 else if (isTimeImmemorial(epoch_))
131 return "";
132 return stringFromDate<boost::gregorian::date>(epoch_);
133}

172void CDnaDatum::SetEpoch(const std::string& epoch)
173{
174 // Parse epoch
175 if (epoch.empty())
176 {
177 if (isEpsgWGS84Ensemble(epsgCode_))
178 epoch_ = timeImmemorial<boost::gregorian::date>();
179 else
180 epoch_ = dateFromString<boost::gregorian::date>(referenceepochFromEpsgCode(epsgCode_));
181 }
182 else
183 epoch_ = dateFromString<boost::gregorian::date>(epoch);
184}

205{
206 // Get epsg code
