148bool CDnaGpsPoint::operator== (const CDnaGpsPoint& rhs) const
149{
150 return (
151 m_strFirst == rhs.m_strFirst &&
152 m_strType == rhs.m_strType &&
153 m_lRecordedTotal == rhs.m_lRecordedTotal &&
154 m_bIgnore == rhs.m_bIgnore &&
155 fabs(m_dX - rhs.m_dX) < PRECISION_1E4 &&
156 fabs(m_dY - rhs.m_dY) < PRECISION_1E4 &&
157 fabs(m_dZ - rhs.m_dZ) < PRECISION_1E4 &&
158 boost::iequals(m_referenceFrame, rhs.m_referenceFrame)
159 );
160}

163bool CDnaGpsPoint::operator< (const CDnaGpsPoint& rhs) const
164{
165 if (m_strFirst == rhs.m_strFirst) {
166 if (m_strType == rhs.m_strType) {
167 if (m_lRecordedTotal == rhs.m_lRecordedTotal) {
168 if (m_bIgnore == rhs.m_bIgnore) {
169 if (m_epoch == rhs.m_epoch) {
170 if (fabs(m_dX - rhs.m_dX) < PRECISION_1E4) {
171 if (fabs(m_dY - rhs.m_dY) < PRECISION_1E4) {
172 return m_dZ < rhs.m_dZ; }
173 else
174 return m_dY < rhs.m_dY; }
175 else
176 return m_dX < rhs.m_dX; }
177 else
178 return m_epoch < rhs.m_epoch; }
179 else
180 return m_bIgnore < rhs.m_bIgnore; }
181 else
182 return m_lRecordedTotal < rhs.m_lRecordedTotal; }
183 else
184 return m_strType < rhs.m_strType; }
185 else
186 return m_strFirst < rhs.m_strFirst;
187}

268void CDnaGpsPoint::WriteDNAMsr(std::ofstream* dna_stream, const dna_msr_fields& dmw, const dna_msr_fields& dml, bool) const
269{
270 *dna_stream << std::setw(dmw.msr_type) << m_strType;
271 if (m_bIgnore)
272 *dna_stream << std::setw(dmw.msr_ignore) << "*";
273 else
274 *dna_stream << std::setw(dmw.msr_ignore) << " ";
275
276 *dna_stream << std::left << std::setw(dmw.msr_inst) << m_strFirst;
277
278 // Print header for first cluster point
279 if (m_lRecordedTotal == m_vPointCovariances.size() + 1)
280 {
281 if (m_ctType == LLH_type_i)
282 *dna_stream << std::left << std::setw(dmw.msr_targ1) << LLH_type;
283 else if (m_ctType == LLh_type_i)
284 *dna_stream << std::left << std::setw(dmw.msr_targ1) << LLh_type;
285 else
286 *dna_stream << std::left << std::setw(dmw.msr_targ1) << XYZ_type;
287
288 *dna_stream << std::left << std::setw(dmw.msr_targ2) << m_lRecordedTotal;
289
290 // print scaling
291 *dna_stream <<
292 std::fixed << std::setprecision(2) <<
293 std::right << std::setw(dmw.msr_gps_vscale) << double_string_width<double, UINT32, std::string>(m_dVscale, dmw.msr_gps_vscale) <<
294 std::right << std::setw(dmw.msr_gps_pscale) << double_string_width<double, UINT32, std::string>(m_dPscale, dmw.msr_gps_vscale) <<
295 std::right << std::setw(dmw.msr_gps_lscale) << double_string_width<double, UINT32, std::string>(m_dLscale, dmw.msr_gps_vscale) <<
296 std::right << std::setw(dmw.msr_gps_hscale) << double_string_width<double, UINT32, std::string>(m_dHscale, dmw.msr_gps_vscale);
297
298 // print reference frame and epoch
299 *dna_stream <<
300 std::right << std::setw(dmw.msr_gps_reframe) << m_referenceFrame <<
301 std::right << std::setw(dmw.msr_gps_epoch) << m_epoch;
302
303 // print database ids
304 if (m_msr_db_map.is_msr_id_set)
305 *dna_stream << std::setw(dmw.msr_id_msr) << m_msr_db_map.msr_id;
306 if (m_msr_db_map.is_cls_id_set)
307 *dna_stream << std::setw(dmw.msr_id_cluster) << m_msr_db_map.cluster_id;
308 }
309 else
310 {
311 // print database ids
312 if (m_msr_db_map.is_msr_id_set)
313 *dna_stream << std::setw(dml.msr_id_msr - dml.msr_targ1) << " " <<
314 std::right << std::setw(dmw.msr_id_msr) << m_msr_db_map.msr_id;
315 if (m_msr_db_map.is_cls_id_set)
316 *dna_stream << std::setw(dmw.msr_id_cluster) << m_msr_db_map.cluster_id;
317 }
318
319 *dna_stream << std::endl;
320
321 UINT32 precision = 4;
322 UINT32 pad(dmw.msr_type + dmw.msr_ignore + dmw.msr_inst + dmw.msr_targ1 + dmw.msr_targ2);
323
324 // X
325 *dna_stream << std::setw(pad) << " ";
326 if (m_ctType == LLH_type_i || m_ctType == LLh_type_i)
327 {
328 precision = 13;
329 *dna_stream << std::right << std::setw(dmw.msr_gps) << std::fixed << std::setprecision(precision) << RadtoDms(m_dX);
330 }
331 else
332 *dna_stream << std::right << std::setw(dmw.msr_gps) << std::fixed << std::setprecision(precision) << m_dX;
333
334 *dna_stream << std::right << std::setw(dmw.msr_gps_vcv_1) << std::scientific << std::setprecision(13) << m_dSigmaXX;
335 *dna_stream << std::endl;
336
337 // Y
338 *dna_stream << std::setw(pad) << " ";
339 if (m_ctType == LLH_type_i || m_ctType == LLh_type_i)
340 *dna_stream << std::right << std::setw(dmw.msr_gps) << std::fixed << std::setprecision(precision) << RadtoDmsL(m_dY);
341 else
342 *dna_stream << std::right << std::setw(dmw.msr_gps) << std::fixed << std::setprecision(precision) << m_dY;
343
344 *dna_stream <<
345 std::right << std::setw(dmw.msr_gps_vcv_1) << std::scientific << std::setprecision(13) << m_dSigmaXY <<
346 std::right << std::setw(dmw.msr_gps_vcv_2) << m_dSigmaYY;
347 *dna_stream << std::endl;
348
349 // Z
350 precision = 4; // whether XYZ or LLH, precision only needs to be at 4
351 *dna_stream << std::setw(pad) << " ";
352 *dna_stream << std::right << std::setw(dmw.msr_gps) << std::fixed << std::setprecision(precision) << m_dZ;
353 *dna_stream <<
354 std::right << std::setw(dmw.msr_gps_vcv_1) << std::scientific << std::setprecision(13) << m_dSigmaXZ <<
355 std::right << std::setw(dmw.msr_gps_vcv_2) << m_dSigmaYZ <<
356 std::right << std::setw(dmw.msr_gps_vcv_3) << m_dSigmaZZ;
357 *dna_stream << std::endl;
358
359 // write GPSPoint covariances (not supported by DNA format)
360 std::vector<CDnaCovariance>::const_iterator _it_cov = m_vPointCovariances.begin();
361 for (_it_cov=m_vPointCovariances.begin(); _it_cov!=m_vPointCovariances.end(); ++_it_cov)
362 _it_cov->WriteDNAMsr(dna_stream, dmw, dml);
363}

366void CDnaGpsPoint::SimulateMsr(vdnaStnPtr* vStations, const CDnaEllipsoid* ellipsoid)
367{
368 m_dX = vStations->at(m_lstn1Index).get()->GetXAxis();
369 m_dY = vStations->at(m_lstn1Index).get()->GetYAxis();
370 m_dZ = vStations->at(m_lstn1Index).get()->GetZAxis();
371
372 if (m_ctType == LLH_type_i || m_ctType == LLh_type_i)
373 {
374 // propagate to geographic
375 CartToGeo<double>(m_dX, m_dY, m_dZ, &m_dX, &m_dY, &m_dZ, ellipsoid);
376
377 // Reduce to orthometric height
378 if (fabs(vStations->at(m_lstn1Index).get()->GetgeoidSep()) > PRECISION_1E4)
379 m_dZ -= vStations->at(m_lstn1Index).get()->GetgeoidSep();
380
381 m_dSigmaXX = 9.402e-009;
382 m_dSigmaXY = 5.876e-010;
383 m_dSigmaYY = 9.402e-009;
384 m_dSigmaXZ = 5.876e-010;
385 m_dSigmaYZ = 5.876e-010;
386 m_dSigmaZZ = 2.500e-001;
387 }
388 else
389 {
390 m_dSigmaXX = 4.022E-04;
391 m_dSigmaXY = -1.369E-04;
392 m_dSigmaXZ = 3.975E-04;
393 m_dSigmaYY = 1.487E-04;
394 m_dSigmaYZ = -2.035E-04;
395 m_dSigmaZZ = 6.803E-04;
396 }
397
398 std::vector<CDnaCovariance>::iterator _it_cov = m_vPointCovariances.begin();
399 for (_it_cov=m_vPointCovariances.begin(); _it_cov!=m_vPointCovariances.end(); ++_it_cov)
400 _it_cov->SimulateMsr(vStations, ellipsoid);
401}

403void CDnaGpsPoint::PopulateMsr(pvstn_t bstRecords, uint32_uint32_map* blockStationsMap, vUINT32* blockStations,
404 const UINT32& map_idx, const CDnaDatum* datum, matrix_2d* estimates, matrix_2d* variances)
405{
406 // populate station coordinate information
407 m_strType = 'Y';
408 m_bIgnore = false;
409 SetCoordType(XYZ_type);
410
411 m_lstn1Index = blockStations->at(map_idx);
412 m_strFirst = bstRecords->at(m_lstn1Index).stationName;
413 m_MSmeasurementStations = ONE_STATION;
414
415 m_referenceFrame = datum->GetName();
416 m_epoch = datum->GetEpoch_s();
417 m_epsgCode = datum->GetEpsgCode_s();
418
419 m_dPscale = 1.0;
420 m_dLscale = 1.0;
421 m_dHscale = 1.0;
422 m_dVscale = 1.0;
423
424 // populate station coordinate estimates and unertainties
425 UINT32 mat_idx = (*blockStationsMap)[m_lstn1Index] * 3;
426 m_dX = estimates->get(mat_idx, 0);
427 m_dY = estimates->get(mat_idx+1, 0);
428 m_dZ = estimates->get(mat_idx+2, 0);
429
430 m_dSigmaXX = variances->get(mat_idx, mat_idx);
431 m_dSigmaXY = variances->get(mat_idx, mat_idx+1);
432 m_dSigmaXZ = variances->get(mat_idx, mat_idx+2);
433 m_dSigmaYY = variances->get(mat_idx+1, mat_idx+1);
434 m_dSigmaYZ = variances->get(mat_idx+1, mat_idx+2);
435 m_dSigmaZZ = variances->get(mat_idx+2, mat_idx+2);
436
437 // populate station covariances
438 UINT32 covariance_count = static_cast<UINT32>(blockStations->size()) - map_idx - 1;
439 m_vPointCovariances.clear();
440 m_vPointCovariances.resize(covariance_count);
441
442 std::vector<CDnaCovariance>::iterator _it_cov = m_vPointCovariances.begin();
443 matrix_2d variances_cart(3, 3);
444 UINT32 ic, jc;
445
446 for (ic=map_idx+1; ic<blockStationsMap->size(); ++ic)
447 {
448 jc = (*blockStationsMap)[blockStations->at(ic)] * 3;
449
450 // get cartesian submatrix corresponding to the covariance
451 variances->submatrix(mat_idx, jc, &variances_cart, 3, 3);
452
453 _it_cov->SetM11(variances_cart.get(0, 0));
454 _it_cov->SetM12(variances_cart.get(0, 1));
455 _it_cov->SetM13(variances_cart.get(0, 2));
456 _it_cov->SetM21(variances_cart.get(1, 0));
457 _it_cov->SetM22(variances_cart.get(1, 1));
458 _it_cov->SetM23(variances_cart.get(1, 2));
459 _it_cov->SetM31(variances_cart.get(2, 0));
460 _it_cov->SetM32(variances_cart.get(2, 1));
461 _it_cov->SetM33(variances_cart.get(2, 2));
462
463 ++_it_cov;
464 }

468UINT32 CDnaGpsPoint::SetMeasurementRec(const vstn_t& binaryStn, it_vmsr_t& it_msr, it_vdbid_t& dbidmap)
469{
470 // first station
471 m_lstn1Index = it_msr->station1;
472 m_strFirst = binaryStn.at(it_msr->station1).stationName;
473
474 m_dPscale = it_msr->scale1;
475 m_dLscale = it_msr->scale2;
476 m_dHscale = it_msr->scale3;
477 m_dVscale = it_msr->scale4;
478
479 m_epoch = it_msr->epoch;
480 m_epsgCode = it_msr->epsgCode;
481 m_referenceFrame = datumFromEpsgCode<std::string, UINT32>(LongFromString<UINT32>(it_msr->epsgCode));
482
483 m_lclusterID = it_msr->clusterID;
484 m_MSmeasurementStations = (MEASUREMENT_STATIONS)it_msr->measurementStations;
485
486 // X, sigmaXX
487 m_bIgnore = it_msr->ignore;
488 m_strType = it_msr->measType;
489 m_dX = it_msr->term1;
490 m_dSigmaXX = it_msr->term2;
491 m_lRecordedTotal = it_msr->vectorCount1;
492 SetCoordType(it_msr->coordType);
493
494 // get data relating to Y, sigmaYY, sigmaXY
495 it_msr++;
496
497 m_dY = it_msr->term1;
498 m_dSigmaXY = it_msr->term2;
499 m_dSigmaYY = it_msr->term3;
500
501 // get data relating to Z, sigmaZZ, sigmaXZ, sigmaYZ
502 it_msr++;
503
504 m_dZ = it_msr->term1;
505 m_dSigmaXZ = it_msr->term2;
506 m_dSigmaYZ = it_msr->term3;
507 m_dSigmaZZ = it_msr->term4;
508
509 m_vPointCovariances.clear();
510 m_vPointCovariances.resize(it_msr->vectorCount2);
511
512 CDnaMeasurement::SetDatabaseMap(*dbidmap);
513 dbidmap += 2;
514 dbidmap += (it_msr->vectorCount2 * 3);
515
516 // now covariances
517 std::vector<CDnaCovariance>::iterator _it_cov = m_vPointCovariances.begin();
518 for (; _it_cov!=m_vPointCovariances.end(); ++_it_cov)
519 _it_cov->SetMeasurementRec(binaryStn, it_msr);
520
521 return it_msr->vectorCount1;
522}

525void CDnaGpsPoint::WriteBinaryMsr(std::ofstream* binary_stream, PUINT32 msrIndex) const
526{
527 measurement_t measRecord;
528
529 // Common
530 measRecord.measType = GetTypeC();
531 measRecord.measStart = xMeas;
532 strcpy(measRecord.coordType, m_strCoordType.c_str());
533 measRecord.ignore = m_bIgnore;
534 measRecord.station1 = m_lstn1Index;
535 measRecord.vectorCount1 = m_lRecordedTotal; // number of GpsPoints in the cluster
536 measRecord.vectorCount2 = static_cast<UINT32>(m_vPointCovariances.size()); // number of Covariances in the GpsPoint
537
538 measRecord.clusterID = m_lclusterID;
539 measRecord.measurementStations = m_MSmeasurementStations;
540
