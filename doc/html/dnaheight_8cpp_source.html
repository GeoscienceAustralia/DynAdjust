55bool CDnaHeight::operator< (const CDnaHeight& rhs) const
56{
57 if (m_strFirst == rhs.m_strFirst) {
58 if (m_strType == rhs.m_strType) { // don't think this is needed
59 if (m_bIgnore == rhs.m_bIgnore) {
60 if (m_epoch == rhs.m_epoch) {
61 if (m_dValue == rhs.m_dValue)
62 return m_dStdDev < rhs.m_dStdDev;
63 else
64 return m_dValue < rhs.m_dValue; }
65 else
66 return m_epoch < rhs.m_epoch; }
67 else
68 return m_bIgnore < rhs.m_bIgnore; }
69 else
70 return m_strType < rhs.m_strType; }
71 else
72 return m_strFirst < rhs.m_strFirst;
73}

108void CDnaHeight::WriteDNAMsr(std::ofstream* dna_stream, const dna_msr_fields& dmw, const dna_msr_fields& dml, bool) const
109{
110 *dna_stream << std::setw(dmw.msr_type) << m_strType;
111 if (m_bIgnore)
112 *dna_stream << std::setw(dmw.msr_ignore) << "*";
113 else
114 *dna_stream << std::setw(dmw.msr_ignore) << " ";
115
116 *dna_stream << std::left << std::setw(dmw.msr_inst) << m_strFirst;
117 *dna_stream << std::setw(dmw.msr_targ1) << " ";
118 *dna_stream << std::setw(dmw.msr_targ2) << " ";
119 *dna_stream << std::right << std::setw(dmw.msr_linear) << std::fixed << std::setprecision(4) << m_dValue; // linear measurement value
120 *dna_stream << std::setw(dmw.msr_ang_d + dmw.msr_ang_m + dmw.msr_ang_s) << " ";
121
122 UINT32 m_stdDevPrec(3); // note change from 6 decimal places to 3
123 *dna_stream << std::setw(dmw.msr_stddev) << StringFromTW(m_dStdDev, dmw.msr_stddev, m_stdDevPrec);
124 //*dna_stream << std::setw(dmw.msr_stddev) << std::fixed << std::setprecision(6) << m_dStdDev;
125
126 *dna_stream << std::setw(dml.msr_gps_epoch - dml.msr_inst_ht) << " ";
127 *dna_stream << std::setw(dmw.msr_gps_epoch) << m_epoch;
128
129 if (m_msr_db_map.is_msr_id_set)
130 *dna_stream << std::setw(dmw.msr_id_msr) << m_msr_db_map.msr_id;
131
132 *dna_stream << std::endl;
133}

136void CDnaHeight::SimulateMsr(vdnaStnPtr* vStations, const CDnaEllipsoid* ellipsoid)
137{
138 // Zn is the z coordinate element of the point on the z-axis
139 // which intersects with the the normal at the given Latitude
140 double nu1, Zn1;
141
142 // calculated height
143 m_dValue = EllipsoidHeight<double>(
144 vStations->at(m_lstn1Index).get()->GetXAxis(),
145 vStations->at(m_lstn1Index).get()->GetYAxis(),
146 vStations->at(m_lstn1Index).get()->GetZAxis(),
147 vStations->at(m_lstn1Index).get()->GetcurrentLatitude(),
148 &nu1, &Zn1, ellipsoid);
149
150 m_dStdDev = 0.024;
151
152 switch (GetTypeC())
153 {
154 case 'H':
155 // N value available?
156 if (fabs(vStations->at(m_lstn1Index).get()->GetgeoidSep()) > PRECISION_1E4)
157 {
158 // reduce to orthometric height
159 m_preAdjCorr = vStations->at(m_lstn1Index).get()->GetgeoidSep();
160 m_dValue -= m_preAdjCorr;
161 }
162 break;
163 }
164
165 m_epoch = "01.10.1985";
166}

169UINT32 CDnaHeight::SetMeasurementRec(const vstn_t& binaryStn, it_vmsr_t& it_msr, it_vdbid_t& dbidmap)
170{
171 m_strType = it_msr->measType;
172 m_bIgnore = it_msr->ignore;
173 m_MSmeasurementStations = (MEASUREMENT_STATIONS)it_msr->measurementStations;
174
175 m_lstn1Index = it_msr->station1;
176 m_strFirst = binaryStn.at(it_msr->station1).stationName;
177
178 m_measAdj = it_msr->measAdj;
179 m_measCorr = it_msr->measCorr;
180 m_measAdjPrec = it_msr->measAdjPrec;
181 m_residualPrec = it_msr->residualPrec;
182 m_preAdjCorr = it_msr->preAdjCorr;
183 m_dValue = it_msr->term1;
184 m_dStdDev = sqrt(it_msr->term2);
185
186 m_epoch = it_msr->epoch;
187
188 CDnaMeasurement::SetDatabaseMap(*dbidmap);
189
190 return 0;
191}

194void CDnaHeight::WriteBinaryMsr(std::ofstream* binary_stream, PUINT32 msrIndex) const
195{
196 measurement_t measRecord;
197 measRecord.measType = GetTypeC();
198 measRecord.measStart = xMeas;
199 measRecord.ignore = m_bIgnore;
200 measRecord.station1 = m_lstn1Index;
201 measRecord.measAdj = m_measAdj;
202 measRecord.measCorr = m_measCorr;
203 measRecord.measAdjPrec = m_measAdjPrec;
204 measRecord.residualPrec = m_residualPrec;
205 measRecord.preAdjCorr = m_preAdjCorr;
206 measRecord.term1 = m_dValue;
207 measRecord.term2 = m_dStdDev * m_dStdDev; // convert to variance
208 measRecord.term3 = 0.;
209 measRecord.term4 = 0.;
210 measRecord.measurementStations = m_MSmeasurementStations;
211 measRecord.fileOrder = ((*msrIndex)++);
212
213 snprintf(measRecord.epoch, sizeof(measRecord.epoch), "%s", m_epoch.substr(0, STN_EPOCH_WIDTH).c_str());
214
215 binary_stream->write(reinterpret_cast<char *>(&measRecord), sizeof(measurement_t));
216}
