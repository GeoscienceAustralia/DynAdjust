54template <typename T, typename U>
55T removeNegativeZero(const T& t, const U& precision)
56{
57 if (t < 0.0 || t == T(-0.0))
58 {
59 T v(floor((t * pow(10.0, precision)) + 0.5));
60 if (fabs(v) > 0.)
61 // t is non-zero at precision
62 return t;
63 // t is a negative zero
64 return 0.;
65 }
66 return t;
67}

115// use this for longitude values only, where 180 degrees
116// marks the boundary between east (positive) and west (negative)
117template <class T>
118T Rad180Mod(const T &dValue)
119{
120 if (dValue < -PI)
121 return dValue + (PI + PI);
122 else if (dValue > PI)
123 return dValue - (PI + PI);
124 return dValue;
125}

210template <class T, typename U>
211void DegtoDms(const T& dDegrees, U* dDegMinSec)
212{
213 T d, m, s;
214
215 *dDegMinSec = fabs(dDegrees); // retain original value
216 d = floor(*dDegMinSec);
217 m = floor((((*dDegMinSec) - d) * 60.0));
218 s = ((*dDegMinSec) - d - (m/60.0)) * 3600.0;
219 if (fabs(s - 60.0) < 0.000000001)
220 {
221 s = 0.0;
222 m += 1.0;
223 }
224 *dDegMinSec = d + (m/100.0) + (s/10000.0);
225 if (dDegrees < 0.0)
226 *dDegMinSec *= -1;
227}

240template <class T, typename U>
241void DmstoDeg(const T& dDegMinSec, U* dDegrees)
242{
243 T dh, dm, ds;
244
245 dh = fabs(dDegMinSec); // retain original value
246 *dDegrees = floor(dh);
247 dm = floor(((dh - (*dDegrees)) * 100.0) + 0.0001);
248 ds = (((dh - (*dDegrees)) * 100.0) - dm) * 100.0;
249 *dDegrees += ((dm / 60.0) + (ds / 3600.0));
250 if (dDegMinSec < 0.0)
251 (*dDegrees) *= -1;
252}

347template <class T>
348// |
349// 4th | 1st
350// |
351// ----------------------
352// |
353// 3rd | 2nd
354// |
355T atan_2(const T& x, const T& y)
356{
357 T theta(atan(x / y)); // first quadrant (default)
358 if (y < 0) // second or third quadrant
359 return theta + PI;
360 else
361 {
362 if (x > 0) // first quadrant
363 return theta;
364 else // fourth quadrant
365 return theta + TWO_PI;
366 }
367 return 0.;
368}
