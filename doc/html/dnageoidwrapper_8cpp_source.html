111{

194bool reportGridProperties(dna_geoid_interpolation* g, const char* gridfilePath, const char* gridfileType)
195{
196 n_file_par grid_properties;
197
198 try {
199 g->ReportGridProperties(gridfilePath, gridfileType, &grid_properties);
200 }
201 catch (const NetGeoidException& e) {
202 std::cout << std::endl << "- Error: " << e.what() << std::endl;
203 return false;
204 }
205
206 bool isRadians(false);
207 std::string shiftType(grid_properties.chGs_type);
208 if (boost::iequals(trimstr(shiftType), "radians"))
209 isRadians = true;
210
211 std::string formattedLimit;
212
213 std::cout << "+ Grid properties for " << gridfilePath << ":" << std::endl;
214 std::cout << " - GS_TYPE = " << grid_properties.chGs_type << std::endl; // grid shift type (GS_TYPE)
215 std::cout << " - VERSION = " << grid_properties.chVersion << std::endl; // grid file version (VERSION)
216 std::cout << " - SYSTEM_F = " << grid_properties.chSystem_f << std::endl; // reference system (SYSTEM_F)
217 std::cout << " - SYSTEM_T = " << grid_properties.chSystem_t << std::endl; // reference system (SYSTEM_T)
218 std::cout << " - MAJOR_F = " << std::setprecision(3) << std::fixed << grid_properties.daf << std::endl; // semi major of from system (MAJOR_F)
219 std::cout << " - MAJOR_T = " << std::setprecision(3) << std::fixed << grid_properties.dat << std::endl; // semi major of to system (MAJOR_T)
220 std::cout << " - MINOR_F = " << std::setprecision(3) << std::fixed << grid_properties.dbf << std::endl; // semi minor of from system (MINOR_F)
221 std::cout << " - MINOR_T = " << std::setprecision(3) << std::fixed << grid_properties.dbt << std::endl; // semi minor of to system (MINOR_T)
222 std::cout << " - NUM_OREC = " << grid_properties.iH_info << std::endl; // Number of header identifiers (NUM_OREC)
223 std::cout << " - NUM_SREC = " << grid_properties.iSubH_info << std::endl; // Number of sub-header idents (NUM_SREC)
224 std::cout << " - NUM_FILE = " << grid_properties.iNumsubgrids << std::endl; // number of subgrids in file (NUM_FILE)
225
226 for (int i=0; i<grid_properties.iNumsubgrids; ++i)
227 {
228 std::string formattedLimit;
229 std::cout << " - SUBGRID " << i << ":" << std::endl;
230 std::cout << " - SUB_NAME = " << grid_properties.ptrIndex[i].chSubname << std::endl; // name of subgrid (SUB_NAME)
231 std::cout << " - PARENT = " << grid_properties.ptrIndex[i].chParent << std::endl; // name of parent grid (PARENT)
232 std::cout << " - CREATED = " << grid_properties.ptrIndex[i].chCreated << std::endl; // date of creation (CREATED)
233 std::cout << " - UPDATED = " << grid_properties.ptrIndex[i].chUpdated << std::endl; // date of last file update (UPDATED)
234
235 // In ptrIndex, all values for the limits of a grid are held in seconds, despite
236 // whether the grid node records are in radians.
237
238 // lower latitude (S_LAT)
239 std::cout << " - S_LAT = " << std::right << std::setw(isRadians ? ZONE : REL) << grid_properties.ptrIndex[i].dSlat;
240 formattedLimit = "(" + FormatDmsString(DegtoDms(grid_properties.ptrIndex[i].dSlat / DEG_TO_SEC), 6, true, false) + ")";
241 std::cout << std::right << std::setw(MEASR) << formattedLimit << std::endl;
242
243 // upper latitude (N_LAT)
244 std::cout << " - N_LAT = " << std::right << std::setw(isRadians ? ZONE : REL) << grid_properties.ptrIndex[i].dNlat;
245 formattedLimit = "(" + FormatDmsString(DegtoDms(grid_properties.ptrIndex[i].dNlat / DEG_TO_SEC), 6, true, false) + ")";
246 std::cout << std::right << std::setw(MEASR) << formattedLimit << std::endl;
247
248 // lower longitude (E_LONG)
249 std::cout << " - E_LONG = " << std::right << std::setw(isRadians ? ZONE : REL) << grid_properties.ptrIndex[i].dElong;
250 formattedLimit = "(" + FormatDmsString(DegtoDms(grid_properties.ptrIndex[i].dElong / DEG_TO_SEC), 6, true, false) + ")";
251 std::cout << std::right << std::setw(MEASR) << formattedLimit << std::endl;
252
253 // upper longitude (W_LONG)

257
258 std::cout << " - LAT_INC = " << std::fixed << grid_properties.ptrIndex[i].dLatinc << std::endl; // latitude interval (LAT_INC)
259 std::cout << " - LONG_INC = " << std::fixed << grid_properties.ptrIndex[i].dLonginc << std::endl; // longitude interval (LONG_INC)
260 std::cout << " - GS_COUNT = " << std::fixed << std::setprecision(0) << grid_properties.ptrIndex[i].lGscount; // number of nodes (GS_COUNT)
261 }
262 std::cout << std::endl;
263 return true;
264}

365int ParseCommandLineOptions(const int& argc, char* argv[], const boost::program_options::variables_map& vm, project_settings& p)
366{
367 // capture command line arguments
368 for (int cmd_arg(0); cmd_arg<argc; ++cmd_arg)
369 {
370 p.n.command_line_arguments += argv[cmd_arg];
371 p.n.command_line_arguments += " ";
372 }
373
374 // Has the user supplied a project file?
375 if (vm.count(PROJECT_FILE))
376 {
377 if (boost::filesystem::exists(p.g.project_file))
378 {
379 try {
380 CDnaProjectFile projectFile(p.g.project_file, geoidSetting);
381 p = projectFile.GetSettings();
382 }
383 catch (const std::runtime_error& e) {
384 std::cout << std::endl << "- Error: " << e.what() << std::endl;
385 return EXIT_FAILURE;
386 }
387
388 return EXIT_SUCCESS;
389 }
390
391 std::cout << std::endl << "- Error: project file " << p.g.project_file << " does not exist." << std::endl << std::endl;
392 return EXIT_FAILURE;
393 }
394
395 if (!vm.count(NETWORK_NAME) && // User wants to populate a binary station file for a particular DynAdjust project
396 !vm.count(INTERACTIVE) && // User wants to interpolate geoid information from the command line
397 !vm.count(CREATE_NTV2) && // User wants to create a NTv2 grid file
398 !vm.count(SUMMARY) && // User wants to print to the screen the details of the grid file
399 !vm.count(INPUT_FILE) && // User wants to interpolate geoid information in text file mode
400 !vm.count(DAT_FILEPATH) && // User supplied file path to WINTER DAT file
401 !vm.count(EXPORT_NTV2_ASCII_FILE) && // User wants to export a geoid file to ASCII
402 !vm.count(EXPORT_NTV2_BINARY_FILE)) // User wants to export a geoid file to binary
403 {
404 std::cout << std::endl << "- Nothing to do - no standard options specified. " << std::endl << std::endl;
405 return EXIT_FAILURE;
406 }
407
408 // check mandatory arguments for NTv2 file creation
409 if (vm.count(CREATE_NTV2) && !vm.count(DAT_FILEPATH))
410 {
411 std::cout << std::endl << "- Error: no dat file specified. " << std::endl << std::endl;
412 return EXIT_FAILURE;
413 }
414
415 // check mandatory arguments for interactive mode
416 if (vm.count(INTERACTIVE) && !vm.count(LATITUDE))
417 {
418 std::cout << std::endl << "- Error: Interpolation latitide not specified. " << std::endl << std::endl;
419 return EXIT_FAILURE;
420 }
421
422 // check mandatory arguments for interactive mode
423 if (vm.count(INTERACTIVE) && !vm.count(LONGITUDE))
424 {
425 std::cout << std::endl << "- Error: Interpolation longitude not specified. " << std::endl << std::endl;
426 return EXIT_FAILURE;
427 }
428
429 // check mandatory arguments for text file mode
430 if (!vm.count(INPUT_FILE) && !vm.count(NTV2_FILEPATH) && !vm.count(CREATE_NTV2))
431 {
432 std::cout << std::endl << "- Error: Interpolation input file not specified. " << std::endl << std::endl;
433 return EXIT_FAILURE;
434 }
435
436 // check mandatory arguments for NTv2 file summary
437 if (vm.count(SUMMARY) && !vm.count(NTV2_FILEPATH))
438 {
439 std::cout << std::endl << "- Error: No NTv2 grid file specified. " << std::endl << std::endl;
440 return EXIT_FAILURE;
441 }
442
443 // check mandatory arguments for DynAdjust station file population
444 if (vm.count(NETWORK_NAME) && !vm.count(NTV2_FILEPATH))
445 {
446 std::cout << std::endl << "- Error: No NTv2 grid file specified. " << std::endl << std::endl;
447 return EXIT_FAILURE;
448 }
449
450 // binary station file location (input)
451 if (vm.count(NETWORK_NAME))
452 {
453 p.g.project_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "dnaproj");
454
455 // define bst file name
456 p.n.bst_file = formPath<std::string>(p.g.input_folder, p.g.network_name, "bst");
457 p.n.file_mode = 1;
458 if (!boost::filesystem::exists(p.n.bst_file))
459 {
460 // Look for it in the input folder
461 p.n.bst_file = formPath<std::string>(p.g.input_folder, leafStr<std::string>(p.n.bst_file));
462
463 if (!boost::filesystem::exists(p.n.bst_file))
464 {
465 std::cout << std::endl << "- Error: ";
466 std::cout << "Binary station file " << p.n.bst_file << " does not exist." << std::endl << std::endl;
467 return EXIT_FAILURE;
468 }
469 }
470 }
471
472 // Geoid DAT grid file file location (input)
473 if (vm.count(DAT_FILEPATH))
474 {
475 if (!boost::filesystem::exists(p.n.rdat_geoid_file))
476 {
477 // Look for it in the input folder
478 p.n.rdat_geoid_file = formPath<std::string>(p.g.input_folder, leafStr<std::string>(p.n.rdat_geoid_file));
479
480 if (!boost::filesystem::exists(p.n.rdat_geoid_file))
481 {
482 std::cout << std::endl << "- Error: ";
483 std::cout << "WINTER DAT grid file " << p.n.rdat_geoid_file << " does not exist." << std::endl << std::endl;
484 return EXIT_FAILURE;
485 }
486 }
487 }
488
489 // Is geoid to run in file mode?
490 if (vm.count(INPUT_FILE))
491 {
492 p.n.file_mode = 1;
493
494 // Geoid DAT grid file file location (input)
495 if (!boost::filesystem::exists(p.n.input_file))
496 {
497 // Look for it in the input folder
498 p.n.input_file = formPath<std::string>(p.g.input_folder, leafStr<std::string>(p.n.input_file));
499
500 if (!boost::filesystem::exists(p.n.input_file))
501 {
502 std::cout << std::endl << "- Error: ";
503 std::cout << "Input coordinates text file " << leafStr<std::string>(p.n.input_file) << " does not exist." << std::endl << std::endl;
504 return EXIT_FAILURE;
505 }
506 }
507 }
508
509 if (vm.count(EXPORT_GEO_FILE))
510 {
511 if (vm.count(NETWORK_NAME))
512 p.n.geo_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "geo"); // dna geoid file
513 p.n.export_dna_geo_file = 1;
514 }
515
516 if (vm.count(DDEG_FORMAT))
517 p.n.coordinate_format = DDEG;
518
519 // Change behaviour of geoid to always convert orthometric heights to ellipsoid
520 // deprecate this command
521 //if (vm.count(CONVERT_BST_HT))
522 p.n.convert_heights = 1;
523 //else
524 // p.n.convert_heights = 0;
525
526 return EXIT_SUCCESS;
527}
