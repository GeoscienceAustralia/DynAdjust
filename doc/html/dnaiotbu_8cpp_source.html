28void dna_io_tbu::assign_typeb_values_global(const vstring& typeBUncertainties, type_b_uncertainty& type_b)
29{
30 double e(0.), n(0.), u(0.);
31 UINT32 i(0);
32 std::stringstream ss;
33
34 if (typeBUncertainties.size() > 2)
35 {
36 // 3 dimensions (east, north, up)
37 if (!typeBUncertainties.at(i).empty())
38 e = DoubleFromString<double>(typeBUncertainties.at(i));
39 if (!typeBUncertainties.at(++i).empty())
40 n = DoubleFromString<double>(typeBUncertainties.at(i));
41 if (!typeBUncertainties.at(++i).empty())
42 u = DoubleFromString<double>(typeBUncertainties.at(i));
43 }
44 else if (typeBUncertainties.size() > 1)
45 {
46 // 2 dimensions (east, north)
47 if (!typeBUncertainties.at(i).empty())
48 e = DoubleFromString<double>(typeBUncertainties.at(i));
49 if (!typeBUncertainties.at(++i).empty())
50 n = DoubleFromString<double>(typeBUncertainties.at(i));
51 }
52 else if (typeBUncertainties.size() > 0)
53 {
54 // 1 dimension (up)
55 if (!typeBUncertainties.at(i).empty())
56 u = DoubleFromString<double>(typeBUncertainties.at(i));
57 }
58 else {
59 ss << " No Type b uncertainties provided." << std::endl;
60 throw boost::enable_current_exception(std::runtime_error(ss.str()));
61 }
62
63 // assign
64 type_b.set_typeb_values_3d(e*e, n*n, u*u);
65}

86void dna_io_tbu::validate_typeb_values(const std::string& argument, vstring& typeBUncertainties)
87{
88 std::stringstream ss;
89
90 if (typeBUncertainties.size() == 0)
91 {
92 ss << " No Type b uncertainties provided:" << std::endl <<
93 " " << argument << std::endl;
94 throw boost::enable_current_exception(std::runtime_error(ss.str()));
95 }
96
97 UINT32 i(0), j;
98
99 for (j = i; j < typeBUncertainties.size(); j++)
100 {
101 if (!is_floatingpoint<std::string>(typeBUncertainties.at(i)))
102 {
103 ss << " Type b uncertainty '" << typeBUncertainties.at(i) <<
104 "' is not a number:" << std::endl <<
105 " " << argument << std::endl;
106 throw boost::enable_current_exception(std::runtime_error(ss.str()));
107 }
108
109 i++;
110 }

113void dna_io_tbu::read_tbu_header(std::ifstream* ptr, std::string& version, INPUT_DATA_TYPE& idt)
114{
115 std::string sBuf;
116 getline((*ptr), sBuf);
117 sBuf = trimstr(sBuf);
118
119 // Set the default version
120 version = "1.00";
121
122 // Attempt to get the file's version
123 try {
124 if (boost::iequals("!#=DNA", sBuf.substr(0, 6)))
125 version = trimstr(sBuf.substr(6, 6));
126 }
127 catch (const std::runtime_error& e) {
128 throw boost::enable_current_exception(std::runtime_error(e.what()));
129 }
130
131 std::string type;
132 std::stringstream ssError;
133 ssError << " File type has not been provided in the header:" << std::endl <<
134 " " << sBuf << std::endl;
135
136 if (sBuf.length() < 15)
137 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
138
139 // Attempt to get the file's type
140 try {
141 type = trimstr(sBuf.substr(12, 3));
142 }
143 catch (const std::runtime_error& e) {
144 ssError << " " << e.what() << std::endl;
145 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
146 }
147
148 // Check this is a Type B file
149 if (boost::iequals(type, "tbu"))
150 idt = tbu_data;
151 else
152 {
153 idt = unknown;
154 std::stringstream ssError;
155 ssError << " The supplied filetype '" << type << "' is not recognised:" << std::endl <<
156 " " << sBuf << std::endl;
157 throw boost::enable_current_exception(std::runtime_error(ssError.str()));
158 }
159}

161void dna_io_tbu::load_tbu_argument(const std::string& argument, type_b_uncertainty& type_b_uncertainties)
162{
163 vstring typeBUncertainties;
164 typeBUncertainties.resize(3);
165
166 // Extract constraints from comma delimited string
167 SplitDelimitedString<std::string>(
168 argument, // the comma delimited string
169 std::string(","), // the delimiter
170 &typeBUncertainties); // the respective values
171
172 // validate values. Throws on exception
173 validate_typeb_values(argument, typeBUncertainties);
174
175 // Assign the type b uncertainties to be applied to all stations, except
176 // when site-specific type b uncertainties have been loaded from a file
177 assign_typeb_values_global(typeBUncertainties, type_b_uncertainties);
178}

194
195 stn_id = it_stnmap_range.first->second;

200void dna_io_tbu::load_tbu_file(const std::string& tbu_filename, v_type_b_uncertainty& type_b_uncertainties, v_string_uint32_pair& vStnsMap)
201{
202 // Type B uncertainty file structure is as follows.
203 // Note - uncertainties are in metres, in the local reference frame (e,n,u)
204 // and are given at 1 sigma (68%).
205 //
206 // station id (20 chars) east uncertainty (13) north uncertainty (13) up uncertainty (13)
207 // ...
208 // EOF
209
210 std::ifstream tbu_file;
211 std::stringstream ss;
212 ss << "load_tbu_file(): An error was encountered when opening " << tbu_filename << "." << std::endl;
213
214 INPUT_DATA_TYPE idt;
215 std::string version;
216
217 try {
218 // open ascii plate boundaries file. Throws runtime_error on failure.
219 file_opener(tbu_file, tbu_filename, std::ios::in, ascii, true);
220
221 // read header information
222 read_tbu_header(&tbu_file, version, idt);
223
224 }
225 catch (const std::runtime_error& e) {
226 ss << e.what();
227 throw boost::enable_current_exception(std::runtime_error(ss.str()));
228 }
229 catch (...) {
230 throw boost::enable_current_exception(std::runtime_error(ss.str()));
231 }
232
233 ss.str("");
234 ss << "load_tbu_file(): An error was encountered when reading from " << tbu_filename << "." << std::endl;
235
236 UINT32 stn_id;
237 std::string record, stn_str;
238 vstring typeb_values;
239 const UINT16 STDDEVn(STATION + STDDEV);
240 const UINT16 STDDEVu(STATION + STDDEV + STDDEV);
241
242 type_b_uncertainty typeBUncertainties;
243
244 type_b_uncertainties.clear();
245
246 try {
247
248 typeb_values.resize(3);
249
250 while (!tbu_file.eof()) // while EOF not found
251 {
252 typeb_values.at(0) = "";
253 typeb_values.at(1) = "";
254 typeb_values.at(2) = "";
255
256 // get the plate identifier
257 getline(tbu_file, record);
258
259 // blank or whitespace?
260 if (trimstr(record).empty())
261 continue;
262
263 // Ignore lines with comments
264 if ((record.compare(0, 1, "*") == 0))
265 continue;
266
267 // obtain station name and unique ID
268 stn_str = trimstr(record.substr(0, STATION));
269 try {
270 identify_station_id(stn_str, stn_id, vStnsMap);
271 }
272 catch (const std::runtime_error&) {
273 // Can't find the station, ignore.
274 continue;
275 }
276
277 typeBUncertainties.set_station_id(stn_id);
278
279 // east uncertainty (may be blank)
280 if (record.length() > STATION)
281 {
282 if (record.length() > STDDEVn)
283 typeb_values.at(0) = trimstr(record.substr(STATION, STDDEV));
284 else
285 typeb_values.at(0) = trimstr(record.substr(STATION));
286 }
287 else
288 // no uncertainties!
289 continue;
290
291 // north uncertainty (may be blank)
292 if (record.length() > STDDEVn)
293 {
294 if (record.length() > STDDEVu)
295 typeb_values.at(1) = trimstr(record.substr(STDDEVn, STDDEV));
296 else
297 typeb_values.at(1) = trimstr(record.substr(STDDEVn));
298 }
299
300 // up uncertainty (may be blank)
301 if (record.length() > STDDEVu)
302 typeb_values.at(2) = trimstr(record.substr(STDDEVu));
303
304 try {
305 // validate values. Throws on exception. Print error and continue
306 // to the next set of values
307 validate_typeb_values(record, typeb_values);
308 }
309 catch (const std::runtime_error&) {
310 continue;
311 }
312
313 // Assign the site-specific type b uncertainties
314 assign_typeb_values_local(typeb_values, typeBUncertainties);
315
316 // add to the list
317 type_b_uncertainties.push_back(typeBUncertainties);
318
319 }
320
321 tbu_file.close();
322
323 }
324 catch (const std::ios_base::failure& f) {
325 if (tbu_file.eof())
326 {
327 tbu_file.close();
328 return;
329 }
330 ss << f.what();
331 throw boost::enable_current_exception(std::runtime_error(ss.str()));
332 }
333 catch (const std::runtime_error& e) {
334 ss << e.what();
335 throw boost::enable_current_exception(std::runtime_error(ss.str()));
336 }
337 catch (...) {
338 if (tbu_file.eof())
339 {
340 tbu_file.close();
341 return;
342 }
343 throw boost::enable_current_exception(std::runtime_error(ss.str()));
344 }
345
346 return;
347}
