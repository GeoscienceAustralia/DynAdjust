30void dna_io_aml::load_aml_file(const std::string& aml_filename, v_aml_pair* vbinary_aml, pvmsr_t bmsRecords)
31{
32 std::ifstream aml_file;
33 std::stringstream ss;
34 ss << "load_aml_file(): An error was encountered when opening " << aml_filename << "." << std::endl;
35
36 try {
37 // open stations aml file. Throws runtime_error on failure.
38 file_opener(aml_file, aml_filename, std::ios::in | std::ios::binary, binary, true);
39 }
40 catch (const std::runtime_error& e) {
41 ss << e.what();
42 throw boost::enable_current_exception(std::runtime_error(ss.str()));
43 }
44 catch (...) {
45 throw boost::enable_current_exception(std::runtime_error(ss.str()));
46 }
47
48 ss.str("");
49 ss << "load_aml_file(): An error was encountered when reading from " << aml_filename << "." << std::endl;
50
51 UINT32 msrValue;
52
53 try {
54 // read the file information
55 readFileInfo(aml_file);
56
57 // Get number of records and resize AML vector
58 aml_file.read(reinterpret_cast<char *>(&msrValue), sizeof(UINT32));
59
60 vbinary_aml->clear();
61 vbinary_aml->resize(msrValue);
62
63 it_aml_pair _it_aml;
64
65 for (_it_aml = vbinary_aml->begin();
66 _it_aml != vbinary_aml->end();
67 ++_it_aml)
68 {
69 aml_file.read(reinterpret_cast<char *>(&msrValue), sizeof(UINT32));
70 // Ignored measurements are retained in the AML, so consume them!
71 _it_aml->bmsr_index = msrValue;
72 if (bmsRecords->at(msrValue).ignore)
73 _it_aml->consume();
74 }
75
76 }
77 catch (const std::ios_base::failure& f) {
78 ss << f.what();
79 throw boost::enable_current_exception(std::runtime_error(ss.str()));
80 }
81 catch (const std::runtime_error& e) {
82 ss << e.what();
83 throw boost::enable_current_exception(std::runtime_error(ss.str()));
84 }
85 catch (...) {
86 throw boost::enable_current_exception(std::runtime_error(ss.str()));
87 }
88
89 aml_file.close();
90}

92void dna_io_aml::write_aml_file(const std::string& aml_filename, pvUINT32 vbinary_aml)
93{
94 std::ofstream aml_file;
95 std::stringstream ss;
96 ss << "write_aml_file(): An error was encountered when opening " << aml_filename << "." << std::endl;
97
98 try {
99 // create binary aml file. Throws runtime_error on failure.
100 file_opener(aml_file, aml_filename,
101 std::ios::out | std::ios::binary, binary);
102 }
103 catch (const std::runtime_error& e) {
104 ss << e.what();
105 throw boost::enable_current_exception(std::runtime_error(ss.str()));
106 }
107 catch (...) {
108 throw boost::enable_current_exception(std::runtime_error(ss.str()));
109 }
110
111 ss.str("");
112 ss << "write_aml_file(): An error was encountered when writing to " << aml_filename << "." << std::endl;
113
114 // Calculate number of measurement records and write to binary file
115 UINT32 amlCount(static_cast<UINT32>(vbinary_aml->size()));
116 it_vUINT32 _it_aml;
117
118 try {
119 // write the file information
120 writeFileInfo(aml_file);
121
122 // write the data
123 aml_file.write(reinterpret_cast<char *>(&amlCount), sizeof(UINT32));
124 for (_it_aml = vbinary_aml->begin();
125 _it_aml != vbinary_aml->end();
126 ++_it_aml)
127 {
128 aml_file.write(reinterpret_cast<char *>(&(*_it_aml)), sizeof(UINT32));
129 }
130 }
131 catch (const std::ios_base::failure& f) {
132 ss << f.what();
133 throw boost::enable_current_exception(std::runtime_error(ss.str()));
134 }
135 catch (const std::runtime_error& e) {
136 ss << e.what();
137 throw boost::enable_current_exception(std::runtime_error(ss.str()));
138 }
139 catch (...) {
140 throw boost::enable_current_exception(std::runtime_error(ss.str()));
141 }
142
143 aml_file.close();

146void dna_io_aml::create_msr_to_stn_tally(const pvASLPtr vAssocStnList, v_aml_pair& vAssocMsrList,
147 vmsrtally& stnmsrTally, vmsr_t& bmsBinaryRecords)
148{
149 stnmsrTally.clear();
150 stnmsrTally.resize(vAssocStnList->size());
151
152 vASLPtr::iterator _it_asl;
153 UINT32 i, m, amlIndex, msrCount;
154
155 for (_it_asl=vAssocStnList->begin(), i=0; _it_asl!=vAssocStnList->end(); ++_it_asl, ++i)
156 {
157 msrCount = _it_asl->get()->GetAssocMsrCount();
158
159 for (m=0; m<msrCount; ++m)
160 {
161 amlIndex = _it_asl->get()->GetAMLStnIndex() + m;
162
163 if (bmsBinaryRecords.at(vAssocMsrList.at(amlIndex).bmsr_index).ignore)
164 continue;
165
166 stnmsrTally.at(i).IncrementMsrType(bmsBinaryRecords.at(vAssocMsrList.at(amlIndex).bmsr_index).measType);
167 }
168
169 stnmsrTally.at(i).TotalCount();
170 }
171}

174void dna_io_aml::write_msr_to_stn(std::ostream &os, pvstn_t bstBinaryRecords,
175 pvUINT32 vStationList, vmsrtally& v_stnmsrTally, MsrTally* parsemsrTally)
176{
177 // Print measurement to station summary header
178 std::string header("MEASUREMENT TO STATIONS ");
179 MsrToStnSummaryHeader(os, header);
180
181 it_vUINT32 _it_stn(vStationList->begin());
182
183 UINT32 msrRedundancies(0);
184 vstring stationRedundantMsrs;
185
186 // Print measurements to each station and the total count for each station
187 for (_it_stn=vStationList->begin(); _it_stn != vStationList->end(); ++_it_stn)
188 {
189 v_stnmsrTally.at(*_it_stn).coutSummaryMsrToStn(os, bstBinaryRecords->at(*_it_stn).stationName);
190
191 // Test for possible redundancies
192 if ((v_stnmsrTally.at(*_it_stn).MeasurementCount('G') ||
193 v_stnmsrTally.at(*_it_stn).MeasurementCount('X') ||
194 v_stnmsrTally.at(*_it_stn).MeasurementCount('Y'))
195 &&
196 (v_stnmsrTally.at(*_it_stn).MeasurementCount('P') ||
197 v_stnmsrTally.at(*_it_stn).MeasurementCount('Q') ||
198 v_stnmsrTally.at(*_it_stn).MeasurementCount('R') ||
199 v_stnmsrTally.at(*_it_stn).MeasurementCount('H') ||
200 v_stnmsrTally.at(*_it_stn).MeasurementCount('I') ||
201 v_stnmsrTally.at(*_it_stn).MeasurementCount('J')))
202 {
203 msrRedundancies++;
204 stationRedundantMsrs.push_back(bstBinaryRecords->at(*_it_stn).stationName);
205 }
206 }
207
208 // Print "the bottom line"
209 MsrToStnSummaryHeaderLine(os);
210
211 // Print the total count per measurement
212 parsemsrTally->coutSummaryMsrToStn(os, "Totals");
213
214 os << std::endl << std::endl;
215
216 if (msrRedundancies > 0)
217 {
218 os << "WARNING: " << msrRedundancies;
219 if (msrRedundancies == 1)
220 os << " station was";
221 else
222 os << " stations were";
223
224 os << " found to have GNSS measurements and absolute terrestrial measurements:" << std::endl << std::endl;
225
226 // Print header
227 os << std::setw(STATION) << std::left << "Station" << std::setw(30) << "Measurement types" << "Count" << std::endl;
228 os << "------------------------------------------------------------" << std::endl;
229
230 // Print measurements to each station and the total count for each station
231 for (_it_stn=vStationList->begin(); _it_stn != vStationList->end(); ++_it_stn)
232 {
233 // Test for possible redundancies
234 if ((v_stnmsrTally.at(*_it_stn).MeasurementCount('G') ||
235 v_stnmsrTally.at(*_it_stn).MeasurementCount('X') ||
236 v_stnmsrTally.at(*_it_stn).MeasurementCount('Y'))
237 &&
238 (v_stnmsrTally.at(*_it_stn).MeasurementCount('P') ||
239 v_stnmsrTally.at(*_it_stn).MeasurementCount('Q') ||
240 v_stnmsrTally.at(*_it_stn).MeasurementCount('R') ||
241 v_stnmsrTally.at(*_it_stn).MeasurementCount('H') ||
242 v_stnmsrTally.at(*_it_stn).MeasurementCount('I') ||
243 v_stnmsrTally.at(*_it_stn).MeasurementCount('J')))
244 {
245 v_stnmsrTally.at(*_it_stn).coutSummaryMsrToStn_Compressed(os, bstBinaryRecords->at(*_it_stn).stationName);
246 }
247 }
248 }

252void dna_io_aml::write_aml_file_txt(const std::string& bms_filename, const std::string& aml_filename, pvUINT32 vbinary_aml, const pvASLPtr vAssocStnList, vdnaStnPtr* vStations)
253{
254 vmsr_t binaryMsrRecords;
255 std::stringstream ss;
256 ss << "write_aml_file(): An error was encountered when opening " << bms_filename << "." << std::endl;
257
258 binary_file_meta_t bms_meta;
259 try {
260 dna_io_bms bms;
261 // Load binary stations data. Throws runtime_error on failure.
262 bms.load_bms_file(bms_filename, &binaryMsrRecords, bms_meta);
263 }
264 catch (const std::runtime_error& e) {
265 ss << e.what();
266 throw boost::enable_current_exception(std::runtime_error(ss.str()));
267 }
268 catch (...) {
269 throw boost::enable_current_exception(std::runtime_error(ss.str()));
270 }
271
272 std::ofstream aml_file;
273 ss.str("");
274 ss << "write_aml_file(): An error was encountered when opening " << aml_filename << "." << std::endl;
275
276 try {
277 // create binary aml file. Throws runtime_error on failure.
278 file_opener(aml_file, aml_filename,
279 std::ios::out | std::ios::binary, binary);
280 }
281 catch (const std::runtime_error& e) {
282 ss << e.what();
283 throw boost::enable_current_exception(std::runtime_error(ss.str()));
284 }
285 catch (...) {
286 throw boost::enable_current_exception(std::runtime_error(ss.str()));
287 }
288
289 ss.str("");
290 ss << "write_aml_file(): An error was encountered when writing to " << aml_filename << "." << std::endl;
291
292 // Write number of measurement records to file
293 std::stringstream ss_aml;
294 ss_aml << vbinary_aml->size() << " records";
295 aml_file << std::left << std::setw(HEADER_18) << ss_aml.str();
296 aml_file << std::left << std::setw(MSR) << "Msr index";
297 aml_file << std::left << std::setw(MSR) << "Msr type";
298 aml_file << std::left << std::setw(MSR) << "Cluster";
299 aml_file << std::left << std::setw(MSR) << "Ignored msr?" << std::endl;
300
301 it_vUINT32_const _it_aml(vbinary_aml->begin());
302
303 UINT32 stn_index(0);
304
305 UINT32 nextAMLStationIndex(vAssocStnList->at(stn_index)->GetAMLStnIndex() + vAssocStnList->at(stn_index)->GetAssocMsrCount());
306 UINT32 currAMLStationIndex(0);
307 vUINT32 msrIndices;
308
309 it_vUINT32 _it_msr;
310
311 try {
312 do
313 {
314 if (stn_index >= vAssocStnList->size())
315 break;
316
317 if (_it_aml >= vbinary_aml->end())
318 break;
319
320 // Is the curent station not connected to any measurements?
321 if (vAssocStnList->at(stn_index)->GetAssocMsrCount() == 0)
322 {
323 // Get the index of the next station in the AML
324 ++stn_index;
325 nextAMLStationIndex = vAssocStnList->at(stn_index)->GetAMLStnIndex() + vAssocStnList->at(stn_index)->GetAssocMsrCount();
326 continue;
327 }
328
329 // Has a new station index been reached?
330 if (currAMLStationIndex == nextAMLStationIndex)
331 {
332 // Get the index of the next station in the AML
333 ++stn_index;
334 nextAMLStationIndex = vAssocStnList->at(stn_index)->GetAMLStnIndex() + vAssocStnList->at(stn_index)->GetAssocMsrCount();
335 continue;
336 }
337
338 // At this point, the current station is connected to measurements.
339 // However, such measurements may be ignored, in which case a station
340 // may be unused.
341
342 // Print station name
343 aml_file << std::setw(HEADER_18) << std::left << vStations->at(stn_index)->GetName();
344
345 // print the bmsr index
346 // For clusters, this will be the index of the first binary element
347 aml_file << std::setw(MSR) << std::left << *_it_aml;
348
349 GetMsrIndices<UINT32>(binaryMsrRecords, *_it_aml, msrIndices);
350
351 ss_aml.str("");
352 ss_aml << binaryMsrRecords.at(*_it_aml).measType;
353
354 for (_it_msr=msrIndices.begin(); _it_msr!=msrIndices.end(); ++_it_msr)
355 {
356 if (stn_index == binaryMsrRecords.at(*_it_msr).station1)
357 {
358 // All measurements have a first station
359 ss_aml << std::left << " (First)";
360 aml_file << std::setw(MSR) << std::left << ss_aml.str();
361 ss_aml.str("");
362 switch (binaryMsrRecords.at(*_it_msr).measType)
363 {
364 case 'D': // Direction set
365 case 'X': // GPS Baseline cluster
366 case 'Y': // GPS point cluster
367 ss_aml << std::setw(MSR) << std::left << binaryMsrRecords.at(*_it_msr).clusterID;
368 break;
369 default:
370 ss_aml << std::setw(MSR) << std::left << " ";
371 break;
372 }
373 continue;
374 }
375
376 if (MsrTally::Stations(binaryMsrRecords.at(*_it_msr).measType) >= TWO_STATION)
377 {
378 // Measurements which have a second or third station
379 switch (binaryMsrRecords.at(*_it_msr).measType)
380 {
381 case 'D': // Direction set
382 if (stn_index == binaryMsrRecords.at(*_it_msr).station2)
383 {
384 if (binaryMsrRecords.at(*_it_msr).vectorCount1 > 0)
385 ss_aml << std::left << " (Second)";
386 else
387 ss_aml << std::left << " (Target)";
388 }
389 aml_file << std::setw(MSR) << std::left << ss_aml.str();
390 ss_aml.str("");
391 ss_aml << std::setw(MSR) << std::left << binaryMsrRecords.at(*_it_msr).clusterID;
392 break;
393 case 'B': // Geodetic azimuth
394 case 'C': // Chord dist
395 case 'E': // Ellipsoid arc
396 case 'G': // GPS Baseline (treat as single-baseline cluster)
397 case 'K': // Astronomic azimuth
398 case 'L': // Level difference
399 case 'M': // MSL arc
400 case 'S': // Slope distance
401 case 'V': // Zenith distance
402 case 'Z': // Vertical angle
403 if (stn_index == binaryMsrRecords.at(*_it_msr).station2)
404 ss_aml << std::left << " (Second)";
405 aml_file << std::setw(MSR) << std::left << ss_aml.str();
406 ss_aml.str("");
407 ss_aml << std::setw(MSR) << std::left << " ";
408 break;
409 case 'X': // GPS Baseline cluster
410 if (stn_index == binaryMsrRecords.at(*_it_msr).station2)
411 ss_aml << std::left << " (Second)";
412 aml_file << std::setw(MSR) << std::left << ss_aml.str();
413 ss_aml.str("");
414 ss_aml << std::setw(MSR) << std::left << binaryMsrRecords.at(*_it_msr).clusterID;
415 break;
416 case 'A': // Horizontal angle
417 if (stn_index == binaryMsrRecords.at(*_it_msr).station2)
418 ss_aml << std::left << " (Second)";
419 else if (stn_index == binaryMsrRecords.at(*_it_msr).station3)
420 ss_aml << std::left << " (Third)";
421 aml_file << std::setw(MSR) << std::left << ss_aml.str();
422 ss_aml.str("");
423 ss_aml << std::setw(MSR) << std::left << " ";
424 break;
425 }
426 }
427 }
428
429 aml_file << std::setw(MSR) << std::left << ss_aml.str();
430
431 if (binaryMsrRecords.at(*_it_aml).ignore)
432 aml_file << std::setw(MSR) << std::left << "*" << std::endl; // Ignored
433 else
434 aml_file << std::setw(MSR) << std::left << " " << std::endl;
435
436 // Move to the next AML record
437 if (++_it_aml == vbinary_aml->end())
438 break;
439
440 ++currAMLStationIndex;
441
442 // has the end of the AML been reached?
443 } while (_it_aml != vbinary_aml->end());
444
445 }
446 catch (const std::ios_base::failure& f) {
447 ss << f.what();
448 throw boost::enable_current_exception(std::runtime_error(ss.str()));
449 }
450 catch (const std::runtime_error& e) {
451 ss << e.what();
452 throw boost::enable_current_exception(std::runtime_error(ss.str()));
453 }
454 catch (...) {
455 throw boost::enable_current_exception(std::runtime_error(ss.str()));
456 }
457
458 aml_file.close();
459}
