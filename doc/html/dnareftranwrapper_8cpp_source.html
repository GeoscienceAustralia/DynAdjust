27void PrintOutputFileHeaderInfo(std::ofstream* f_out, const std::string& out_file, project_settings* p, const std::string& header, UINT32& epsgCode, bool userSuppliedFrame, bool userSuppliedEpoch)
28{
29 // Print formatted header
30 print_file_header(*f_out, header);
31
32 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "File name:" << boost::filesystem::system_complete(out_file).string() << std::endl << std::endl;
33
34 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Command line arguments: ";
35 *f_out << p->r.command_line_arguments << std::endl << std::endl;
36
37 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Network name:" << p->g.network_name << std::endl;
38 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Input folder: " << boost::filesystem::system_complete(p->g.input_folder).string() << std::endl;
39 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Output folder: " << boost::filesystem::system_complete(p->g.output_folder).string() << std::endl;
40 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Stations file:" << boost::filesystem::system_complete(p->r.bst_file).string() << std::endl;
41 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Measurements file:" << boost::filesystem::system_complete(p->r.bms_file).string() << std::endl;
42 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Target reference frame:" << p->r.reference_frame;
43
44 if (userSuppliedFrame)
45 *f_out << " (user supplied)" << std::endl;
46 else
47 *f_out << " (project reference frame)" << std::endl;
48
49 if (!p->r.epoch.empty())
50 {
51 // Has the user supplied the year only?
52 if (p->r.epoch.rfind(".") == std::string::npos)
53 p->r.epoch.insert(0, "01.01.");
54
55 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Target epoch: " << p->r.epoch;
56 if (isEpsgDatumStatic(epsgCode))
57 *f_out << " (adopted reference epoch of " << p->r.reference_frame << ")";
58 else
59 {
60 if (userSuppliedEpoch)
61 *f_out << " (user supplied)" << std::endl;
62 else
63 {
64 if (userSuppliedFrame)
65 *f_out << " (adopted reference epoch of " << p->r.reference_frame << ")" << std::endl;
66 else
67 *f_out << " (project epoch)" << std::endl;
68 }
69 }
70 *f_out << std::endl;
71 }
72
73 if (p->r.plate_model_option > 0)
74 {
75 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Plate boundaries file: " << p->r.tpb_file << std::endl;
76 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "Plate pole parameter file: " << p->r.tpp_file << std::endl;
77 }
78
79 if (p->i.export_dynaml)
80 {
81 if (p->i.export_single_xml_file)
82 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "DynaML output file: " << p->i.xml_outfile << std::endl;
83 else
84 {
85 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "DynaML station file: " << p->i.xml_stnfile << std::endl;
86 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "DynaML measurement file: " << p->i.xml_msrfile << std::endl;
87 }

90 {
91 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "DNA station file: " << p->i.dna_stnfile << std::endl;
92 *f_out << std::setw(PRINT_VAR_PAD) << std::left << "DNA measurement file: " << p->i.dna_msrfile << std::endl;
93 }
94
95// if (p->o._export_snx_file)
96// *f_out << std::setw(PRINT_VAR_PAD) << std::left << "SINEX file: " << p->o._snx_file << std::endl;
97
98
99 *f_out << OUTPUTLINE << std::endl << std::endl;
100}

103int ParseCommandLineOptions(const int& argc, char* argv[], const boost::program_options::variables_map& vm, project_settings& p, UINT32& epsgCode)
104{
105 // capture command line arguments
106 for (int cmd_arg(0); cmd_arg<argc; ++cmd_arg)
107 {
108 p.r.command_line_arguments += argv[cmd_arg];
109 p.r.command_line_arguments += " ";
110 }
111
112 if (vm.count(PROJECT_FILE))
113 {
114 if (boost::filesystem::exists(p.g.project_file))
115 {
116 try {
117 CDnaProjectFile projectFile(p.g.project_file, reftranSetting);
118 p = projectFile.GetSettings();
119 }
120 catch (const std::runtime_error& e) {
121 std::cout << std::endl << "- Error: " << e.what() << std::endl;
122 return EXIT_FAILURE;
123 }
124
125 return EXIT_SUCCESS;
126 }
127
128 std::cout << std::endl << "- Error: project file " << p.g.project_file << " does not exist." << std::endl << std::endl;
129 return EXIT_FAILURE;
130 }
131
132 if (!vm.count(NETWORK_NAME))
133 {
134 std::cout << std::endl << "- Nothing to do - no network name specified. " << std::endl << std::endl;
135 return EXIT_FAILURE;
136 }
137
138 p.g.project_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "dnaproj");
139 p.r.rft_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "rft");
140
141 bool frameSupplied(true);
142 project_settings tmp;
143
144 if (!vm.count(REFERENCE_FRAME))
145 {
146 if (boost::filesystem::exists(p.g.project_file))
147 {
148 try {
149 CDnaProjectFile projectFile(p.g.project_file, reftranSetting);
150 tmp = projectFile.GetSettings();
151 p.r.reference_frame = tmp.r.reference_frame;
152 frameSupplied = false;
153 }
154 catch (const std::runtime_error& e) {
155 std::cout << std::endl << "- Error: " << e.what() << std::endl;
156 return EXIT_FAILURE;
157 }
158 }
159
160 //cout << std::endl << "- Reference frame was not supplied. Using project default (" << p.r.reference_frame << ")" << std::endl << std::endl;
161
162 try
163 {
164 // Okay, no frame supplied, set the frame epsg code from the project settings.
165 // The following throws an exception if the frame is unknown
166 epsgCode = epsgCodeFromName<UINT32>(p.r.reference_frame);
167 }
168 catch (const std::runtime_error& e) {
169 std::cout << std::endl << "- Error: " << e.what() << std::endl;
170 return EXIT_FAILURE;
171 }
172 }
173
174 if (vm.count(EPOCH))
175 {
176 // Get today's date?
177 if (boost::iequals(p.r.epoch, "today"))
178 p.r.epoch = stringFromToday<boost::gregorian::date>();
179 // Has the user supplied the year only?
180 else if (p.r.epoch.rfind(".") == std::string::npos)
181 p.r.epoch.insert(0, "01.01.");
182
183 if (p.r.epoch.length() < 10)
184 p.r.epoch = FormatDateString(p.r.epoch);
185 }
186 else
187 {
188 try
189 {
190 if (frameSupplied)
191 // Okay, frame supplied, but no epoch supplied.
192 // Set the epoch to be the reference epoch of the supplied reference frame
193 p.r.epoch = referenceepochFromEpsgCode<UINT32>(epsgCodeFromName<UINT32, std::string>(p.r.reference_frame));
194 else
195 // Take the epoch from the project file
196 p.r.epoch = tmp.r.epoch;
197 }
198 catch (const std::runtime_error& e) {
199 std::cout << std::endl << "- Error: " << e.what() << std::endl;
200 return EXIT_FAILURE;
201 }
202 }
203
204 if (vm.count(TECTONIC_PLATE_BDY_FILE))
205 {
206 if (!boost::filesystem::exists(p.r.tpb_file))
207 {
208 std::cout << std::endl << "- Error: ";
209 std::cout << std::endl << "tectonic plate boundary file " << std::endl << " ";
210 std::cout << p.r.tpb_file << " does not exist." << std::endl << std::endl;
211 return EXIT_FAILURE;
212 }
213 }
214
215 if (vm.count(TECTONIC_PLATE_POLE_FILE))
216 {
217 if (!boost::filesystem::exists(p.r.tpp_file))
218 {
219 std::cout << std::endl << "- Error: ";
220 std::cout << std::endl << "Euler pole parameters file " << std::endl << " ";
221 std::cout << p.r.tpp_file << " does not exist." << std::endl << std::endl;
222 return EXIT_FAILURE;
223 }
224 }
225
226 // binary station file location (input)
227 if (vm.count(BIN_STN_FILE))
228 p.r.bst_file = formPath<std::string>(p.g.input_folder, p.r.bst_file);
229 else
230 p.r.bst_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "bst");
231
232 // binary station file location (input)
233 if (vm.count(BIN_MSR_FILE))
234 p.r.bms_file = formPath<std::string>(p.g.input_folder, p.r.bms_file);
235 else
236 p.r.bms_file = formPath<std::string>(p.g.output_folder, p.g.network_name, "bms");
237
238 if (!boost::filesystem::exists(p.r.bst_file) || !boost::filesystem::exists(p.r.bms_file))
239 {
240 std::cout << std::endl << "- Nothing to do: ";
241
242 if (p.g.network_name.empty())
243 std::cout << std::endl << "network name has not been specified specified, and " << std::endl << " ";
244 std::cout << p.r.bst_file << " and " << p.r.bms_file << " do not exist." << std::endl << std::endl;
245 return EXIT_FAILURE;
246 }
247
248 // convert to upper case
249 str_toupper<int>(p.r.reference_frame);
250
252 // Export options
253
254 // Create file name based on the provided block
255 std::string fileName(p.g.network_name);
256 fileName.append(".").append(p.r.reference_frame);
257
258 if (!isEpsgDatumStatic(epsgCode))
259 {
260 if (boost::iequals(p.r.epoch, "today"))
261 p.r.epoch = stringFromToday<boost::gregorian::date>();
262 fileName.append(".").append(p.r.epoch);
263 }
264
265 // Export to dynaml?
266 if (vm.count(EXPORT_XML_FILES))
267 {
268 p.i.export_dynaml = 1;
269
270 // single file for both stations and measurements
271 if (vm.count(EXPORT_SINGLE_XML_FILE))
272 {
273 p.i.export_single_xml_file = 1;
274 p.i.xml_outfile = formPath<std::string>(p.g.output_folder,
275 fileName, "xml");
276 }
277 // unique files for stations and measurements
278 else
279 {
280 p.i.export_single_xml_file = 0;
281 p.i.xml_stnfile = formPath<std::string>(p.g.output_folder,
282 fileName + "stn", "xml");
283 p.i.xml_msrfile = formPath<std::string>(p.g.output_folder,
284 fileName + "msr", "xml");
285 }
286 }
287
288 // Export dna files
289 if (vm.count(EXPORT_DNA_FILES))
290 {
291 p.i.export_dna_files = 1;
292 p.i.dna_stnfile = formPath<std::string>(p.g.output_folder,
293 fileName, "stn");
294 p.i.dna_msrfile = formPath<std::string>(p.g.output_folder,
295 fileName, "msr");
296 }
297
298// // Export dna files
299// if (vm.count(EXPORT_SNX_FILE))
300// {
301// p.o._export_snx_file = 1;
302// p.o._snx_file = formPath<std::string>(p.g.output_folder,
303// fileName, "snx");
304// }
305
306 return EXIT_SUCCESS;
307}

309int main(int argc, char* argv[])
310{
311 // create banner message
312 std::string cmd_line_banner;
313 fileproc_help_header(&cmd_line_banner);
314
315 project_settings p;
316
317 boost::program_options::variables_map vm;
318 boost::program_options::positional_options_description positional_options;
319
320 boost::program_options::options_description standard_options("+ " + std::string(ALL_MODULE_STDOPT), PROGRAM_OPTIONS_LINE_LENGTH);
321 boost::program_options::options_description transformation_options("+ " + std::string(REFTRAN_MODULE_TRANSFORM), PROGRAM_OPTIONS_LINE_LENGTH);
322 boost::program_options::options_description export_options("+ " + std::string(ALL_MODULE_EXPORT), PROGRAM_OPTIONS_LINE_LENGTH);
323 boost::program_options::options_description generic_options("+ " + std::string(ALL_MODULE_GENERIC), PROGRAM_OPTIONS_LINE_LENGTH);
324
325 std::string cmd_line_usage("+ ");
326 cmd_line_usage.append(__BINARY_NAME__).append(" usage: ").append(__BINARY_NAME__).append(" ").append(NETWORK_NAME).append(" [options]");
327 boost::program_options::options_description allowable_options(cmd_line_usage, PROGRAM_OPTIONS_LINE_LENGTH);
328
329 try {
330 // Declare a group of options that will be
331 // allowed only on command line
332 standard_options.add_options()
333 (PROJECT_FILE_P, boost::program_options::value<std::string>(&p.g.project_file),

335 (NETWORK_NAME_N, boost::program_options::value<std::string>(&p.g.network_name),
336 "Network name. User defined name for all output files. Default is \"network#\".")
337 (INPUT_FOLDER_I, boost::program_options::value<std::string>(&p.g.input_folder),
338 "Path containing all input files")
339 (OUTPUT_FOLDER_O, boost::program_options::value<std::string>(&p.g.output_folder), // default is ./,
340 "Path for all output files")
341 (BIN_STN_FILE, boost::program_options::value<std::string>(&p.r.bst_file),
342 "Binary station file name. Overrides network name.")
343 (BIN_MSR_FILE, boost::program_options::value<std::string>(&p.r.bms_file),
344 "Binary measurement file name. Overrides network name.")
345 ;
346
347 transformation_options.add_options()
348 (REFERENCE_FRAME_R, boost::program_options::value<std::string>(&p.r.reference_frame),
349 "Target reference frame for all stations and datum-dependent measurements.")
350 (EPOCH_E, boost::program_options::value<std::string>(&p.r.epoch),
351 "Projected date for the transformed stations and measurements. arg is a dot delimited string \"dd.mm.yyyy\", or \"today\" if today's date is required. If no date is supplied, the reference epoch of the supplied reference frame will be used.")
352 (TECTONIC_PLATE_MODEL_OPTION, boost::program_options::value<UINT16>(&p.r.plate_model_option),
353 std::string("Plate motion model option.\n"
354 " 0: Assume all stations are on the Australian plate (default)\n"
355 " 1: Interpolate plate motion model parameters from a defined\n"
356 " set of global tectonic plates. For this option, a global\n"
357 " tectonic plate boundary file and corresponding Euler\n"
358 " pole parameters file must be provided.").c_str())
359 (TECTONIC_PLATE_BDY_FILE_B, boost::program_options::value<std::string>(&p.r.tpb_file),
360 std::string("Global tectonic plate boundaries.").c_str())
361 (TECTONIC_PLATE_POLE_FILE_M, boost::program_options::value<std::string>(&p.r.tpp_file),
362 std::string("Euler pole parameters corresponding to the global tectonic plate boundaries supplied with option --" +
363 StringFromT(TECTONIC_PLATE_BDY_FILE) +
364 ".").c_str())
365 ;
366
367 export_options.add_options()
368 (EXPORT_XML_FILES,
369 "Export transformed stations and measurements to DynaML (DynAdjust XML) format.")
370 (EXPORT_SINGLE_XML_FILE,
371 "Create a single DynaML file for stations and measurements.")
372 (EXPORT_DNA_FILES,
373 "Export transformed stations and measurements to DNA STN and MSR format.")
374 //(EXPORT_SNX_FILE,
375 // "Export transformed station coordinates and full variance matrix to SINEX file.")
376 ;
377
378 generic_options.add_options()
379 (VERBOSE, boost::program_options::value<UINT16>(&p.g.verbose),
380 std::string("Give detailed information about what ").append(__BINARY_NAME__).append(" is doing.\n 0: No information (default)\n 1: Helpful information\n 2: Extended information\n 3: Debug level information").c_str())
381 (QUIET,
382 std::string("Suppresses all explanation of what ").append(__BINARY_NAME__).append(" is doing unless an error occurs").c_str())
383 (VERSION_V, "Display the current program version")
384 (HELP_H, "Show this help message")
385 (HELP_MODULE_H, boost::program_options::value<std::string>(),
386 "Provide help for a specific help category.")
387 ;
388
389 allowable_options.add(standard_options).add(transformation_options).add(export_options).add(generic_options);
390
391 // add "positional options" to handle command line tokens which have no option name
392 positional_options.add(NETWORK_NAME, -1);
393
394 boost::program_options::command_line_parser parser(argc, argv);
395 store(parser.options(allowable_options).positional(positional_options).run(), vm);
396 notify(vm);
397 }
398 catch (const std::exception& e) {
399 std::cout << "- Error: " << e.what() << std::endl;
400 std::cout << cmd_line_banner << allowable_options << std::endl;
401 return EXIT_FAILURE;
402 }
403 catch (...)
404 {
405 std::cout << "+ Exception of unknown type!\n";
406 return EXIT_FAILURE;
407 }
408
409 if (argc < 2)
410 {
411 std::cout << std::endl << "- Nothing to do - no options provided. " << std::endl << std::endl;
412 std::cout << cmd_line_banner << allowable_options << std::endl;
413 return EXIT_FAILURE;
414 }
415
416 if (vm.count(VERSION))
417 {
418 std::cout << cmd_line_banner << std::endl;
419 return EXIT_SUCCESS;
420 }
421
422 if (vm.count(HELP))
423 {
424 std::cout << cmd_line_banner << allowable_options << std::endl;
425 return EXIT_SUCCESS;
426 }
427
428 if (vm.count(HELP_MODULE))
429 {
430 std::cout << cmd_line_banner;
431 std::string original_text = vm[HELP_MODULE].as<std::string>();
432 std::string help_text = str_upper<std::string>(original_text);
433
434 if (str_upper<std::string, char>(ALL_MODULE_STDOPT).find(help_text) != std::string::npos) {
435 std::cout << standard_options << std::endl;
436 }
437 else if (str_upper<std::string, char>(REFTRAN_MODULE_TRANSFORM).find(help_text) != std::string::npos) {
438 std::cout << transformation_options << std::endl;
439 }
440 else if (str_upper<std::string, char>(ALL_MODULE_EXPORT).find(help_text) != std::string::npos) {
441 std::cout << export_options << std::endl;
442 }
443 else if (str_upper<std::string, char>(ALL_MODULE_GENERIC).find(help_text) != std::string::npos) {
444 std::cout << generic_options << std::endl;
445 }
446 else {
447 std::cout << std::endl << "- Error: Help module '" <<
448 original_text << "' is not in the list of options." << std::endl;
449 return EXIT_FAILURE;
450 }
451
452 return EXIT_SUCCESS;
453 }
454
455 if (vm.count(TECTONIC_PLATE_MODEL_OPTION))
456 {
457 if (!vm.count(TECTONIC_PLATE_BDY_FILE))
458 {
459 std::cout << std::endl << "- Error: A plate boundary file must be supplied in order to interpolate plate motion model parameters. See command line help for further information." << std::endl;
460 return EXIT_FAILURE;
461 }
462
463 if (!vm.count(TECTONIC_PLATE_POLE_FILE))
464 {
465 std::cout << std::endl << "- Error: A Euler pole parameters file must be supplied in order to interpolate plate motion model parameters. See command line help for further information." << std::endl;
466 return EXIT_FAILURE;
467 }
468 }
469
470 // Test for a valid reference frame label
471 UINT32 epsgCode;
472 try {
473 epsgCode = epsgCodeFromName<UINT32>(p.r.reference_frame);
474 }
475 catch (const std::runtime_error& e) {
476 std::cout << std::endl << cmd_line_banner;
477
478 std::cout << "- Error: " << e.what() << std::endl;
479 return EXIT_FAILURE;
480 }
481
482 if (ParseCommandLineOptions(argc, argv, vm, p, epsgCode) != EXIT_SUCCESS)
483 return EXIT_FAILURE;
484
485 // If the reference frame supplied is static, then force the epoch
486 // to be the reference epoch
487 if (isEpsgDatumStatic(epsgCode))
488 p.r.epoch = referenceepochFromEpsgCode<UINT32>(epsgCode);
489
490 std::ofstream rft_file;
491 try {
492 // Create import log file. Throws runtime_error on failure.
493 file_opener(rft_file, p.r.rft_file);
494 }
495 catch (const std::runtime_error& e) {
496 std::stringstream ss;
497 ss << "- Error: Could not open " << p.r.rft_file << ". \n Check that the file exists and that the file is not already opened." << std::endl;
498 std::cout << ss.str() << e.what() << std::endl;
499 return EXIT_FAILURE;
500 }
501
502
503
504 if (vm.count(QUIET))
505 p.g.quiet = 1;
506
507 if (!p.g.quiet)
508 {
509 std::cout << std::endl << cmd_line_banner;
510
511 std::cout << "+ Options:" << std::endl;
512 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Network name: " << p.g.network_name << std::endl;
513 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Input folder: " << p.g.input_folder << std::endl;
514 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Output folder: " << p.g.output_folder << std::endl;
515 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Binary station file: " << p.r.bst_file << std::endl;
516 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Binary measurement file: " << p.r.bms_file << std::endl;
517 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Target reference frame: " << p.r.reference_frame;
518 if (vm.count(REFERENCE_FRAME))
519 std::cout << " (user supplied)" << std::endl;
520 else
521 std::cout << " (project reference frame)" << std::endl;
522
523 // try to parse user supplied string. If this fails, then there's not much point in attempting
524 // to transform stations and measurements
525 try
526 {
527 if (!p.r.epoch.empty())
528 {
529 std::cout << std::setw(PRINT_VAR_PAD) << std::left << " Target epoch: " << p.r.epoch;
530 if (isEpsgDatumStatic(epsgCode))
531 std::cout << " (adopted reference epoch of " << p.r.reference_frame << ")";
532 else
533 {
534 if (vm.count(EPOCH))
535 std::cout << " (user supplied)" << std::endl;
536 else
537 {
538 if (vm.count(REFERENCE_FRAME))
539 std::cout << " (adopted reference epoch of " << p.r.reference_frame << ")" << std::endl;
540 else
541 std::cout << " (project epoch)" << std::endl;
542 }
543 }
544 std::cout << std::endl;
545 }
546 }
547 catch (const std::runtime_error& e) {
548 std::cout << std::endl << "- Error: " << e.what() << std::endl;
549 rft_file.close();
550 return EXIT_FAILURE;
551 }
552 catch (const RefTranException& e) {
553 std::cout << std::endl << "- Error: " << e.what() << std::endl;
554 rft_file.close();
555 return EXIT_FAILURE;
556 }
557 catch (...) {
558 std::cout << std::endl << "- Error: Unknown error." << std::endl;
559 rft_file.close();
560 return EXIT_FAILURE;
561 }
562
563 if (p.r.plate_model_option > 0)
564 {
