29UINT16 dna_io_bms::create_msr_input_file_meta(vifm_t& vinput_file_meta, input_file_meta_t** input_file_meta)
30{
31 UINT16 msr_file_count(0);
32 std::stringstream ss;
33 ss << "create_msr_input_file_meta(): An error was encountered when creating " << std::endl <<
34 " the binary measurement file metadata." << std::endl;
35
36 // Determine how many measurement files were supplied
37 try {
38 for_each(vinput_file_meta.begin(), vinput_file_meta.end(),
39 [&msr_file_count] (input_file_meta_t& ifm) {
40 if (ifm.datatype == msr_data || ifm.datatype == stn_msr_data)
41 msr_file_count++;
42 });
43
44 if (*input_file_meta != NULL)
45 delete [] *input_file_meta;
46
47 (*input_file_meta) = new input_file_meta_t[msr_file_count];
48
49 msr_file_count = 0;
50 for_each(vinput_file_meta.begin(), vinput_file_meta.end(),
51 [&msr_file_count, &input_file_meta] (input_file_meta_t& ifm) {
52 if (ifm.datatype == msr_data || ifm.datatype == stn_msr_data)
53 {
54 (*input_file_meta)[msr_file_count] = ifm;
55 msr_file_count++;
56 }
57 });
58
59 }
60 catch (...) {
61 throw boost::enable_current_exception(std::runtime_error(ss.str()));
62 }
63
64 return msr_file_count;
65}

125 throw boost::enable_current_exception(std::runtime_error(ss.str()));

127

130
131 ss.str("");

148 }

150 ss << f.what();

156 }
157 catch (...) {
158 throw boost::enable_current_exception(std::runtime_error(ss.str()));
159 }
160
161 bms_file.close();
162
163 return bms_meta.binCount;
