72// Define typedef struct interpolant
73typedef struct {
74 double dLatitude; // Latitude of the interpolation point
75 double dLongitude; // Longitude of the interpolation point
76 double dHeight; // Height of the interpolation point (either Orthometric or Ellipsoidal)
77 int iDatum; // ANS/GRS80
78 int iHeightSystem; // 0 (zero) indicates Orthometric, 1 (one) Ellipsoidal
79 int IO_Status; // Holds status type for Point Input Output success\n
80 /* -1: The grid file has not been opened. Returned when a call is made to BiLinearTransformation without prviously calling CreateGridIndex.
81 0: Success...Point read and written.
82 1: Could not read from grid file...perhaps a wrong format or file is corrupted
83 2: Unable to read this type of grid file
84 3: The grid file is corrupt
85 12: Invalid Zone...cannot transform to geographic from grid coords
86 13: Point is outside limits of the grid
87 14: There is an error in the grid
88 15: Could not retrieve shifts from Ascii file
89 16: Could not retrieve shifts from Binary file
90 */
91} interpolant;

107// NTv2 specific typedef struct gridfileindex
108typedef struct gridfileindex {
109public:
110 gridfileindex()
111 : dSlat(-165600.), dNlat(-28800.), dElong(-576000.), dWlong(-388800.)
112 , dLatinc(60.), dLonginc(60.), lGscount(0), iGridPos(0)
113 {
114 strcpy(chSubname, "AUSGEOID");
115 strcpy(chParent, "NONE ");
116 strcpy(chCreated, "01012010");
117 strcpy(chUpdated, "01012010");
118 }
119
120 char chSubname[9]; // name of subgrid (SUB_NAME)
121 char chParent[9]; // name of parent grid (PARENT)
122 char chCreated[9]; // date of creation (CREATED)
123 char chUpdated[9]; // date of last file update (UPDATED)
124 double dSlat; // lower latitude (S_LAT)
125 double dNlat; // upper latitude (N_LAT)
126 double dElong; // lower longitude (E_LONG)
127 double dWlong; // upper longitude (W_LONG)
128 double dLatinc; // latitude interval (LAT_INC)
129 double dLonginc; // longitude interval (LONG_INC)
130 long lGscount; // number of nodes (GS_COUNT)
131 int iGridPos; // position of first record of grid shifts
132} n_gridfileindex;

135// NTv2 specific typedef struct filepar
136typedef struct file_par {
137public:
138 file_par()
139 : ptrIndex(0), daf(6378137.), dat(6378137.), dbf(6356752.314), dbt(6356752.314)
140 , dLat(0.), dLatacc(0.), dLong(0.), dLongacc(0.)
141 , iH_info(11), iSubH_info(11), iNumsubgrids(1), Can_Format(true)
142 {
143 memset(filename, 0, 601);
144 strcpy(filetype, "GSB");
145 strcpy(chGs_type, "SECONDS");
146 strcpy(chVersion, "1.0.0.0");
147 strcpy(chSystem_f, "GDA94 ");
148 strcpy(chSystem_t, "AHD_1971");
149 }
150
151 n_gridfileindex* ptrIndex; // pointer to an index (an array) of grid sub files
152 char filename[601]; // Distortion grid file name
153 char filetype[4]; // Distortion grid file type (ascii: "asc" = 0; binary: "gsb" = 1)
154 char chGs_type[9]; // grid shift type (GS_TYPE)
155 char chVersion[9]; // grid file version (VERSION)
156 char chSystem_f[9]; // reference system (SYSTEM_F)
157 char chSystem_t[9]; // reference system (SYSTEM_T)
158 double daf; // semi major of from system (MAJOR_F)
159 double dat; // semi major of to system (MAJOR_T)
160 double dbf; // semi minor of from system (MINOR_F)
161 double dbt; // semi minor of to system (MINOR_T)
162 double dLat; // user input latitude
163 double dLatacc; // interpolated latitude accuracy
164 double dLong; // user input longitude
165 double dLongacc; // interpolated longitude accuracy
166 int iH_info; // Number of header identifiers (NUM_OREC)
167 int iSubH_info; // Number of sub-header idents (NUM_SREC)
168 int iNumsubgrids; // number of subgrids in file (NUM_FILE)
169 size_t iGfilelength; // Size of the grid file in bytes
170 int iPointFlag; // a four way flag indicating the location of the point within subgrid
171 int iTheGrid; // The array index of the most dense sub grid
172 bool Can_Format; // Format of Binary Grid File (true = Canadian; false = Australian)
173} n_file_par;
