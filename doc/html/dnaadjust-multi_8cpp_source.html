37bool combineAdjustmentExceptionThrown(const std::vector<boost::exception_ptr>& cmb_errors_)
38{
39 // got a forward or reverse exception, or
40 // have any of the other combination adjustments failed?
41 if (fwd_error || rev_error)
42 {
43 if (combineAdjustmentQueue.is_empty())
44 combineAdjustmentQueue.push_and_notify(99999999);
45 return true;
46 }
47
48 UINT32 err_size(static_cast<UINT32>(cmb_errors_.size()));
49 for (UINT32 err=0; err<err_size; ++err)
50 {
51 if (cmb_errors_.at(err))
52 {
53 if (combineAdjustmentQueue.is_empty())
54 combineAdjustmentQueue.push_and_notify(99999999);
55 return true;
56 }
57 }
58 return false;
59}

77// Multi thread phased adjustment
78// Notes for general understanding of the multi thread operation:
79// - Thread 1 undertakes a forward adjustment
80// - Thread 2 undertakes a reverse adjustment
81// - Thread 3 undertakes a combine adjustment
82// - Threads 1 and 2 are executed simultaneously without any due regard for each other. The
83// coordinates and uncertainties produced from the forward and reverse runs are managed in
84// separate matrices, hence the two threads may run without conflict.
85// After each block is adjusted, the corresponding forward and reverse mutexes for each
86// block are unlocked.
87// - Immediately after block n has been adjusted in both forward and reverse directions,
88// Thread 3 undertakes combination adjustment of forward/reverse blocks. This is handled
89// by waiting for the two (forward and reverse) mutexes for each block. Hence, it is
90// only when both Threads 1 and 2 have triggered 'unlock' on the respective forward/reverse
91// mutexes that Thread 3 commence.
92//
93void dna_adjust::AdjustPhasedMultiThread()
94{
95 initialiseIteration();
96
97 std::string corr_msg;
98 std::ostringstream ss;
99 UINT32 i;
100 bool iterate(true);
101
102 boost::posix_time::milliseconds iteration_time(boost::posix_time::milliseconds(0));
103 boost::timer::cpu_timer it_time, tot_time;
104
105#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
106 boost::shared_ptr<thread> f, r, c;
107 std::vector< boost::shared_ptr<thread> > mt_adjust_threads_sp;
108#else
109 std::vector<boost::thread> mt_adjust_threads;
110#endif
111
112 concurrentAdjustments.resize_runs(blockCount_);
113
114 // do until convergence criteria is met
115 for (i=0; i<projectSettings_.a.max_iterations; ++i)
116 {
117 if (IsCancelled())
118 break;
119
120 SetcurrentBlock(0);
121 blockLargeCorr_ = 0;
122 largestCorr_ = 0.0;
123 maxCorr_ = 0.0;
124
126 // Print the iteration # to adj file.
127 // protected write to adj file (not needed here since write to
128 // adj file at this stage is via single thread
129 PrintIteration(incrementIteration());
131
132 concurrentAdjustments.reset_adjustment_runs();
133 concurrentAdjustments.combine_run_stopped();
134
135 combineAdjustmentQueue.reset_blocks_coming();
136
137 // Forward and reverse threads sequentially adjust all blocks in
138 // forward and reverse directions.
139#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
140 f.reset(new thread(adjust_forward_thread(this, boost::ref(fwd_error))));
141 mt_adjust_threads_sp.push_back(f);
142 r.reset(new thread(adjust_reverse_thread(this, boost::ref(rev_error))));
143 mt_adjust_threads_sp.push_back(r);
144#else
145 mt_adjust_threads.push_back(boost::thread(adjust_forward_thread(this, boost::ref(fwd_error))));
146 mt_adjust_threads.push_back(boost::thread(adjust_reverse_thread(this, boost::ref(rev_error))));
147#endif
148
149 // Combination thread fires up as many threads as there are cores
150 // on the current PC to concurrently combine blocks
151#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
152 if (CombinationThreadRequired())
153 {
154 c.reset(new thread(adjust_combine_thread(this, boost::ref(cmb_error))));
155 mt_adjust_threads_sp.push_back(c);
156 }
157#else
158 if (CombinationThreadRequired())
159 mt_adjust_threads.push_back(boost::thread(adjust_combine_thread(this, boost::ref(cmb_error))));
160#endif
161
162 // Start the clock
163 it_time.start();
164
165 // Start the forward, reverse and combine threads, which commences the
166 // network adjustment
167#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
168 for_each(mt_adjust_threads_sp.begin(), mt_adjust_threads_sp.end(), boost::mem_fn(&thread::join));
169#else
170 for_each(mt_adjust_threads.begin(), mt_adjust_threads.end(), boost::mem_fn(&boost::thread::join));
171#endif
172 // This point is reached when the threads have finished
173 iteration_time = boost::posix_time::milliseconds(it_time.elapsed().wall/MILLI_TO_NANO);
174
175 //delete mt_adjust_threads;
176#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
177 mt_adjust_threads_sp.clear();
178#else
179 mt_adjust_threads.clear();
180#endif
181
182 concurrentAdjustments.combine_run_stopped();
183
184 // Were any exceptions thrown? If so, re-throw and let
185 // test stub handle the exception
186 if (fwd_error)
187 // exception thrown in the forward pass
188 boost::rethrow_exception(fwd_error);
189 else if (rev_error)
190 // exception thrown in the reverse pass
191 boost::rethrow_exception(rev_error);
192 else if (cmb_error)
193 // exception thrown in one of the combination adjustments
194 boost::rethrow_exception(cmb_error);
195
196 if (IsCancelled())
197 break;
198
199 ss.str("");
200 if (iteration_time > boost::posix_time::seconds(1))
201 ss << boost::posix_time::seconds(static_cast<long>(iteration_time.total_seconds()));
202 else
203 ss << iteration_time;
204
206 // protected write to adj file (not needed here since write to
207 // adj file at this stage is via single thread
208 adj_file << std::setw(PRINT_VAR_PAD) << std::left << "Elapsed time" << ss.str() << std::endl;
209 OutputLargestCorrection(corr_msg);
211
212 if (projectSettings_.g.verbose)
213 debug_file << concurrentAdjustments.print_adjusted_blocks();
214
215 iterationCorrections_.add_message(corr_msg);
216 iterationQueue_.push_and_notify(CurrentIteration()); // currentIteration begins at 1, so not zero-indexed
217
218 // continue iterating?
219 iterate = fabs(maxCorr_) > projectSettings_.a.iteration_threshold;
220
221 // Update normals and measured-computed matrices for the next iteration.
222 // Similar to PrepareAdjustment, UpdateAdjustment prepares every block
223 // in the network so that forward and reverse adjustments can commence
224 // at the same time.
225 UpdateAdjustment(iterate);
226
227 if (!iterate)
228 break;
229 }
230
231 isAdjusting_ = false;
232
233 if (adjustStatus_ > ADJUST_TEST_FAILED)
234 return;
235
236 // This is similar to the check in ValidateandFinaliseAdjustment API for
237 // phased single-thread mode.
238 if (IsCancelled())
239 {
240 adjustStatus_ = ADJUST_CANCELLED;
241 return;
242 }
243
244 if (CurrentIteration() == projectSettings_.a.max_iterations)
245 adjustStatus_ = ADJUST_MAX_ITERATIONS_EXCEEDED;
246
247 // Print status
248 PrintAdjustmentStatus();
249 // Compute and print time taken to run adjustment
250 PrintAdjustmentTime(tot_time, total_time);
251}

274void dna_adjust::SolveMTTry(bool COMPUTE_INVERSE, const UINT32& block)
275{
276 // Least Squares Solution
277 try {
278 SolveMT(COMPUTE_INVERSE, block);
279 }
280 catch (const std::runtime_error& e) {
281
282 // could not invert matrix. throw error
283
284 if (projectSettings_.g.verbose)
285 {
286 debug_file << "Block " << block + 1 << " (reverse, multi-thread)" << std::endl;
287 debug_file << "Pre-adjustment Estimates" << std::fixed << std::setprecision(16) << v_estimatedStationsR_.at(block);
288 debug_file << "Design" << std::fixed << std::setprecision(16) << v_designR_.at(block);
289 debug_file << "Measurements" << std::fixed << std::setprecision(16) << v_measMinusCompR_.at(block);
290 debug_file << "At * V-inv" << std::fixed << std::setprecision(16) << v_AtVinvR_.at(block);
291 debug_file << "Normals " << std::fixed << std::setprecision(16) << v_normalsR_.at(block) << std::endl;
292 debug_file.flush();
293 }
294
295 SignalExceptionAdjustment(e.what(), block);
296 }

299void dna_adjust::SolveMT(bool COMPUTE_INVERSE, const UINT32& block)
300{
301 if (COMPUTE_INVERSE)
302 {
303 // Compute inverse of normals (aposteriori variance matrix)
304 // (AT * V-1 * A)-1
305 FormInverseVarianceMatrix(&(v_normalsR_.at(block)));
306 }
307
308 // compute weighted "measured minus computed"
309 matrix_2d At_Vinv_m(v_designR_.at(block).columns(), 1);
310 At_Vinv_m.multiply(v_AtVinvR_.at(block), "N", v_measMinusCompR_.at(block), "N");
311
312 // Solve corrections from normal equations
313 v_correctionsR_.at(block).redim(v_designR_.at(block).columns(), 1);
314 v_correctionsR_.at(block).multiply(v_normalsR_.at(block), "N", At_Vinv_m, "N");
315
316 // debug output?
317 if (projectSettings_.g.verbose > 3)
318 {
319 dbg_file_mutex.lock();
320
321 UINT32 i;
322 switch (projectSettings_.a.adjust_mode)
323 {
324 case PhasedMode:
325 for (i=0; i<v_parameterStationList_.at(block).size(); ++i)
326 {
327 debug_file << bstBinaryRecords_.at(v_parameterStationList_.at(block).at(i)).stationName;
328 if (find(v_ISL_.at(block).begin(), v_ISL_.at(block).end(), v_parameterStationList_.at(block).at(i)) != v_ISL_.at(block).end())
329 debug_file << "i ";
330 else if (find(v_JSL_.at(block).begin(), v_JSL_.at(block).end(), v_parameterStationList_.at(block).at(i)) != v_JSL_.at(block).end())
331 debug_file << "j ";
332 else
333 debug_file << " ";
334 }
335 break;
336 }
337
338 debug_file << std::endl;
339
340 debug_file << "Block " << block + 1 << " (Reverse)";
341 debug_file << std::endl;
342 debug_file << "Pre-adjustment Estimates" << std::fixed << std::setprecision(16) << v_estimatedStationsR_.at(block);
343 debug_file << "Block " << block + 1 << " (Reverse)";
344 debug_file << std::endl;
345 debug_file << "Design" << std::fixed << std::setprecision(16) << v_designR_.at(block);
346 debug_file << "Block " << block + 1 << " (Reverse)";
347 debug_file << std::endl;
348 debug_file << "Measurements" << std::fixed << std::setprecision(16) << v_measMinusCompR_.at(block);
349 debug_file << "Block " << block + 1 << " (Reverse)";
350 debug_file << std::endl;
351 debug_file << "At * V-inv" << std::fixed << std::setprecision(16) << v_AtVinvR_.at(block) << std::endl;
352 debug_file << "Block " << block + 1 << " (Reverse)";
353 debug_file << std::endl;
354 debug_file << "Normals " << std::fixed << std::setprecision(16) << v_normalsR_.at(block) << std::endl;
355 debug_file << "Block " << block + 1 << " (Reverse)";
356 debug_file << std::endl;
357 debug_file << "Precisions" << std::fixed << std::setprecision(16) << v_normalsR_.at(block) << std::endl;
358 debug_file << "Block " << block + 1 << " (Reverse)";
359 debug_file << std::endl;
360 debug_file << "Corrections" << std::fixed << std::setprecision(16) << v_correctionsR_.at(block) << std::endl;
361 debug_file.flush();
362
363 dbg_file_mutex.unlock();
364
365 }

369void adjust_forward_thread::operator()()
370{
371 std::stringstream ss;
372 UINT32 currentBlock;
373
374 concurrentAdjustments.forward_run_started();
375
376 try {
377 // Set exception
378 error_ = boost::exception_ptr();
379
380 for (currentBlock=0; currentBlock<main_adj_->blockCount_; ++currentBlock)
381 {
382 if (main_adj_->IsCancelled())
383 return;
384
385 //ss.str("");
386 //ss << "1> Adjusted block " << currentBlock + 1 << " (forward, in isolation)... " << std::endl;
387 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
388
389 // Check if an exception was thrown in the reverse or combine threads
390 if (rev_error || cmb_error)
391 return;
392
393 // At this point, whether first iteration or not, if currentBlock is the first block,
394 // the normals will have been initialised. For all later blocks, the normals will contain
395 // the contribution of junction station coordinates and variances from preceding blocks.
396 // In either case, the block is ready for adjustment. Junction station coordinates and
397 // variances are carried forward below
398
399 // Least Squares Solution
400 main_adj_->SolveTry(true, currentBlock);
401
402 // Since Solve() can be computationally intensive, re-check
403 // if an exception was thrown in the reverse or combine threads
404 if (rev_error || cmb_error)
405 return;
406
407 main_adj_->UpdateEstimatesForward(currentBlock);
408
411 //ss.str("");
412 //ss << "1> Forward adjustment (in isolation) of block " << currentBlock + 1 << " complete." << std::endl;
413
414 //adj_file_mutex.lock();
415 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
416 //adj_file_mutex.unlock();
418
419 // OK, now shrink matrices back to normal size
420 main_adj_->ShrinkForwardMatrices(currentBlock);
421
422 // This step is needed to carry coordinates and variance estimates for junctions only
423 // for the forward run. It is not needed for the combination stage
424 main_adj_->CarryForwardJunctions(currentBlock, currentBlock+1);
425
426 // Does this block need combining?
427 if (!main_adj_->CombineRequired(currentBlock))
428 continue;
429
430 // Ok finished adjustment, set adjusted
431 // state of this block in the forward list
432 concurrentAdjustments.set_forward_block_adjusted(currentBlock);
433
434 // Has the reverseThread dealt with this block?
435 if (concurrentAdjustments.reverse_block_adjusted(currentBlock))
436 // add this block # to the queue and notify the waiting
437 // combine thread
438 combineAdjustmentQueue.push_and_notify(currentBlock);
439
441 // protected write to adj file
442 //ss.str("");
443 //ss << "1> Forward adjustment of block " << currentBlock << " complete." << std::endl;
444
445 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
447 }
448
449 concurrentAdjustments.forward_run_finished();
450
451 if (concurrentAdjustments.finished_all_runs())
452 combineAdjustmentQueue.queue_exhausted();
453
454 // notify all threads waiting on combineAdjustmentQueue
455 combineAdjustmentQueue.notify_all();
456
457 // Reset exception
458 error_ = boost::exception_ptr();
459
460 }
461 catch (...) {
462 error_ = boost::current_exception();
463 if (combineAdjustmentQueue.is_empty())
464 combineAdjustmentQueue.push_and_notify(99999999);
465 //combineAdjustmentQueue.notify_all();
466 return;
467 }
468
469 // notify all threads waiting on combineAdjustmentQueue
470 //combineAdjustmentQueue.notify_all();
471
472 //this_thread::sleep(boost::posix_time::milliseconds(10));
473
474 // notify all threads waiting on combineAdjustmentQueue
475 //combineAdjustmentQueue.notify_all();
476}

479void adjust_reverse_thread::operator()()
480{
481 std::stringstream ss;
482 UINT32 currentBlock(main_adj_->blockCount_-1), block;
483
484 concurrentAdjustments.reverse_run_started();
485
486 try {
487 // Set exception
488 error_ = boost::exception_ptr();
489
490 for (block=0; block<main_adj_->blockCount_; ++block, --currentBlock)
491 {
492 if (main_adj_->IsCancelled())
493 return;
494
495 // Check if an exception was thrown in the reverse or combine threads
496 if (fwd_error || cmb_error)
497 return;
498
499 // if this is a single block, then there is no need to perform a reverse adjustment
500 if (!main_adj_->PrepareAdjustmentReverse(currentBlock, true))
501 continue;
502
505 //ss.str("");
506 //ss << "2> Adjusted block " << currentBlock + 1 << " (reverse, in isolation)... " << std::endl;
507 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
509
510 // Backup normals prior to inversion for re-use in combination
511 // adjustment... only if a combination is required
512 main_adj_->BackupNormals(currentBlock, true);
513
514 // At this point, whether first iteration or not, if currentBlock is the last block,
515 // the normals will have been initialised. For all subsequent blocks, the normals will contain
516 // the contribution of junction station coordinates and variances from preceding blocks.
517 // In either case, the block is ready for adjustment. Junction station coordinates and
518 // variances are carried in the reverse direction below
519
520 // Least Squares Solution
521 main_adj_->SolveMTTry(true, currentBlock);
522
523 // Since SolveMT() can be computationally intensive, re-check
524 // if an exception was thrown in the forward or combine threads
525 if (fwd_error || cmb_error)
526 return;
527
528 main_adj_->UpdateEstimatesReverse(currentBlock, true);
529
532 //ss.str("");
533 //ss << "2> Reverse adjustment (in isolation) of block " << currentBlock + 1 << " complete." << std::endl;
534
535 //adj_file_mutex.lock();
536 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
537 //adj_file_mutex.unlock();
539
540 // Now, carry the estimated junction station coordinates and variances
541 // to the next block (to be used in the next loop), except when currentBlock
542 // is the first block and currentBlock-1 is an isolated block.
543 //
544 // Remember - the junction station estimates and variances of the next block
545 // (obtained during the forward pass) were copied during the forward pass
546 // in CarryStnEstimatesandVariancesForward(..), so no need to re-copy.
547 main_adj_->CarryReverseJunctions(currentBlock, currentBlock-1, true);
548
549 // Does this block need combining?
550 if (!main_adj_->CombineRequired(currentBlock))
551 {
552 if (main_adj_->FirstBlock(currentBlock))
553 main_adj_->UpdateEstimatesFinal(currentBlock);
554
555 // This block does not need combination - continue
556 continue;
557 }
558
559 // Ok finished adjustment, set adjusted
560 // state of this block in the reverse list
561 concurrentAdjustments.set_reverse_block_adjusted(currentBlock);
562
563 // Has the forwardThread dealt with this block?
564 if (concurrentAdjustments.forward_block_adjusted(currentBlock))
565 // add this block # to the queue and notify the waiting
566 // combine thread
567 combineAdjustmentQueue.push_and_notify(currentBlock);
568
569
571 // protected write to adj file
572 //ss.str("");
573 //ss << "2> Reverse adjustment of block " << currentBlock << " complete." << std::endl;
574
575 //main_adj_->ThreadSafeWritetoAdjFile(ss.str());
577 }
578
579 concurrentAdjustments.reverse_run_finished();
580
581 if (concurrentAdjustments.finished_all_runs())
582 combineAdjustmentQueue.queue_exhausted();
583
584 // Reset exception
585 error_ = boost::exception_ptr();
586
587 }
588 catch (...) {
589 error_ = boost::current_exception();
590 if (combineAdjustmentQueue.is_empty())
591 combineAdjustmentQueue.push_and_notify(99999999);
592 return;
593 }

597void adjust_process_combine_thread::operator()()
598{
599 std::stringstream ss;
600 UINT32 pseudomsrJSLCount;
601 UINT32 currentBlock;
602
603 try {
604 while (true)
605 {
606 if (main_adj_->IsCancelled())
607 return;
608
609 if (combineAdjustmentQueue.is_queue_exhausted())
610 break;
611
612 // Have any other adjustments failed?
613 if (combineAdjustmentExceptionThrown(cmb_errors_))
614 return;
615
616 // Wait here until blocks have been placed on the queue
617 if (!combineAdjustmentQueue.front_and_pop(currentBlock))
618 {
619 boost::this_thread::sleep(boost::posix_time::milliseconds(2));
620 continue;
621 }
622
623 // Have any other adjustments failed?
624 if (combineAdjustmentExceptionThrown(cmb_errors_))
625 return;
626
627 main_adj_->isCombining_ = true;
628 main_adj_->SetcurrentBlock(currentBlock);
629
630 if (main_adj_->PrepareAdjustmentCombine(currentBlock, pseudomsrJSLCount, true))
631 {
632 // Least Squares Solution
633 main_adj_->SolveMTTry(true, currentBlock);
634 main_adj_->UpdateEstimatesCombine(currentBlock, pseudomsrJSLCount);
635 main_adj_->UpdateEstimatesFinal(currentBlock);
636 }
637 }
638
639 error_ = boost::exception_ptr();
640 }
641 catch (...) {
642 error_ = boost::current_exception();
643 return;
644 }
645}

647
648void adjust_combine_thread::operator()()
649{
650 main_adj_->isCombining_ = false;
651
652 // Wait here until notification has been received
653 combineAdjustmentQueue.wait_if_queue_is_empty();
654
655 // Has either a forward or reverse adjustment failed?
656 if (fwd_error || rev_error)
657 return;
658
659 // OK, at this point, there is a block # in the queue - let's combine!
660
661 // Get number of cores available
662 UINT32 thread_id, cores(boost::thread::hardware_concurrency());
663
664 // Set up exception pointers
665 std::vector<boost::exception_ptr> cmb_errors;
666 cmb_errors.resize(cores);
667
668 // Create the thread pool
669#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
670 boost::shared_ptr<thread> c;
671 std::vector< boost::shared_ptr<thread> > mt_combine_threads_sp;
672#else
673 std::vector<boost::thread> mt_combine_threads;
674#endif
675
676 for (thread_id=0; thread_id<cores; ++thread_id)
677 {
678
679#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
680 c.reset(new thread(
681 adjust_process_combine_thread(
682 main_adj_, // pointer to main adjustment object
683 thread_id, // this new thread's ID (1..n, where n = # cores)
684 boost::ref(cmb_errors.at(thread_id)), // reference to this thread's exception_ptr
685 boost::ref(cmb_errors)))); // reference to all other threads' exception_ptrs
686
687 mt_combine_threads_sp.push_back(c);
688#else
689 mt_combine_threads.push_back(boost::thread(
690 adjust_process_combine_thread(
691 main_adj_, // pointer to main adjustment object
692 thread_id, // this new thread's ID (1..n, where n = # cores)
693 boost::ref(cmb_errors.at(thread_id)), // reference to this thread's exception_ptr
694 boost::ref(cmb_errors)))); // reference to all other threads' exception_ptrs
695#endif
696
697 }
698
699#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
700 for_each(mt_combine_threads_sp.begin(), mt_combine_threads_sp.end(), boost::mem_fn(&thread::join));
701#else
702 for_each(mt_combine_threads.begin(), mt_combine_threads.end(), boost::mem_fn(&boost::thread::join));
703#endif
704
705 // cmb_error is set during combineThreadProcessor
706 for (UINT32 err=0; err<cores; ++err)
707 {
708 if (cmb_errors.at(err))
709 {
710 error_ = cmb_errors.at(err);
711 return;
712 }
713 }
714}

721{
722 std::stringstream ss;
723 UINT32 currentBlock;
724
725 try {
726 while (true)
727 {
728 if (prepareAdjustmentQueue.is_queue_exhausted())
729 break;
730
731 // Have any other adjustments failed?
732 if (prepareAdjustmentExceptionThrown(prep_errors_))
733 return;
734
735 // Wait here until blocks have been placed on the queue
736 if (!prepareAdjustmentQueue.front_and_pop(currentBlock))
737 {
738 boost::this_thread::sleep(boost::posix_time::milliseconds(2));
739 continue;
740 }
741
742 // Have any other adjustments failed?
743 if (prepareAdjustmentExceptionThrown(prep_errors_))
744 return;
745
746 main_adj_->SetcurrentBlock(currentBlock);
747 main_adj_->CreateMeasurementTally(currentBlock);
748 main_adj_->PrepareAdjustmentBlock(currentBlock, thread_id_);
749
750 if (prepareAdjustmentQueue.is_empty())
751 prepareAdjustmentQueue.queue_exhausted();
752 }
753
754 error_ = boost::exception_ptr();
755 }
756 catch (...) {
757 error_ = boost::current_exception();
758 }

762void adjust_prepare_thread::operator()()
763{
764 // load up the queue with the block ids
765 prepareAdjustmentQueue.push_data(createIncrementingIntegerVector<UINT32>(main_adj_->blockCount()));
766
767 // OK, at this point, there is a block # in the queue - let's combine!
768 // Get number of cores available
769 UINT32 thread_id, cores(boost::thread::hardware_concurrency());
770
771 // Set up exception pointers
772 std::vector<boost::exception_ptr> prep_errors;
773 prep_errors.resize(cores);
774
775 // Create the thread pool
776#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
777 boost::shared_ptr<thread> p;
778 std::vector< boost::shared_ptr<thread> > mt_prepare_threads_sp;
779#else
780 std::vector<boost::thread> mt_prepare_threads;
781#endif
782
783
784 for (thread_id=0; thread_id<cores; ++thread_id)
785 {
786#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
787 p.reset(new thread(
788 adjust_process_prepare_thread(
789 main_adj_, // pointer to main adjustment object
790 thread_id, // this new thread's ID (1..n, where n = # cores)
791 boost::ref(prep_errors.at(thread_id)), // reference to this thread's exception_ptr
792 boost::ref(prep_errors)))); // reference to all other threads' exception_ptrs
793
794 mt_prepare_threads_sp.push_back(p);
795#else
796 mt_prepare_threads.push_back(boost::thread(
797 adjust_process_prepare_thread(
798 main_adj_, // pointer to main adjustment object
799 thread_id, // this new thread's ID (1..n, where n = # cores)
800 boost::ref(prep_errors.at(thread_id)), // reference to this thread's exception_ptr
801 boost::ref(prep_errors)))); // reference to all other threads' exception_ptrs
802#endif
803 }
804
805#if defined(__ICC) || defined(__INTEL_COMPILER) // Intel compiler
806 for_each(mt_prepare_threads_sp.begin(), mt_prepare_threads_sp.end(), boost::mem_fn(&thread::join));
807#else
808 for_each(mt_prepare_threads.begin(), mt_prepare_threads.end(), boost::mem_fn(&boost::thread::join));
809#endif
810
811 // prep_error is set during prepareThreadProcessor
812 for (UINT32 err=0; err<cores; ++err)
813 {
814 if (prep_errors.at(err))
815 {
816 error_ = prep_errors.at(err);
817 return;
818 }
819 }
820}
