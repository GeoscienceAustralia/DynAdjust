57// Custom struct to manage occurrences of a station in a SINEX file
58//T1 file_index;
59//T1 solution_id;
60//T2 site_name;
61//T2 formatted_name;
62//T2 formatted_date;
63//T3 last_occurrence;
64template <class T1=UINT32, class T2=std::string, class T3=bool>
65struct site_id_tuple_t
66{
67 T1 file_index;
68 T1 solution_id;
69 T2 site_name;
70 T2 formatted_name;
71 T2 formatted_date;
72 T3 last_occurrence;
73
74 site_id_tuple_t()
75 : file_index(0), solution_id(0)
76 , site_name(""), formatted_name(""), formatted_date(""), last_occurrence(false) {}
77 site_id_tuple_t(const T1& index, const T1& solution,
78 const T2& name, const T2& formattedname, const T2& formatteddate, const T3& lastoccurrence)
79 : file_index(index), solution_id(solution)
80 , site_name(name), formatted_name(formattedname), formatted_date(formatteddate), last_occurrence(lastoccurrence) {}
81
82 template <class A, class B, class C>
83 site_id_tuple_t (const site_id_tuple_t<A, B, C> &s)
84 : file_index(s.file_index)
85 , solution_id(s.solution_id)
86 , site_name(s.site_name)
87 , formatted_date(s.formatted_date)
88 , formatted_name(s.formatted_name)
89 , last_occurrence(s.last_occurrence) {}
90};

97// Used to sort site tuples by file order
98// Can be used to sort site_id elements or discontinuity elements
99template <typename T>
100class CompareSiteTuples {
101public:
102 bool operator()(const T& left, const T& right) {
103 // sort on file order
104 return left.file_index < right.file_index;
105 }
106};
107
108// Used to sort site tuples by original name
109// Can be used to sort site_id elements or discontinuity elements
110template <typename T>
111class CompareSiteTuplesByName {
112public:
113 bool operator()(const T& left, const T& right) {
114 // if the station names are equal
115 if (boost::equals(left.site_name, right.site_name))
116 // sort on file order
117 return left.file_index < right.file_index;
118 // sort on station name
119 return left.site_name < right.site_name;
120 }
121};

124// Custom struct to manage site discontinuities
125template <class T1=UINT32, class T2=std::string, class T3=boost::gregorian::date, class T4=bool>
126struct discontinuity_tuple_t
127{
128 T1 file_index;
129 T1 solution_id;
130 T2 site_name;
131 T3 date_start;
132 T3 date_end;
133 T4 discontinuity_exists;
134
135 discontinuity_tuple_t()
136 : file_index(0), solution_id(0), site_name("")
137 , date_start(boost::gregorian::date(TIME_IMMEMORIAL, 1, 1)), date_end(boost::gregorian::date(TIME_IMMEMORIAL, 1, 1)), discontinuity_exists(false) {}
138 discontinuity_tuple_t(const T1& index, const T1& solution, const T2& name,
139 const T3& from, const T3& to, const T4& discontinuity)
140 : file_index(index), solution_id(solution), site_name(name)
141 , date_start(from), date_end(to), discontinuity_exists(discontinuity) {}
142
143 template <class A, class B, class C, class D>
144 discontinuity_tuple_t (const discontinuity_tuple_t<A, B, C, D> &d)
145 : file_index(d.file_index)
146 , solution_id(d.solution_id)
147 , site_name(d.site_name)
148 , date_start(d.date_start)
149 , date_end(d.date_end)
150 , discontinuity_exists(d.discontinuity_exists) {}
151};

158template<typename T, typename S, typename D>
159bool rename_discont_station(T& begin, S& site_name, D& site_date, S& site_renamed)
160{
161 UINT32 doy, year;
162
163 // find the next occurrence of this site
164 while (boost::equals(site_name, begin->site_name))
165 {
166 // Check if discontinuities exist at this site, if not, skip
167 if (!begin->discontinuity_exists)
168 {
169 begin++;
170 continue;
171 }
172
173 // Test if the start epoch of this site is within this discontinuity window
174 if (site_date >= begin->date_start &&
175 site_date < begin->date_end)
176 {
177 std::stringstream ss;
178 doy = begin->date_start.day_of_year();
179 year = begin->date_start.year();
180 ss << site_name << "_";
181
182 ss << year;
183 if (doy < 100)
184 ss << "0";
185 if (doy < 10)
186 ss << "0";
187 ss << doy;
188
189 site_renamed = ss.str();
190
191 return true;
192 }
193
194 begin++;
195 }
196
197 return false;
198}

217// tweak the binary search so it returns the iterator of the object found
218template<typename T, typename S>
219T binary_search_discontinuity_site(T begin, T end, S value)
220{
221 T i = lower_bound(begin, end, value, CompareDiscontinuityOnSite<discontinuity_tuple, std::string>());
222 if (i != end && i->site_name == value)
223 return i;
224 else
225 return end;
226}
