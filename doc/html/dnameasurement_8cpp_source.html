147{

189UINT32 CDnaCovariance::SetMeasurementRec(const vstn_t&, it_vmsr_t& it_msr)
190{
191 // get data relating to Z, sigmaZZ, sigmaXZ, sigmaYZ
192 it_msr++;
193 m_lstn1Index = it_msr->station1;
194 m_lstn2Index = it_msr->station2;
195 m_strType = it_msr->measType;
196 m_dM11 = it_msr->term1;
197 m_dM12 = it_msr->term2;
198 m_dM13 = it_msr->term3;
199
200 m_lclusterID = it_msr->clusterID;
201
202 // get data relating to Z, sigmaZZ, sigmaXZ, sigmaYZ
203 it_msr++;
204
205 m_lstn1Index = it_msr->station1;
206 m_lstn2Index = it_msr->station2;
207 m_dM21 = it_msr->term1;
208 m_dM22 = it_msr->term2;
209 m_dM23 = it_msr->term3;
210
211 // get data relating to Z, sigmaZZ, sigmaXZ, sigmaYZ
212 it_msr++;
213
214 m_lstn1Index = it_msr->station1;
215 m_lstn2Index = it_msr->station2;
216 m_dM31 = it_msr->term1;
217 m_dM32 = it_msr->term2;
218 m_dM33 = it_msr->term3;
219
220 return 0;
221}

224void CDnaCovariance::WriteBinaryMsr(std::ofstream *binary_stream, PUINT32 msrIndex, const std::string& epsgCode, const std::string& epoch) const
225{
226 *msrIndex += 3;
227 measurement_t measRecord;
228
229 // Common
230 measRecord.measType = GetTypeC();
231 measRecord.station1 = m_lstn1Index;
232 measRecord.station2 = m_lstn2Index;
233 measRecord.clusterID = m_lclusterID;
234
235 snprintf(measRecord.epsgCode, sizeof(measRecord.epsgCode), "%s", epsgCode.substr(0, STN_EPSG_WIDTH).c_str());
236 snprintf(measRecord.epoch, sizeof(measRecord.epoch), "%s", epoch.substr(0, STN_EPOCH_WIDTH).c_str());
237
238 // X
239 measRecord.measStart = xCov;
240 measRecord.term1 = m_dM11;
241 measRecord.term2 = m_dM12;
242 measRecord.term3 = m_dM13;
243
244 binary_stream->write(reinterpret_cast<char *>(&measRecord), sizeof(measurement_t));
245
246 // Y
247 measRecord.measStart = yCov;
248 measRecord.term1 = m_dM21;
249 measRecord.term2 = m_dM22;
250 measRecord.term3 = m_dM23;
251
252 binary_stream->write(reinterpret_cast<char *>(&measRecord), sizeof(measurement_t));
253
254 // Z
255 measRecord.measStart = zCov;
256 measRecord.term1 = m_dM31;
257 measRecord.term2 = m_dM32;
258 measRecord.term3 = m_dM33;
259
260 binary_stream->write(reinterpret_cast<char *>(&measRecord), sizeof(measurement_t));
261}

263void CDnaCovariance::SerialiseDatabaseMap(std::ofstream* os, const msr_database_id_map& dbid)
264{
265 UINT16 msr, cls;
266 msr = static_cast<UINT16>(dbid.is_msr_id_set);
267 cls = static_cast<UINT16>(dbid.is_cls_id_set);
268
269 // X
270 os->write(reinterpret_cast<const char *>(&dbid.msr_id), sizeof(UINT32));
271 os->write(reinterpret_cast<const char *>(&dbid.cluster_id), sizeof(UINT32));
272 os->write(reinterpret_cast<const char *>(&msr), sizeof(UINT16));
273 os->write(reinterpret_cast<const char *>(&cls), sizeof(UINT16));
274 // Y
275 os->write(reinterpret_cast<const char *>(&dbid.msr_id), sizeof(UINT32));
276 os->write(reinterpret_cast<const char *>(&dbid.cluster_id), sizeof(UINT32));
277 os->write(reinterpret_cast<const char *>(&msr), sizeof(UINT16));
278 os->write(reinterpret_cast<const char *>(&cls), sizeof(UINT16));
279 // Z
280 os->write(reinterpret_cast<const char *>(&dbid.msr_id), sizeof(UINT32));
281 os->write(reinterpret_cast<const char *>(&dbid.cluster_id), sizeof(UINT32));
282 os->write(reinterpret_cast<const char *>(&msr), sizeof(UINT16));
283 os->write(reinterpret_cast<const char *>(&cls), sizeof(UINT16));
284}
