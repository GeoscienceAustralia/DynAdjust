192 HDDEDATA hdata, DWORD dwData1, DWORD dwData2)

208int StartServer(LPCTSTR prog)
209{
210 STARTUPINFO si;
211 PROCESS_INFORMATION pi;
212
213 /* To be sure ... */
214 if (! prog)
215 return 1;
216
217 ZeroMemory (&si, sizeof (si));
218 si.cb = sizeof (si);
219 if (!CreateProcess (NULL, (LPTSTR)prog, NULL, NULL, FALSE, 0,
220 NULL, NULL, &si, &pi))
221 {
222 error("Could not start process %s", prog);
223 return 1;
224 }
225
226 /* Wait for the process to enter an idle state */
227 WaitForInputIdle (pi.hProcess, MAX_INPUT_IDLE_WAIT);
228
229 /* Close the handles */
230 CloseHandle (pi.hThread);
231 CloseHandle (pi.hProcess);
232 return 0;
233}

248void OpenConversation(LPCTSTR topic_name)
249{
250 UINT ret;
251 HSZ service;
252 HSZ topic;
253 int n;
254
255 ret = DdeInitialize(&DDEsession, Callback, APPCLASS_STANDARD| APPCMD_CLIENTONLY, 0);
256 if (ret != DMLERR_NO_ERROR)
257 error("Cannot start DDE");
258
259 service = DDEString(ACROBAT);
260 topic = DDEString(topic_name);
261 DDEconversation = DdeConnect(DDEsession, service, topic, 0);
262
263 if (DDEconversation == 0)
264 {
265 if (StartServer(FindAcrobatModule()) == 0)
266 {
267 /* Try to connect */
268 for (n = 0; n < 5; n++)
269 {
270 Sleep (CONNECT_DELAY);
271 DDEconversation = DdeConnect(DDEsession, service, topic, 0);
272 if (DDEconversation)
273 break;
274 }
275 if (DDEconversation == 0)
276 error("Cannot contact server %s", ACROBAT);
277 }
278 }
279 FreeString(service);
280 FreeString(topic);
281}

283bool __cdecl ExecuteCommand(LPCTSTR command, ...)
284{
285 HDDEDATA ret;
286 va_list args;
287 int len;
288 char *buffer;
289
290 va_start( args, command );
291 len = _vscprintf(command, args) // _vscprintf doesn't count
292 + 1; // terminating '\0'
293 buffer = (char *)malloc( len * sizeof(char) );
294 vsprintf( buffer, command, args );
295
296 va_end(args);
297
298 ret = DdeClientTransaction((LPBYTE)buffer, strlen(buffer) + 1,
299 DDEconversation, 0, 0, XTYP_EXECUTE,
300 TRANSACTION_TIMEOUT, 0);
301 if (ret == 0)
302 {
303 UINT DDEErr = DdeGetLastError(DDEsession);
304 error("Cannot execute command \"%s\" (error %ld)", command, DDEErr);
305 return(false);
306 } // if
307
308 return(true);
309}

357char *CheckFileName(char *p)
358{
359 char sTmp[MAXPATHLEN];
360 char *fp;
361
362 sLongFileName[0] = '\0';
363
364 if (GetFileAttributes(p) == -1) {
365 fprintf(stderr, "%s: non existent file %s\n", argv0, p);
366 return NULL;
367 }
368 if (GetFullPathName(p, sizeof(sTmp), sTmp, &fp) == 0) {
369 fprintf(stderr, "%s: failed to get full path name for %s (Error %d)\n", argv0, p, GetLastError());
370 return NULL;
371 }
372 if (GetLongPathName(sTmp, sLongFileName, sizeof(sLongFileName)) == 0) {
373 fprintf(stderr, "%s: failed to get long path name for %s (Error %d)\n", argv0, sTmp, GetLastError());
374 return NULL;
375 }
376 return sLongFileName;

447int pdfserver()
448{
449 char line[1024];
450
451 OpenConversation(EVAL_TOPIC);
452
453 fprintf(stdout, "Type Ctrl-Z to quit.\n");
454
455 while (fgets(line, sizeof(line), stdin)) {
456#define WHITE_DELIM " \t\n"
457 char *p = strtok(line, WHITE_DELIM);
458 if (stricmp(p, "open") == 0) {
459 p = strtok(NULL, WHITE_DELIM);
460 if (CheckFileName(p) == NULL) goto next;
461
462 ExecuteCommand("[DocOpen(\"%s\")]", sLongFileName);
463 ExecuteCommand("[DocOpen(\"%s\")]", sLongFileName); /* acrord32 ver8 bug */
464 }
465 else if (stricmp(p, "close") == 0) {
466 p = strtok(NULL, WHITE_DELIM);
467 if (CheckFileName(p) == NULL) goto next;
468 ExecuteCommand("[DocClose(\"%s\")]", sLongFileName);
469 }
470 else if (stricmp(p, "closeall") == 0) {
471 ExecuteCommand("[CloseAllDocs()]");
472 }
473 else if (stricmp(p, "goto") == 0) {
474 p = strtok(NULL, WHITE_DELIM);
475 if (p == NULL) goto next;
476 if (CheckFileName(p) == NULL) goto next;
477 p = strtok(NULL, WHITE_DELIM);
478 if (p == NULL) goto next;
479 nPage = atoi(p) - 1;
480 ExecuteCommand("[DocGoTo(\"%s\", %d)]", sLongFileName, nPage);
481 }
482 else if (stricmp(p, "gotoname") == 0) {
483 p = strtok(NULL, WHITE_DELIM);
484 if (p == NULL) goto next;
485 if (CheckFileName(p) == NULL) goto next;
486 p = strtok(NULL, WHITE_DELIM);
487 if (p == NULL) goto next;
488 sGoto = p;
489 ExecuteCommand("[DocGoToNameDest(\"%s\", %s)]", sLongFileName, sGoto);
490 }
491 else if (stricmp(p, "show") == 0) {
492 ExecuteCommand("[AppShow()]");
493
494 }
495 else if (stricmp(p, "hide") == 0) {
496 ExecuteCommand("[AppHide()]");
497
498 }
499 else if (stricmp(p, "exit") == 0) {
500 ExecuteCommand("[AppExit()]");
501 goto exit;
502 }
503 else if (stricmp(p, "back") == 0) {
504 ExecuteCommand("[MenuitemExecute(\"GoBack\")]");
505 }
506 else if (stricmp(p, "list") == 0) {
507 fprintf(stdout, "open <file> open the file\n");
508 fprintf(stdout, "close <file> close the file\n");
509 fprintf(stdout, "closeall close all files\n");
510 fprintf(stdout, "goto <file> <page> goto the given <page> in <file>\n");
511 fprintf(stdout, "gotoname <file> <dest> goto the <dest> named destination in <file>\n");
512 fprintf(stdout, "back goto the previous view\n");
513 fprintf(stdout, "show show acroview\n");
514 fprintf(stdout, "hide hide acroview\n");
515 fprintf(stdout, "exit exit acroview and %s\n", argv0);
516 }
517 next: ;
518 }
519
520#if 0
521 fprintf(stderr, "cmd : %s\n", buf);
522#endif
523exit:
524 CloseConversation();
525
526 return EXIT_SUCCESS;
