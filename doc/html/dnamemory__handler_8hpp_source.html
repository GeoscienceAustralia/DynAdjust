56 new_handler_holder(const new_handler_holder&); // prevent copying
57 new_handler_holder& operator=(const new_handler_holder&); // '' ''

60template<typename T>
61class new_handler_support {
62public:
63 static std::new_handler set_new_handler(std::new_handler p) noexcept;
64 static void* operator new(size_t size) noexcept(false);
65
66private:
67 static std::new_handler _current_handler;
68};
69
70template<typename T>
71std::new_handler new_handler_support<T>::set_new_handler(std::new_handler p) noexcept
72{
73 std::new_handler old_handler = _current_handler;
74 _current_handler = p;
75 return old_handler;
76}
77
78template<typename T>
79void* new_handler_support<T>::operator new(size_t size) noexcept(false)
80{
81 new_handler_holder h(std::set_new_handler(_current_handler));
82 return ::operator new(size);
83}

87std::new_handler new_handler_support<T>::_current_handler = 0;
