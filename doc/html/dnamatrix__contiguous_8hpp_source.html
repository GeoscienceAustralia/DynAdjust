131/*
132 By default uses column-major order
133*/
134#define DNAMATRIX_INDEX(num_rows, num_cols, row, column) \
135 ((column) * (num_rows) + (row))
136
137/*
138 Retrieves an element based on DNAMATRIX_INDEX based indexing
139*/
140#define DNAMATRIX_ELEMENT(A, no_rows, no_cols, row, column) \
141 A[DNAMATRIX_INDEX(no_rows, no_cols, row, column)]
142
143/*
144 byteSize template: calculates memory size for 'elements' of type T
145*/
146template <typename T> std::size_t byteSize(const index_t elements = 1) {
147 return elements * sizeof(T);
148}

150/*
151 A matrix implementation using column-major storage.

153class matrix_2d : public new_handler_support<matrix_2d> {

168 /* Copy constructor: makes a deep copy of newmat */

191
192 /* Returns pointer to underlying buffer */
193 inline double* getbuffer() const { return _buffer; }
194
195 /* Returns reference to element at row r, column c */
196 inline double& get(const index_t& r, const index_t& c) const {
197 return DNAMATRIX_ELEMENT(_buffer, _mem_rows, _mem_cols, r, c);
198 }
199
200 /* Returns pointer to element at row r, column c */
201 inline double* getbuffer(const index_t& r, const index_t& c) const {
202 return _buffer + DNAMATRIX_INDEX(_mem_rows, _mem_cols, r, c);

217 /* Returns the current maximum value in the matrix */
218 inline double maxvalue() const { return get(_maxvalRow, _maxvalCol); }

220 /* Returns the row index of the current maximum value */
221 inline index_t maxvalueRow() const { return _maxvalRow; }
222
223 /* Returns the column index of the current maximum value */
224 inline index_t maxvalueCol() const { return _maxvalCol; }
225
226 /* Returns pointer to the element reference at row r, column c */
227 inline double* getelementref(const index_t& r, const index_t& c) const {
228 return &(DNAMATRIX_ELEMENT(_buffer, _mem_rows, _mem_cols, r, c));
229 }
230
231 /* Returns pointer to the element reference at row r, column c (non-const) */
232 inline double* getelementref(const index_t& r, const index_t& c) {
233 return &(DNAMATRIX_ELEMENT(_buffer, _mem_rows, _mem_cols, r, c));
234 }
235
236 /* Sets the total allocated rows (memory) */
237 inline void mem_rows(const index_t& r) { _mem_rows = r; }

239 /* Sets the total allocated columns (memory) */

242 /* Sets the actual number of rows */
243 inline void rows(const index_t& r) { _rows = r; }
244
245 /* Sets the actual number of columns */

247

250

253

375 /* Equality operator */
376 bool operator==(const matrix_2d& rhs) const {
377 if (_mem_cols != rhs._mem_cols)
378 return false;
379 if (_mem_rows != rhs._mem_rows)
380 return false;
381 if (_cols != rhs._cols)
382 return false;
383 if (_rows != rhs._rows)
384 return false;
385 if (*_buffer != *rhs._buffer)
386 return false;
387 if (_maxvalCol != rhs._maxvalCol)
388 return false;
389 if (_maxvalRow != rhs._maxvalRow)
390 return false;
391 if (_matrixType != rhs._matrixType)
392 return false;
393 return true;
394 }
395
396 /* Inequality operator */
397 bool operator!=(const matrix_2d& rhs) const {
398 if (_mem_cols == rhs._mem_cols)
399 return false;
400 if (_mem_rows == rhs._mem_rows)
401 return false;
402 if (_cols == rhs._cols)
403 return false;
404 if (_rows == rhs._rows)
405 return false;
406 if (*_buffer == *rhs._buffer)
407 return false;
408 if (_maxvalCol == rhs._maxvalCol)
409 return false;
410 if (_maxvalRow == rhs._maxvalRow)
411 return false;
412 if (_matrixType == rhs._matrixType)
413 return false;
414 return true;
415 }
