58bool CDnaAngle::operator< (const CDnaAngle& rhs) const
59{
60 if (m_strFirst == rhs.m_strFirst) {
61 if (m_strTarget == rhs.m_strTarget) {
62 if (m_strTarget2 == rhs.m_strTarget2) {
63 if (m_bIgnore == rhs.m_bIgnore) {
64 if (m_epoch == rhs.m_epoch) {
65 if (m_drValue == rhs.m_drValue)
66 return m_dStdDev < rhs.m_dStdDev;
67 else
68 return m_drValue < rhs.m_drValue; }
69 else
70 return m_epoch < rhs.m_epoch; }
71 else
72 return m_bIgnore < rhs.m_bIgnore; }
73 else
74 return m_strTarget2 < rhs.m_strTarget2; }
75 else
76 return m_strTarget < rhs.m_strTarget; }
77 else
78 return m_strFirst < rhs.m_strFirst;
79}

116void CDnaAngle::WriteDNAMsr(std::ofstream* dna_stream, const dna_msr_fields& dmw, const dna_msr_fields& dml, bool) const
117{
118 *dna_stream << std::setw(dmw.msr_type) << m_strType;
119 if (m_bIgnore)
120 *dna_stream << std::setw(dmw.msr_ignore) << "*";
121 else
122 *dna_stream << std::setw(dmw.msr_ignore) << " ";
123 *dna_stream << std::left << std::setw(dmw.msr_inst) << m_strFirst;
124 *dna_stream << std::left << std::setw(dmw.msr_targ1) << m_strTarget;
125 *dna_stream << std::left << std::setw(dmw.msr_targ2) << m_strTarget2;
126 *dna_stream << std::setw(dmw.msr_linear) << " "; // linear measurement value
127 *dna_stream << std::setw(dmw.msr_ang_d + dmw.msr_ang_m + dmw.msr_ang_s) <<
128 std::right << FormatDnaDmsString(RadtoDms(m_drValue), 8);
129
130 UINT32 m_stdDevPrec(3);
131 *dna_stream << std::setw(dmw.msr_stddev) << StringFromTW(Seconds(m_dStdDev), dmw.msr_stddev, m_stdDevPrec);
132 //*dna_stream << std::setw(dmw.msr_stddev) << std::fixed << std::setprecision(3) << Seconds(m_dStdDev);
133
134 *dna_stream << std::setw(dml.msr_gps_epoch - dml.msr_inst_ht) << " ";
135 *dna_stream << std::setw(dmw.msr_gps_epoch) << m_epoch;
136
137 if (m_msr_db_map.is_msr_id_set)
138 *dna_stream << std::setw(dmw.msr_id_msr) << m_msr_db_map.msr_id;
139
140 *dna_stream << std::endl;
141}

144void CDnaAngle::SimulateMsr(vdnaStnPtr* vStations, const CDnaEllipsoid*)
145{
146 _it_vdnastnptr stn1_it(vStations->begin() + m_lstn1Index);
147 _it_vdnastnptr stn2_it(vStations->begin() + m_lstn2Index);
148 _it_vdnastnptr stn3_it(vStations->begin() + m_lstn3Index);
149
150 double direction12, direction13;
151
152 // calculated angle
153 m_drValue = HorizontalAngle(
154 stn1_it->get()->GetXAxis(),
155 stn1_it->get()->GetYAxis(),
156 stn1_it->get()->GetZAxis(),
157 stn2_it->get()->GetXAxis(),
158 stn2_it->get()->GetYAxis(),
159 stn2_it->get()->GetZAxis(),
160 stn3_it->get()->GetXAxis(),
161 stn3_it->get()->GetYAxis(),
162 stn3_it->get()->GetZAxis(),
163 stn1_it->get()->GetcurrentLatitude(),
164 stn1_it->get()->GetcurrentLongitude(),
165 &direction12, &direction13);
166
167 // Deflections available?
168 if (fabs(stn1_it->get()->GetverticalDef()) > E4_SEC_DEFLECTION || fabs(stn1_it->get()->GetmeridianDef()) > E4_SEC_DEFLECTION)
169 {
171 // Angles (observed or derived from directions) must be corrected for deflection
172 // of the vertical via "Laplace correction". This correction requires zenith
173 // distance (zenith12, zenith13) and geodetic azimuth (direction12, direction13),
174 // both of which must be computed from coordinates.
175
176 // 1. Compute zenith distance 1 -> 2
177 double zenith12(ZenithDistance<double>(
178 stn1_it->get()->GetXAxis(),
179 stn1_it->get()->GetYAxis(),
180 stn1_it->get()->GetZAxis(),
181 stn2_it->get()->GetXAxis(),
182 stn2_it->get()->GetYAxis(),
183 stn2_it->get()->GetZAxis(),
184 stn1_it->get()->GetcurrentLatitude(),
185 stn1_it->get()->GetcurrentLongitude(),
186 stn2_it->get()->GetcurrentLatitude(),
187 stn2_it->get()->GetcurrentLongitude(),
188 0.0, // instrument height
189 0.0)); // target height
190
191 // 2. Compute zenith distance 1 -> 3
192 double zenith13(ZenithDistance<double>(
193 stn1_it->get()->GetXAxis(),
194 stn1_it->get()->GetYAxis(),
195 stn1_it->get()->GetZAxis(),
196 stn3_it->get()->GetXAxis(),
197 stn3_it->get()->GetYAxis(),
198 stn3_it->get()->GetZAxis(),
199 stn1_it->get()->GetcurrentLatitude(),
200 stn1_it->get()->GetcurrentLongitude(),
201 stn3_it->get()->GetcurrentLatitude(),
202 stn3_it->get()->GetcurrentLongitude(),
203 0.0, // instrument height
204 0.0)); // target height
205
206 // 3. Laplace correction 1 -> 2 -> 3
207 m_preAdjCorr = HzAngleDeflectionCorrection<double>(
208 direction12, // geodetic azimuth 1 -> 2
209 zenith12, // zenith distance 1 -> 2
210 direction13, // geodetic azimuth 1 -> 3
211 zenith13, // zenith distance 1 -> 3
212 stn1_it->get()->GetverticalDef(), // deflection in prime vertical
213 stn1_it->get()->GetmeridianDef()); // deflection in prime meridian
214
215 // 4. apply deflection correction
216 m_drValue += m_preAdjCorr;
217 }
218
219 // compute standard deviation in radians
220 m_dStdDev = SecondstoRadians(0.01);
221
222 m_epoch = "01.10.1985";
223}

226UINT32 CDnaAngle::SetMeasurementRec(const vstn_t& binaryStn, it_vmsr_t& it_msr, it_vdbid_t& dbidmap)
227{
228 m_strType = it_msr->measType;
229 m_bIgnore = it_msr->ignore;
230 m_MSmeasurementStations = (MEASUREMENT_STATIONS)it_msr->measurementStations;
231
232 // first station
233 m_lstn1Index = it_msr->station1;
234 m_strFirst = binaryStn.at(it_msr->station1).stationName;
235
236 // target station
237 m_lstn2Index = it_msr->station2;
238 m_strTarget = binaryStn.at(it_msr->station2).stationName;
239
240 // target2 station
241 m_lstn3Index = it_msr->station3;
242 m_strTarget2 = binaryStn.at(it_msr->station3).stationName;
243
244 m_measAdj = it_msr->measAdj;
245 m_measCorr = it_msr->measCorr;
246 m_measAdjPrec = it_msr->measAdjPrec;
247 m_residualPrec = it_msr->residualPrec;
248 m_preAdjCorr = it_msr->preAdjCorr;
249 m_drValue = it_msr->term1;
250 m_dStdDev = sqrt(it_msr->term2);
251
252 m_epoch = it_msr->epoch;
253
254 CDnaMeasurement::SetDatabaseMap(*dbidmap);
255
256 return 0;
257}

259void CDnaAngle::WriteBinaryMsr(std::ofstream* binary_stream, PUINT32 msrIndex) const
260{
261 measurement_t measRecord;
262 measRecord.measType = GetTypeC();
263 measRecord.measStart = xMeas;
264 measRecord.ignore = m_bIgnore;
265 measRecord.station1 = m_lstn1Index;
266 measRecord.station2 = m_lstn2Index;
267 measRecord.station3 = m_lstn3Index;
268 measRecord.measAdj = m_measAdj;
269 measRecord.measCorr = m_measCorr;
270 measRecord.measAdjPrec = m_measAdjPrec;
271 measRecord.residualPrec = m_residualPrec;
272 measRecord.preAdjCorr = m_preAdjCorr;
273 measRecord.term1 = m_drValue;
274 measRecord.term2 = m_dStdDev * m_dStdDev; // convert to variance
275 measRecord.measurementStations = m_MSmeasurementStations;
276 measRecord.fileOrder = ((*msrIndex)++);
277
278 snprintf(measRecord.epoch, sizeof(measRecord.epoch), "%s", m_epoch.substr(0, STN_EPOCH_WIDTH).c_str());
279
280 binary_stream->write(reinterpret_cast<char *>(&measRecord), sizeof(measurement_t));
281}
