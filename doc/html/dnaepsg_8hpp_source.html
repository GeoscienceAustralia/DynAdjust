119template <typename U, typename S>

121{
122 if (boost::iequals(datumName, AGD66_s))

125 return AGD84_i;
126 if (boost::iequals(datumName, GDA94_s))
127 return GDA94_i_xyz;
128 if (boost::iequals(datumName, GDA2020_s))
129 return GDA2020_i_xyz;
130 // ITRF
131 if (boost::iequals(datumName, ITRF2020_s))
132 return ITRF2020_i_xyz;
133 if (boost::iequals(datumName, ITRF2014_s))
134 return ITRF2014_i_xyz;
135 if (boost::iequals(datumName, ITRF2008_s))
136 return ITRF2008_i_xyz;
137 if (boost::iequals(datumName, ITRF2005_s))
138 return ITRF2005_i_xyz;
139 if (boost::iequals(datumName, ITRF2000_s))
140 return ITRF2000_i_xyz;
141 if (boost::iequals(datumName, ITRF1997_s) || boost::iequals(datumName, ITRF1997_s_brief))
142 return ITRF1997_i_xyz;
143 if (boost::iequals(datumName, ITRF1996_s) || boost::iequals(datumName, ITRF1996_s_brief))
144 return ITRF1996_i_xyz;
145 if (boost::iequals(datumName, ITRF1994_s) || boost::iequals(datumName, ITRF1994_s_brief))
146 return ITRF1994_i_xyz;

154 return ITRF1990_i_xyz;

161 boost::iequals(datumName, WGS84_ensemble_s) ||
162 boost::iequals(datumName, WGS84_alias_s))
163 return WGS84_i_xyz;

165 boost::iequals(datumName, WGS84_transit_alias_s))
166 return WGS84_transit_i_xyz;
167 if (boost::iequals(datumName, WGS84_G730_s) ||
168 boost::iequals(datumName, WGS84_G730_alias_s))
169 return WGS84_G730_i_xyz;
170 if (boost::iequals(datumName, WGS84_G873_s) ||
171 boost::iequals(datumName, WGS84_G873_alias_s))
172 return WGS84_G873_i_xyz;
173 if (boost::iequals(datumName, WGS84_G1150_s) ||
174 boost::iequals(datumName, WGS84_G1150_alias_s))
175 return WGS84_G1150_i_xyz;

178 return WGS84_G1674_i_xyz;
179 if (boost::iequals(datumName, WGS84_G1762_s) ||
180 boost::iequals(datumName, WGS84_G1762_alias_s))
181 return WGS84_G1762_i_xyz;
182 if (boost::iequals(datumName, WGS84_G2139_s) ||
183 boost::iequals(datumName, WGS84_G2139_alias_s))
184 return WGS84_G2139_i_xyz;
185
186 std::stringstream ss;
187 ss << " epsgCodeFromName: '" << datumName << "' is not a supported reference frame label." << std::endl;
188 throw boost::enable_current_exception(std::runtime_error(ss.str()));
189}
