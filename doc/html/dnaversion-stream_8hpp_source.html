87template <class T>
88void output_build(T& stream, bool PAD=false)
89{
90 if (PAD)
91 stream << std::setw(PRINT_VAR_PAD) << std::left << "Build: ";
92 else
93 stream << "Build: ";
94 stream <<
95 // __DATE__ The compilation date of the current source file.
96 // The date is a constant length string literal of the form
97 // Mmm dd yyyy. The month name Mmm is the same as the abbreviated
98 // month name generated by the C Runtime Library(CRT) asctime
99 // function.The first character of date dd is a space if the
100 // value is less than 10. This macro is always defined.
101 __DATE__ << ", " <<
102 // __TIME__ The time of translation of the preprocessed
103 // translation unit. The time is a character string literal of
104 // the form hh:mm:ss, the same as the time returned by the CRT
105 // asctime function. This macro is always defined.
106 __TIME__;
107 if (strlen(__COMPILER__) > 2)
108 stream << " - " << __COMPILER__ << " " << __COMPILER_VERSION__ << "";
109}
