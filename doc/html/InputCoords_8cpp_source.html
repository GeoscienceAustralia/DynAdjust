144CString CInputCoords::AddSymbology(CString strNumber, bool bLatisPosSouth, int iFlag)
145{
146 int precision, decimal;
147 CString strBuf;
148
149 strBuf = strNumber;
150
151 decimal = strNumber.Find('.');
152
153 //iFlag: (1) Lat/East Input, (2) Long/North Input, (3) Lat/East Output, (4) Long/North Output
154 precision = strBuf.GetLength() - decimal - 1;
155
156 if (m_iDMS == DMS)
157 {
158 // Add symbols for degrees minutes and seconds
159 if (decimal > -1)
160 {
161 // found a decimal point!
162 if (decimal == 0) // decimal point at start, ie >> .0123
163 {
164 strBuf.Insert(decimal, '0');
165 decimal++;
166 }
167
168 strBuf.SetAt(decimal, ' ');
169 strBuf.Insert(decimal, '°');
170
171 // add zero after "tens" minute or "tens" second value
172 if ((iFlag == 1 || iFlag == 2) && (precision == 1 || precision == 3) )
173 {
174 strBuf += "0";
175 precision++;
176 }
177
178 if (precision > 1) // add minute symbol
179 strBuf.Insert((decimal+4), "\' ");
180
181 if (precision > 3)
182 strBuf.Insert((decimal+8), '"');
183
184 if (precision > 4)
185 {
186 strBuf.SetAt((decimal+8), '.');
187 strBuf += "\"";
188 }
189
190 }
191 else
192 strBuf += "°"; // couldn't find a decimal point
193 }
194 else
195 {
196 // must be decimal degrees format...simply truncate value at required precision
197 if (decimal == 0)
198 {
199 strBuf.Insert(decimal, '0');
200 decimal++;
201 }
202
203 strBuf = strBuf.Mid(0, decimal + precision + 1);
204 strBuf += "°";
205 }
206
207 if (iFlag < 3 || iFlag > 2)
208 {
209 // Show North and South notation
210 if (strNumber[0] == '-' || strNumber[0] == 's' || strNumber[0] == 'S' ||
211 strNumber[0] == 'w' || strNumber[0] == 'W')
212 {
213 if (iFlag == 1 || iFlag == 3) // input/output latitude
214 strBuf.SetAt(0, 'S');
215 else // input/output longitude
216 strBuf.SetAt(0, 'W');
217 }
218 else
219 {
220 if (iFlag == 1 || iFlag == 3)
221 strBuf = "N" + strBuf;
222 else
223 strBuf = "E" + strBuf;
224 }
225
226 strBuf = strBuf.Mid(0, 1) + " " + strBuf.Mid(1);
227 }
228
229 return strBuf;
230}

255LRESULT CInputCoords::OnUpdateInterpolationResults(WPARAM wParam, LPARAM lParam)
256{
257 char buf[50];
258
259 sprintf(buf, "%.3f", m_lfNvalue);
260 GetDlgItem(IDC_N_VALUE)->SetWindowText(buf);
261 sprintf(buf, "%.3f", m_lfPmeridian);
262 GetDlgItem(IDC_DEFL_PMERIDIAN)->SetWindowText(buf);
263 sprintf(buf, "%.3f", m_lfPvertical);
264 GetDlgItem(IDC_DEFL_PVERTICAL)->SetWindowText(buf);
265
266 // convert to decimal degrees
267 if (m_iDMS == DMS)
268 {
269 m_lfLatitude = DmstoDeg(m_lfLatitude);
270 m_lfLongitude = DmstoDeg(m_lfLongitude);
271 }
272
273 // Astronomic longitude
274 m_lfLongitude += m_lfPvertical / 3600. / cos(Radians(m_lfLatitude));
275
276 // Astronomic latitude correction
277 m_lfLatitude += m_lfPmeridian / 3600.;
278
279 if (m_iDMS == DMS)
280 {
281 m_lfLatitude = DegtoDms(m_lfLatitude);
282 m_lfLongitude = DegtoDms(m_lfLongitude);
283 }
284
285 sprintf(buf, "%.9f", m_lfLatitude);
286 GetDlgItem(IDC_LAT_OUT)->SetWindowText(AddSymbology(buf, true, 1));
287
288 sprintf(buf, "%.9f", m_lfLongitude);
289 GetDlgItem(IDC_LONG_OUT)->SetWindowText(AddSymbology(buf, true, 2));
290
291 // Height
292 if (m_iEllipsoidtoOrthoRadio == 0)
293 m_lfHeight2 = m_lfHeight - m_lfNvalue; // H = h - N
294 else
295 m_lfHeight2 = m_lfHeight + m_lfNvalue; // h = H + N
296
297 sprintf(buf, "%.3f", m_lfHeight2);
298 GetDlgItem(IDC_HT2)->SetWindowText(buf);
299
300 m_bInterpolateSuccess = true;
301
302 return 0;
303}
