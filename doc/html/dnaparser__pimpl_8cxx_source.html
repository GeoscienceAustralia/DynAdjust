742 }

748 switch (_dnaCurrentMsr->GetTypeC())

763 }

767 // Test for non-ignored measurements
768 for_each(_dnaCurrentMsr->GetDirections_ptr()->begin(),
769 _dnaCurrentMsr->GetDirections_ptr()->end(),
770 [this, &found](CDnaDirection& d) {
771 if (d.NotIgnored())
772 found++;
773 });
774
775 if (found == 0 && _dnaCurrentMsr->NotIgnored())
776 {
777 std::stringstream ss, ss2;
778 ss <<
779 "<Directions>...</Directions>', total of " << total << " element(s)" << std::endl <<
780 " - There aren't any non-ignored directions in the set. ";
781 ss2 << " ~ <Total>" << total << "</Total>";
782 throw ::xsd::cxx::parser::expected_element< char >(
783 ss.str(), ss2.str());
784 }

787
788 // Okay, capture all the elements for this direction set
789 for_each(_dnaCurrentMsr->GetDirections_ptr()->begin(),
790 _dnaCurrentMsr->GetDirections_ptr()->end(),
791 [this](CDnaDirection& d) {
792 d.SetType(_dnaCurrentMsr->GetType());
793 d.SetFirst(_dnaCurrentMsr->GetFirst());

796 // total will be the number of target directions

800 break;

802 case 'G':
803 case 'X':

805 total = _dnaCurrentMsr->GetTotal();

816 }

846 }

850 _dnaCurrentMsr->GetPoints_ptr()->end(),
