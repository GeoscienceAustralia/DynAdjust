31// open the asl file. the asl file is required to ensure invalid stations are
32// excluded from the adjustment. invalid stations are those with no measurements.
33UINT32 dna_io_asl::load_asl_file(const std::string& asl_filename, vASL* vbinary_asl, vUINT32* vfree_stn)
34{
35 std::ifstream asl_file;
36 std::stringstream ss;
37 ss << "load_asl_file(): An error was encountered when opening " << asl_filename << "." << std::endl;
38
39 try {
40 // open stations asl file. Throws runtime_error on failure.
41 file_opener(asl_file, asl_filename, std::ios::in | std::ios::binary, binary, true);
42 }
43 catch (const std::runtime_error& e) {
44 ss << e.what();
45 throw boost::enable_current_exception(std::runtime_error(ss.str()));
46 }
47 catch (...) {
48 throw boost::enable_current_exception(std::runtime_error(ss.str()));
49 }
50
51 ss.str("");
52 ss << "load_asl_file(): An error was encountered when reading from " << asl_filename << "." << std::endl;
53
54 UINT32 stnCount;
55
56 try {
57 // read the file information
58 readFileInfo(asl_file);
59
60 // Get number of records and resize AML vector
61 asl_file.read(reinterpret_cast<char *>(&stnCount), sizeof(UINT32));
62
63 // initialise free station list
64 initialiseIncrementingIntegerVector(vfree_stn, stnCount);
65
66 // initialise assocaiated station list
67 vbinary_asl->clear();
68 vbinary_asl->resize(stnCount);
69
70 _it_vasl _it_asl;
71
72 for (_it_asl=vbinary_asl->begin();
73 _it_asl!=vbinary_asl->end();
74 ++_it_asl)
75 {
76 asl_file >> &(*_it_asl);
77 }
78
79 }
80 catch (const std::ios_base::failure& f) {
81 ss << f.what();
82 throw boost::enable_current_exception(std::runtime_error(ss.str()));
83 }
84 catch (const std::runtime_error& e) {
85 ss << e.what();
86 throw boost::enable_current_exception(std::runtime_error(ss.str()));
87 }
88 catch (...) {
89 throw boost::enable_current_exception(std::runtime_error(ss.str()));
90 }
91
92 asl_file.close();
93
94 return stnCount;
95}

97void dna_io_asl::write_asl_file(const std::string& asl_filename, pvASLPtr vbinary_asl)
98{
99 std::ofstream asl_file;
100 std::stringstream ss;
101 ss << "write_asl_file(): An error was encountered when opening " << asl_filename << "." << std::endl;
102
103 try {
104 // create binary asl file. Throws runtime_error on failure.
105 file_opener(asl_file, asl_filename,
106 std::ios::out | std::ios::binary, binary);
107 }
108 catch (const std::runtime_error& e) {
109 ss << e.what();
110 throw boost::enable_current_exception(std::runtime_error(ss.str()));
111 }
112 catch (...) {
113 throw boost::enable_current_exception(std::runtime_error(ss.str()));
114 }
115
116 ss.str("");
117 ss << "write_asl_file(): An error was encountered when writing to " << asl_filename << "." << std::endl;
118
119 // Calculate number of station records and write to binary file
120 UINT32 aslCount = static_cast<UINT32>(vbinary_asl->size());
121 vASLPtr::const_iterator _it_asl(vbinary_asl->begin());
122
123 try {
124 // write the file information
125 writeFileInfo(asl_file);
126
127 // write the data
128 asl_file.write(reinterpret_cast<char *>(&aslCount), sizeof(UINT32));
129 for (_it_asl=vbinary_asl->begin(); _it_asl!=vbinary_asl->end(); ++_it_asl)
130 if (_it_asl->get())
131 asl_file << _it_asl->get(); // see dnastation.hpp (region CAStationList stream handlers)
132 }
133 catch (const std::ios_base::failure& f) {
134 ss << f.what();
135 throw boost::enable_current_exception(std::runtime_error(ss.str()));
136 }
137 catch (const std::runtime_error& e) {
138 ss << e.what();
139 throw boost::enable_current_exception(std::runtime_error(ss.str()));
140 }
141 catch (...) {
142 throw boost::enable_current_exception(std::runtime_error(ss.str()));
143 }
144
145 asl_file.close();
146}

148// prints ASL entries sorted according to measurement count, which is useful when determining
149// the starting station to begin network segmentation.
150void dna_io_asl::write_asl_file_txt(const std::string& asl_filename, pvASLPtr vbinary_asl, vdnaStnPtr* vStations)
151{
152 std::ofstream asl_file;
153 std::stringstream ss;
154 ss << "write_asl_file_txt(): An error was encountered when opening " << asl_filename << "." << std::endl;
155
156 try {
157 // create binary asl file. Throws runtime_error on failure.
158 file_opener(asl_file, asl_filename);
159 }
160 catch (const std::runtime_error& e) {
161 ss << e.what();
162 throw boost::enable_current_exception(std::runtime_error(ss.str()));
163 }
164 catch (...) {
165 throw boost::enable_current_exception(std::runtime_error(ss.str()));
166 }
167
168 ss.str("");
169 ss << "write_asl_file_txt(): An error was encountered when writing to " << asl_filename << "." << std::endl;
170
171 // Calculate number of station records and write to binary file
172 size_t aslCount = vbinary_asl->size();
173
174 // Pump to text file
175 std::stringstream ss_asl;
176 ss_asl << aslCount << " stations";
177 asl_file << std::left << std::setw(STATION) << ss_asl.str();
178 asl_file << std::setw(HEADER_20) << std::right << "No. connected msrs";
179 asl_file << std::setw(STATION) << std::right << "AML index";
180 asl_file << std::setw(STATION) << std::right << "Unused?" << std::endl;
181
182 // Create an incrementing list and sort on number of measurements to each station
183 vUINT32 aslPtrs(vbinary_asl->size());
184 initialiseIncrementingIntegerVector<UINT32>(aslPtrs, static_cast<UINT32>(vbinary_asl->size()));
185
186 // Sort on measurement count
187 CompareMeasCount2<ASLPtr, UINT32> msrcountCompareFunc(vbinary_asl);
188 std::sort(aslPtrs.begin(), aslPtrs.end(), msrcountCompareFunc);
189
190 vbinary_asl->at(0).get()->GetAMLStnIndex();
191
192 try {
193
194 it_vUINT32 _it_asl;
195 vASLPtr::iterator _it_pasl;
196
197 for (_it_asl=aslPtrs.begin(); _it_asl!=aslPtrs.end(); ++_it_asl)
198 {
199 if (!vbinary_asl->at(*_it_asl)) // unused station
200 continue;
201
202 _it_pasl = vbinary_asl->begin() + *_it_asl;
203
204 // Name and measurement count
205 asl_file << std::setw(STATION) << std::left <<
206 vStations->at(*_it_asl)->GetName() <<
207 std::setw(HEADER_20) << std::right << _it_pasl->get()->GetAssocMsrCount();
208 // Aml station index
209 if (_it_pasl->get()->GetAssocMsrCount() == 0)
210 asl_file << std::setw(STATION) << "-";
211 else
212 asl_file << std::setw(STATION) << std::right << _it_pasl->get()->GetAMLStnIndex();
213 // Valid station?
214 asl_file << std::setw(STATION) << std::right << (_it_pasl->get()->IsValid() ? " " : "*") << std::endl;
215 }
216 }
217 catch (const std::ios_base::failure& f) {
218 ss << f.what();
219 throw boost::enable_current_exception(std::runtime_error(ss.str()));
220 }
221 catch (const std::runtime_error& e) {
222 ss << e.what();
223 throw boost::enable_current_exception(std::runtime_error(ss.str()));
224 }
225 catch (...) {
226 throw boost::enable_current_exception(std::runtime_error(ss.str()));
227 }
228
229 asl_file.close();
230}
