48// Custom type to manage type b uncertainties
49template <class T1 = UINT32, class T2 = double, class T3 = matrix_2d>
50struct type_b_uncertainty_t
51{
52 T1 station_id;
53 T3 type_b;
54
55 type_b_uncertainty_t()
56 : station_id(0) {
57 type_b.redim(3, 3);
58 }
59
60 void set_station_id(const T1& stn=0) { station_id =stn; }
61
62 void set_typeb_values_3d(const T2& e, const T2& n, const T2& u) {
63 type_b.put(0, 0, e);
64 type_b.put(1, 1, n);
65 type_b.put(2, 2, u);
66 }
67
68 bool operator== (const type_b_uncertainty_t<T1, T2, T3>& rhs) {
69 return equal(station_id, rhs.station_id);
70 }
71
72 bool operator< (const type_b_uncertainty_t<T1, T2, T3>& rhs) const
73 {
74 return station_id < rhs.station_id;
75 }
76};

84template <class T1=UINT32, class T2= type_b_uncertainty>
85class CompareTypeBStationID
86{
87public:
88 bool operator()(const T2& lhs, const T2& rhs) const {
89 return keyLess(lhs.second, rhs.second);
90 }
91 bool operator()(const T2& lhs, const T1& rhs) {
92 return keyLess(lhs.station_id, rhs);
93 }
94 bool operator()(const T1& lhs, const T2& rhs) {
95 return keyLess(lhs, rhs.station_id);
96 }
97private:
98 bool keyLess(const T1& s1, const T1& s2) const {
99 return s1 < s2;
100 }
101};
