84void dna_io_base::writeFileMetadata(std::ofstream& file_stream, binary_file_meta_t& file_meta)
85{
86 // Write the metadata
87 file_stream.write(reinterpret_cast<char *>(&file_meta.binCount), sizeof(UINT32));
88 file_stream.write(reinterpret_cast<char *>(&file_meta.reduced), sizeof(bool));
89 file_stream.write(reinterpret_cast<char *>(file_meta.modifiedBy), MOD_NAME_WIDTH);
90
91 // Write the epsg code and epoch
92 file_stream.write(reinterpret_cast<char *>(file_meta.epsgCode), STN_EPSG_WIDTH);
93 file_stream.write(reinterpret_cast<char *>(file_meta.epoch), STN_EPOCH_WIDTH);
94
95 file_stream.write(reinterpret_cast<char*>(&file_meta.reftran), sizeof(bool));
96 file_stream.write(reinterpret_cast<char*>(&file_meta.geoid), sizeof(bool));
97
98 // Write file count and file meta
99 file_stream.write(reinterpret_cast<char *>(&file_meta.inputFileCount), sizeof(UINT16));
100 for (UINT16 i(0); i<file_meta.inputFileCount; ++i)
101 {
102 file_stream.write(reinterpret_cast<char *>(file_meta.inputFileMeta[i].filename), FILE_NAME_WIDTH);
103 file_stream.write(reinterpret_cast<char *>(file_meta.inputFileMeta[i].epsgCode), STN_EPSG_WIDTH);
104 file_stream.write(reinterpret_cast<char *>(file_meta.inputFileMeta[i].epoch), STN_EPOCH_WIDTH);
105 file_stream.write(reinterpret_cast<char *>(&file_meta.inputFileMeta[i].filetype), sizeof(UINT16));
106 file_stream.write(reinterpret_cast<char *>(&file_meta.inputFileMeta[i].datatype), sizeof(UINT16));
107 }
108}

173void dna_io_base::writeDate(std::ofstream& file_stream)
174{
175 char dateField[identifier_field_width+1];
176
177 // Form date string
178 boost::gregorian::date today(boost::gregorian::day_clock::local_day());
179 std::stringstream date_string;
180 date_string << std::right << to_iso_extended_string(today);
181 m_strDate = date_string.str();
182
183 // write creation date field name
184 dateField[identifier_field_width] = '\0';
185 file_stream.write(const_cast<char *>(create_date_header), identifier_field_width);
186
187 // write creation date, with safeguard against overwriting by substr
188 snprintf(dateField, sizeof(dateField), "%*s", identifier_field_width, m_strDate.substr(0, identifier_field_width).c_str());
189 file_stream.write(reinterpret_cast<char *>(dateField), identifier_field_width);
190
