45void dna_segment::LoadNetFile()
46{
47 // Get stations listed in net file
48 std::ifstream net_file;
49 try {
50 // Load net file. Throws runtime_error on failure.
51 file_opener(net_file, projectSettings_.s.net_file,
52 std::ios::in, ascii, true);
53 }
54 catch (const std::runtime_error& e) {
55 SignalExceptionSerialise(e.what(), 0, NULL);
56 }
57
58 std::string station;
59 char line[PROGRAM_OPTIONS_LINE_LENGTH];
60
61 // read header lines
62 net_file.ignore(PRINT_LINE_LENGTH, '\n'); // -------------------------------------------
63 net_file.getline(line, PROGRAM_OPTIONS_LINE_LENGTH); // DYNADJUST BLOCK 1 STATIONS FILE
64
65 vinitialStns_.clear();
66
67 try {
68
69 while (!net_file.eof())
70 {
71 net_file.getline(line, PROGRAM_OPTIONS_LINE_LENGTH); // Stations
72 station = line;
73 if ((station = trimstr(station)).empty())
74 continue;
75 // add this station to the list
76 vinitialStns_.push_back(station);
77 }
78 }
79 catch (const std::ios_base::failure& f) {
80
81 if (!net_file.eof())
82 {
83 std::stringstream ss;
84 ss << "ParseStartingStations(): An error was encountered when reading " << projectSettings_.s.net_file << "." << std::endl << f.what();
85 SignalExceptionSerialise(ss.str(), 0, "i", &net_file);
86 }
87 }
88
89 net_file.close();
90}

92void dna_segment::ParseStartingStations()
93{
94 // Get stations listed in net file (if it exists)
95 // Rememeber, calling app should make sure that
96 // projectSettings_.s.net_file contains a valid path if
97 // the user wants to use a net file.
98 if (boost::filesystem::exists(projectSettings_.s.net_file.c_str()))
99 LoadNetFile();
100
101 // OK, now get the additionalstations on the command line
102 if (projectSettings_.s.seg_starting_stns.empty())
103 return;
104
105 vstring cmd_line_stations;
106
107 try {
108 // Extract stations from comma delimited string
109 SplitDelimitedString<std::string>(
110 projectSettings_.s.seg_starting_stns, // the comma delimited string
111 std::string(","), // the delimiter
112 &cmd_line_stations); // the respective values
113 }
114 catch (...) {
115 return;
116 }
117
118 if (!cmd_line_stations.empty())
119 vinitialStns_.insert(vinitialStns_.end(), cmd_line_stations.begin(), cmd_line_stations.end());
120
121 if (!vinitialStns_.empty())
122 strip_duplicates(vinitialStns_);
123}

171{
172 isProcessing_ = true;
173
174 output_folder_ = p->g.output_folder;
175
176 if ((debug_level_ = p->g.verbose) > 2)
177 {
178 std::string s(p->s.seg_file + ".trace");
179 try {
180 // Create trace file. Throws runtime_error on failure.
181 file_opener(trace_file, s);
182 print_file_header(trace_file, "DYNADJUST SEGMENTATION TRACE FILE");
183 trace_file << OUTPUTLINE << std::endl << std::endl;
184 }
185 catch (const std::runtime_error& e) {
186 SignalExceptionSerialise(e.what(), 0, NULL);
187 }
188 }
189
190 if ((debug_level_ = p->g.verbose) > 1)
191 {
192 std::string s(p->s.seg_file + ".debug");
193
194 try {
195 // Create debug file. Throws runtime_error on failure.
196 file_opener(debug_file, s);
197 }
198 catch (const std::ios_base::failure& f) {
199 SignalExceptionSerialise(f.what(), 0, NULL);
200 }
201 }
202
203 segmentStatus_ = SEGMENT_SUCCESS;
204 std::ostringstream ss;
205
206 if (debug_level_ > 2)
207 WriteFreeStnListSortedbyASLMsrCount();
208
209 currentBlock_ = 1;
210
211 if (debug_level_ > 2)
212 trace_file << "Block " << currentBlock_ << "..." << std::endl;
213
214 boost::timer::cpu_timer time;
215
216 v_ContiguousNetList_.clear();
217 v_ContiguousNetList_.push_back(currentNetwork_ = 0);
218
219 if (bstBinaryRecords_.empty())
220 SignalExceptionSerialise("SegmentNetwork(): the binary stations file has not been loaded into memory yet.", 0, NULL);
221 if (bmsBinaryRecords_.empty())
222 SignalExceptionSerialise("SegmentNetwork(): the binary measurements file has not been loaded into memory yet.", 0, NULL);
223 if (vAssocStnList_.empty())
224 SignalExceptionSerialise("SegmentNetwork(): the Associated Station List (ASL) has not been loaded into memory yet.", 0, NULL);
225 if (vAssocFreeMsrList_.empty())
226 SignalExceptionSerialise("SegmentNetwork(): the Associated Measurements List (AML) has not been loaded into memory yet.", 0, NULL);
227 if (stnsMap_.empty())
228 SignalExceptionSerialise("SegmentNetwork(): the station map has not been loaded into memory yet.", 0, NULL);
229
230 // The inner stations for the first block are created from segmentCriteria._initialStns
231 // Junction stations are retrieved from measurements connected to the inner stations
232 BuildFirstBlock();
233
234 while (!vfreeStnList_.empty())
235 {
236 isProcessing_ = true;
237
238 currentBlock_++;
239
240 if (debug_level_ > 2)
241 trace_file << "Block " << currentBlock_ << "..." << std::endl;
242
243 v_ContiguousNetList_.push_back(currentNetwork_);
244
245 // The inner stations for the next block are the junction stations from the previous block
246 // The junction stations are retrieved from measurements connected to the inner stations
247 // BuildNextBlock applies the min and max station constraints using segmentCriteria
248 BuildNextBlock();
249
250 if (vfreeStnList_.empty())
251 break;
252 }
253
254 boost::posix_time::milliseconds elapsed_time(boost::posix_time::milliseconds(0));
255 if (debug_level_ > 1)
256 elapsed_time = boost::posix_time::milliseconds(time.elapsed().wall/MILLI_TO_NANO);
257
258 isProcessing_ = false;
259
260 CalculateAverageBlockSize();
261
262 char valid_stations(0);
263 char valid_measurements_not_included(0);
264 if (vfreeStnList_.size())
265 {
266 valid_stations = 1;
267 ss.str("");
268 ss << std::endl << "- Warning: The following stations were not used:" << std::endl;
269 ss << " ";
270 it_vUINT32_const _it_freestn(vfreeStnList_.begin());
271 for (; _it_freestn!=vfreeStnList_.end(); ++_it_freestn)
272 ss << bstBinaryRecords_.at(*_it_freestn).stationName << " ";
273 ss << std::endl;
274 ss << "- Possible reasons why these stations were not used include:" << std::endl;
275 ss << " - No measurements to these stations were found," << std::endl;
276 ss << " - The network is made up of non-contiguous blocks," << std::endl;
277 ss << " - There is a bug in this program." << std::endl;
278
279 if (debug_level_ > 1)
280 debug_file << ss.str();
281 }
282
283 if (debug_level_ > 1)
284 {
285 debug_file << formatedElapsedTime<std::string>(&elapsed_time, "+ Network segmentation took ") <<
286 " The network is now ready for adjustment." << std::endl;
287 debug_file.close();
288 }
289
290 isProcessing_ = false;
291
292 if (valid_measurements_not_included || valid_stations)
293 return (segmentStatus_ = SEGMENT_BLOCK_ERROR);
294
295 return (segmentStatus_ = SEGMENT_SUCCESS);
296}

299void dna_segment::CalculateAverageBlockSize()
300{
301 vUINT32 blockSizes;
302 blockSizes.resize(vISL_.size());
303
304 it_vUINT32 _it_count;
305 it_vvUINT32 _it_isl, _it_jsl;
306 for (_it_count=blockSizes.begin(), _it_isl=vISL_.begin(), _it_jsl=vJSL_.begin();
307 _it_isl!=vISL_.end();
308 ++_it_count, ++_it_isl, ++_it_jsl)
309 *_it_count = static_cast<UINT32>(_it_isl->size() + _it_jsl->size());
310
311 averageBlockSize_ = average<double, UINT32, it_vUINT32>(blockSizes.begin(), blockSizes.end(), stationSolutionCount_);
312
313 maxBlockSize_ = *(max_element(blockSizes.begin(), blockSizes.end()));
314 minBlockSize_ = *(min_element(blockSizes.begin(), blockSizes.end()));
315}

334// throws NetSegmentException on failure
335void dna_segment::BuildFirstBlock()
336{
337 vCurrJunctStnList_.clear();
338 vCurrInnerStnList_.clear();
339 vCurrMeasurementList_.clear();
340
341 if (vinitialStns_.empty())
342 // no station specified? then pick the first one on the free list
343 // Remember, vfreeStnList_ is not sorted alphabetically, but according
344 // to the number of measurements connected to each station
345 vinitialStns_.push_back(bstBinaryRecords_.at(vfreeStnList_.at(0)).stationName);
346 else
347 RemoveDuplicateStations(&vinitialStns_); // remove duplicates
348
349 // First pass to validate stations
350 VerifyStationsandBuildBlock(true);
351
352 // Second pass to build the block
353 VerifyStationsandBuildBlock();
354
355 FinaliseBlock();
356}

358void dna_segment::VerifyStationsandBuildBlock(bool validationOnly)
359{
360#ifdef _MSDEBUG
361 UINT32 stn_index;
362#endif
363
364 it_vUINT32 _it_freeisl;
365 _it_vstr_const _it_name(vinitialStns_.begin());
366 it_pair_string_vUINT32 it_stnmap_range;
367 v_string_uint32_pair::iterator _it_stnmap(stnsMap_.begin());
368
369 // For each station name specified as the initial stations
370 for (_it_name=vinitialStns_.begin(); _it_name!=vinitialStns_.end(); ++_it_name)
371 {
372 // Retrieve the ASL index
373 it_stnmap_range = equal_range(stnsMap_.begin(), stnsMap_.end(), *_it_name, StationNameIDCompareName());
374
375 if (it_stnmap_range.first == it_stnmap_range.second)
376 {
377 // If this point is reached, _it_stnmap->second is not a known network station
378 std::stringstream ss;
379 ss << "VerifyStationsandBuildBlock(): " << *_it_name << " is not in the list of network stations.";
380 SignalExceptionSerialise(ss.str(), 0, NULL);
381 }
382
383 _it_stnmap = it_stnmap_range.first;
384
385#ifdef _MSDEBUG
386 stn_index = _it_stnmap->second;
387#endif
388 // add this station to inner station list if it is on the free stations list only.
389 // vfreeStnList_ is sorted according to the number of measurements connected to each station
390 // hence, a binary_search (or lower_bound or similar) cannot be used here
391
392 // Check if this station is available before trying to find it. Why?
393 // find_if over a large vector is not fast like binary_search
394 if (vfreeStnAvailability_.at(_it_stnmap->second).isfree())
395 {
396 // Ok, find it and remove it
397 //if ((_it_freeisl = find_if(vfreeStnList_.begin(), vfreeStnList_.end(),
398 // bind1st(std::equal_to<UINT32>(), _it_stnmap->second))) != vfreeStnList_.end())
399 if ((_it_freeisl = find_if(vfreeStnList_.begin(), vfreeStnList_.end(),
400 [&_it_stnmap](const UINT32& freeStn) {
401 return freeStn == _it_stnmap->second; } )) != vfreeStnList_.end())
402 {
403 if (!validationOnly)
404 MoveFreeStnToInnerList(_it_freeisl, _it_stnmap->second);
405 }
406 }
407 else
408 {
409 if (validationOnly)
410 {
411 // If this point is reached, _it_stnmap->second is a known network station but is
412 // invalid or has no measurements connected to it.
413 std::stringstream ss;
414 ss << "VerifyStationsandBuildBlock(): " << bstBinaryRecords_.at(_it_stnmap->second).stationName << " does not have any measurements connected to it.";
415 SignalExceptionSerialise(ss.str(), 0, NULL);
416 }
417 }
418
419 if (!validationOnly)
420 // Retrieve JSL (all stations connected to this station) and
421 // add AML indices to CML for all measurements connected to this station
422 GetInnerMeasurements(_it_stnmap->second);
423 }

426UINT32 dna_segment::SelectInner()
427{
428 // Sort the list of junction stations by the number of measurements
429 // connected to them, and begin with the station that has the lowest
430 // number of measurements
431 SortbyMeasurementCount(&vCurrJunctStnList_);
432
433 it_vUINT32 it_currjsl(vCurrJunctStnList_.begin());
434 UINT32 stn_index = *it_currjsl;
435
436 if (debug_level_ > 2)
437 trace_file << " + Inner '" << bstBinaryRecords_.at(stn_index).stationName << "'" << std::endl;
438
439#ifdef _MSDEBUG
440 std::string station_name = bstBinaryRecords_.at(stn_index).stationName;
441#endif
442
443 // add this junction station to inner station list
444 MoveStation(vCurrJunctStnList_, it_currjsl, vCurrInnerStnList_, stn_index);
445
446 return stn_index;
447}

504// throws NetSegmentException on failure
505void dna_segment::BuildNextBlock()
506{
507 // vCurrJunctStnList_ carries the previous block's junction stations. Thus,
508 // don't clear it as these stations will become the inners for the next block
509 vCurrInnerStnList_.clear();
510 vCurrMeasurementList_.clear();
511
512#ifdef _MSDEBUG
513 std::string station_name;
514#endif
515
516 // Select a new junction station if there are none on the JSL
517 // that have free measurements left
518 if (vCurrJunctStnList_.empty() && !vfreeStnList_.empty())
519 {
520 if (!projectSettings_.s.force_contiguous_blocks)
521 v_ContiguousNetList_.back() = ++currentNetwork_;
522
523 if (debug_level_ > 1)
524 {
525 debug_file << "+ Non-contiguous block found... creating a new block using " << bstBinaryRecords_.at(vfreeStnList_.at(0)).stationName << std::endl;
526 if (debug_level_ > 2)
527 trace_file << " + Non-contiguous block found... creating a new block using " << bstBinaryRecords_.at(vfreeStnList_.at(0)).stationName << std::endl;
528 }
529
530 // Select a new junction from the free station list
531 SelectJunction();
532 }
533
534 if (vCurrJunctStnList_.empty())
535 {
536 coutSummary();
537 SignalExceptionSerialise("BuildNextBlock(): An invalid junction list has been created. This is most likely a bug.", 0, NULL);
538 }
539
540 UINT32 stn_index;
541 UINT32 currentTotalSize;
542 bool block_threshold_reached = false;
543
544 // Until the threshold is reached...
545 // - Go through the list of junction stations (copied from the previous block) and
546 // make them inner stations if there are new stations connected to them
547 // - If the list is exhausted, add new junctions
548 while (!block_threshold_reached)
549 {
550 // Attempt to add non-contiguous blocks to this block if the
551 // station limit hasn't been reached
552 if (vfreeStnList_.empty() /*|| vfreeMsrList_.empty()*/)
553 break;
554
555 // Is the junction list empty? force_contiguous_blocks determines what to
556 // do in this instance.
557 if (projectSettings_.s.force_contiguous_blocks)
558 {
559 // Add non-contiguous blocks to this block if the
560 // station limit hasn't been reached
561 if (vCurrJunctStnList_.empty())
562 SelectJunction();
563 }
564 else if (vCurrJunctStnList_.empty())
565 break;
566
567 // Get next station from current junction list
568 stn_index = SelectInner();
569
570 // Retrieve JSL (all stations connected to this station) and
571 // add AML indices to CML for all measurements connected to this station
572 GetInnerMeasurements(stn_index);
573
574 currentTotalSize = static_cast<UINT32>(vCurrInnerStnList_.size() + vCurrJunctStnList_.size());
575
576 // Has the station count threshold been exceeded?
577 //if (currentTotalSize >= seg_max_total_stations_)
578 if (currentTotalSize >= projectSettings_.s.max_total_stations)
579 {
580 if (vCurrInnerStnList_.size() < projectSettings_.s.min_inner_stations)
581 continue;
582 block_threshold_reached = true;
583 break;
584 }
585 }
586
587 if (debug_level_ > 2)
588 {
589 if (block_threshold_reached)
590 trace_file << " + Block size threshold exceeded... finishing block." << std::endl;
591 else
592 trace_file << " + Block " << currentBlock_ << " complete." << std::endl;
593 }
594
595 FinaliseBlock();
596}

598void dna_segment::FinaliseBlock()
599{

602
603 // Sort lists
604 std::sort(vCurrInnerStnList_.begin(), vCurrInnerStnList_.end());
605 std::sort(vCurrJunctStnList_.begin(), vCurrJunctStnList_.end());
606 strip_duplicates(vCurrMeasurementList_); // remove duplicates and sort
607
608 vISL_.push_back(vCurrInnerStnList_);
609 vJSL_.push_back(vCurrJunctStnList_);
610 vCML_.push_back(vCurrMeasurementList_);
611
612 if (debug_level_ > 1)

615
616 if (debug_level_ > 2)
617 {

685// Name: GetInnerMeasurements
686// Purpose: Gets all measurements tied to the subject station (innerStation) and stations tied to
687// those measurements
688// Called by: BuildFirstBlock(), BuildNextBlock()
689// Calls: AddtoCurrentMsrList(), AddtoJunctionStnList()
690void dna_segment::GetInnerMeasurements(const UINT32& innerStation)
691{
692 if (debug_level_ > 2)
693 trace_file << " + GetInnerMeasurements()" << std::endl;
694
695#ifdef _MSDEBUG
696 std::string from, to, to2;
697#endif
698
699 measurement_t measRecord;
700 vUINT32 msrStations;
701
