39// Custom type to manage euler plate motion model parameters
40template <class T1 = std::string, class T2 = double>
41struct plate_motion_euler_t
42{
43 T1 plate_name;
44 T1 pole_param_author;
45 T2 pole_latitude;
46 T2 pole_longitude;
47 T2 pole_rotation_rate;
48
49 plate_motion_euler_t()
50 : plate_name(""), pole_param_author("")
51 , pole_latitude(T2()), pole_longitude(T2()), pole_rotation_rate(T2()) {}
52 plate_motion_euler_t(const T1& name, const T1& author, const T2& lat, const T2& lon, const T2& rot)
53 : plate_name(name), pole_param_author(author)
54 , pole_latitude(lat), pole_longitude(lon), pole_rotation_rate(rot) {}
55 plate_motion_euler_t(const plate_motion_euler_t <T1, T2> &p)
56 : plate_name(p.plate_name), pole_param_author(p.pole_param_author)
57 , pole_latitude(p.pole_latitude), pole_longitude(p.pole_longitude)
58 , pole_rotation_rate(p.pole_rotation_rate) { }
59
60 bool operator== (const plate_motion_euler_t<T1, T2>& rhs) {
61 return equals(plate_name, rhs.plate_name);
62 }
63
64 bool operator< (const plate_motion_euler_t<T1, T2>& rhs) const
65 {
66 if (plate_name == rhs.plate_name) {
67 if (pole_latitude == rhs.pole_latitude) // don't think this is needed
68 return pole_longitude < rhs.pole_longitude;
69 else
70 return pole_latitude < rhs.pole_latitude;
71 }
72 else
73 return plate_name < rhs.plate_name;
74 }
75
76};

79typedef std::vector<plate_motion_euler> v_plate_motion_eulers;
80typedef v_plate_motion_eulers::iterator it_plate_motion_euler;

108

117 }
118 else
119 return x_rotation < rhs.x_rotation;
120 }
121 else
122 return plate_name < rhs.plate_name;
123 }

125};
